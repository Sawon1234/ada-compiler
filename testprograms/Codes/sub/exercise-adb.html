The Ada Program: exercise.adb

  1 with Ada.Text_IO, Ada.Integer_Text_IO; use Ada;
  2 procedure Exercise is
  3    procedure Put (X:String; N:Integer) is
  4    begin
  5       Text_IO.Put (X);
  6       Integer_Text_IO.Put (Item=>N, Width=>0);
  7       Text_IO.New_Line;
  8    end Put;
  9    procedure One (G, H: in out Integer) is
 10    begin
 11       G := 5;
 12       H := 3 * G;
 13       G := G + 1;
 14       Put ("G=", G);  Put ("H=", H);
 15    end One;
 16    J, K: Integer;
 17 begin
 18    J := 7;   K := 1;
 19    One (J, K);
 20    Put ("J=", J);   Put ("K=", K);
 21    J := 10;   K := 15;
 22    One (K, J);
 23    Put ("J=", J);   Put ("K=", K);
 24 end Exercise;
