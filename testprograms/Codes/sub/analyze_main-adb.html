  3 procedure Analyze_Main is
  4 
  5    type Data_List_Type is array (Integer range <>) of Float;
  6 
  7    procedure Analyze
  8       (Data_Points: in Data_List_Type; Mean, Variance: out Float) is
  9 
 10       Number_Of_Points   : constant Float := Float (Data_Points'Length);
 11       Sum, Sum_Of_Squares: Float := 0.0;
 12 
 13    begin
 14 
 15       for I in Data_Points'Range loop
 16          Sum := Sum + Data_Points(I);
 17          Sum_Of_Squares := Sum_Of_Squares + Data_Points(I)**2;
 18       end loop;
 19 
 20       Mean := Sum / Number_Of_Points;
 21       Variance := (Sum_Of_Squares - Number_Of_Points * Mean ** 2) /
 22         (Number_Of_Points - 1.0);
 23 
 24    end Analyze;
