  1 -- repeat_main.adb: illustrate default parameters

  3 with Ada.Text_IO;
  4 use Ada;
  5 
  6 procedure Repeat_Main is
  7 
  8    procedure Repeat_Put (C: in Character; By: in Natural:=1) is
  9    begin
 10       for I in 1..By loop
 11          Text_IO.Put (C);
 12       end loop;
 13    end Repeat_Put;
 14 
 15 begin
 16 
 17    Repeat_Put ('!', 10);
 18    Repeat_Put ('=', 1);
 19    Repeat_Put ('=');
 20    Repeat_Put (C=>'_');
 21    Repeat_Put ('X', By=>0);
 22    Repeat_Put (C=>'#');
 23    Text_IO.New_Line;
 24    Repeat_Put (C=>'*', By=>60);   Text_IO.New_Line;
 25    Repeat_Put (By=>60, C=>'*');   Text_IO.New_Line;
 26 
 27 end Repeat_Main;
