The Ada Program: demo.adb

  1 -- demo.adb: illustrate parameter passing
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Demo is
  7    procedure Nonsense (A: in Integer;
  8                        B: in out Integer;
  9                        C: out Integer) is
 10    begin
 11       B := B+A;
 12       C := 0;
 13    end Nonsense;
 14 
 15    procedure Print (A,B,C: Integer) is
 16    begin
 17       Integer_Text_IO.Put (A);
 18       Integer_Text_IO.Put (B);
 19       Integer_Text_IO.Put (C);
 20       Text_IO.New_Line;
 21    end Print;
 22 
 23    X,Y,Z : Integer;
 24 
 25 begin
 26    X := 1; Y:= 5; Z := 10;
 27    Print (X,Y,Z);
 28    Nonsense (X,Y,Z);
 29    Print (X,Y,Z);
 30 end Demo;
