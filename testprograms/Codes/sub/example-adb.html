  1 -- example.adb:  Ada program illustrating subprocedures
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Example is
  7 
  8     procedure Greeting is
  9     begin
 10         Text_IO.New_Line (2);
 11         Text_IO.Put_Line ("* * * * * * * * * * * * * * * * * * *");
 12         Text_IO.Put_Line ("* Hello, please enter two integers. *");
 13         Text_IO.Put_Line ("* * * * * * * * * * * * * * * * * * *");
 14         Text_IO.New_Line;
 15     end Greeting;
 16 
 17     procedure Put_Line (Label: in String; N: in Integer) is
 18     begin
 19         Text_IO.Put (Label);
 20         Integer_Text_IO.Put (N, Width=>0);
 21         Text_IO.New_Line;
 22     end Put_Line;
 23 
 24     procedure Get_2 (X,Y: out Integer) is
 25     begin
 26        Integer_Text_IO.Get (X);
 27        Integer_Text_IO.Get (Y);
 28        Text_IO.Skip_Line;
 29     end Get_2;
 30 
 31     A,B: Integer;
 32 
 33 begin
 34 
 35     Greeting;
 36     Get_2 (A, B);
 37     Text_IO.New_Line;
 38     Text_IO.Put_Line ("Thank you.");
 39     Put_Line ("The first integer is ", A);
 40     Put_Line ("The second integer is ", B);
 41     Text_IO.Put_Line ("The end.");
 42 
 43 end Example;
