  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Even is
  7 
  8    function Is_Even (X: Integer) return Boolean is
  9    begin
 10       return (X mod 2 = 0);
 11    end Is_Even;
 12 
 13    function Jacobi (A,N: Positive) return Integer is
 14       R: Integer;
 15    begin
 16       if A=1 then
 17          return 1;
 18       elsif Is_Even (A) then
 19          R := Jacobi (A/2,N);
 20          if Is_Even(((N**2)-1)/8) then return R; else return -R; end if;
 21       else
 22          R := Jacobi (N mod A, A);
 23          if Is_Even((A-1)*(N-1)/4) then return R; else return -R; end if;
 24       end if;
 25    end Jacobi;
 26

 32 end Even;
