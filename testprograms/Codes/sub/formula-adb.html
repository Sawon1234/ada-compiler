The Ada Program: formula.adb

  1 -- formula.adb: print the sum 1..N of powers 1, 2, 3, and 4
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Formula is
  7 
  8    function Sum_1 (N: Integer) return Integer is
  9    begin
 10       return (N*(N+1)/2);
 11    end Sum_1;
 12 
 13    function Sum_2 (N: Integer) return Integer is
 14    begin
 15       return (N*(N+1)*(2*N+1)/6);
 16    end Sum_2;
 17 
 18    function Sum_3 (N: Integer) return Integer is
 19    begin
 20       return (N**2 * (N+1)**2 / 4);
 21    end Sum_3;
 22 
 23    function Sum_4 (N: Integer) return Integer is
 24    begin
 25       return (Sum_2(N)*(3*N**2+3*N-1)/5);
 26    end Sum_4;
 27 
 28    N: Positive;
 29 
 30 begin
 31    Input: loop
 32       Text_IO.Put ("Enter an integer greater than 0 => ");
 33       -- We must test for EOF after prompt, because it blocks
 34       -- waiting for input, if the input comes from the user.
 35       exit when Text_IO.End_Of_File;
 36       Integer_Text_IO.Get (N);
 37       Text_IO.Put_Line ("  n   sum k  sum k*k  sum k**3   sum k**4");
 38       Text_IO.Put_Line ("---   -----  -------  --------   --------");
 39       Integer_Text_IO.Put (N, Width=>3);
 40       Integer_Text_IO.Put (Sum_1(N), Width=>8);
 41       Integer_Text_IO.Put (Sum_2(N), Width=>9);
 42       Integer_Text_IO.Put (Sum_3(N), Width=>10);
 43       Integer_Text_IO.Put (Sum_4(N), Width=>11);
 44       Text_IO.New_Line (2);
 45    end loop Input;
 46 end Formula;
