The Ada Program: grade.adb

  1 -- grade.adb:  get student records from standard input and print some
  2 
  3 with Ada.Text_IO, Ada.Float_Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Grade is
  7 
  8    type Student_Record_Type is record
  9        Name  : String (1..30);
 10        GPA   : Float;
 11        Grade : Integer;
 12    end record;
 13 
 14    Records :  array (1..100) of Student_Record_Type;
 15 
 16    Record_Number : Integer := Records'First-1;
 17 
 18 begin
 19 
 20    while not (Text_IO.End_Of_File) loop
 21       Record_Number := Record_Number+1;
 22       Text_IO.Get (Records(Record_Number).Name);
 23       Float_Text_IO.Get (Records(Record_Number).GPA);
 24       Integer_Text_IO.Get (Records(Record_Number).Grade);
 25    end loop;
 26 
 27    for I in Records'First .. Record_Number loop
 28       -- print student record if grade same as last student
 29       if Records(I).Grade = Records(Record_Number).Grade then
 30          Text_IO.Put (Records(I).Name & "  ");
 31          Float_Text_IO.Put (Records(I).GPA, Aft=>2, Exp=>0);
 32          Text_IO.New_Line;
 33       end if;
 34    end loop;
 35 
 36 end Grade;
