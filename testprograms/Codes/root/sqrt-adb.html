The Ada Program: sqrt.adb

  1 -- This function calculates an approximation of the square root
  2 -- on the input value.  It works only for nonegative values.
  3 
  4 function Sqrt (X: in Float) return Float is
  5 
  6    Tolerance : constant Float := 0.000001;
  7    Approx    : Float := X / 2.0;
  8 
  9 begin
 10 
 11    -- Find the square root by iterative approximation
 12    while abs (Approx**2 - X) > Tolerance loop
 13       Approx := 0.5 * (Approx + X / Approx);  -- compute new approximation
 14    end loop;
 15 
 16    return (Approx);
 17 
 18 end Sqrt;
