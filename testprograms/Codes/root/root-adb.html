The Ada Program: root.adb

  1 -- root.adb
  2 --
  3 --   Program to compute the square root of a float number to 3 decimal
  4 --   places using Newton's method of approximations
  5 
  6 with Ada.Text_IO, Ada.Float_Text_IO;
  7 use Ada;
  8 
  9 procedure Root is
 10 
 11     Num      : Float;   -- the number to find the root of
 12     Root     : Float;   -- the current approximation
 13     Old_Root : Float;   -- the last approximation
 14 
 15 begin
 16 
 17     -- Get the number
 18     Text_IO.Put_Line ("Enter a float number (>0.0)");
 19     Float_Text_IO.Get (Num);
 20 
 21     -- Get the initial approximation to the sqrt of Num
 22     Text_IO.Put_Line ("Enter initial approx to its square root (>0.0)");
 23     Float_Text_IO.Get (Old_Root);
 24 
 25     -- Find the square root by iterative approximation
 26     loop
 27         Root := 0.5 * (Old_Root + Num / Old_Root); -- compute new approx
 28         exit when abs (Root-Old_Root) < 0.0005;    -- exit when close
 29         Old_Root := Root;
 30     end loop;
 31 
 32     -- Print out the Root
 33     Text_IO.Put ("The square root is ");
 34     Float_Text_IO.Put (Root, 8, 3, 0);
 35     Text_IO.New_Line;
 36 
 37 end Root;
