  1 -- cubic.adb
  2 --
  3 --   Program to compute the square root of a float number to 4 decimal
  4 --   places using Newton's method of approximations
  5 
  6 with Ada.Text_IO, Ada.Float_Text_IO;
  7 use Ada;
  8 
  9 procedure Cubic is
 10 
 11     Num      : Float;          -- the number to find the root of
 12     Root     : Float;          -- the current approximation
 13     Old_Root : Float := 1.0;   -- the last approximation
 14     Tolerance: constant Float := 5.0E-5;
 15 
 16 begin
 17 
 18     -- Get the number
 19     Text_IO.Put_Line ("Enter a real number -- ");
 20     Float_Text_IO.Get (Num);
 21 
 22     -- Find the cube root by iterative approximation
 23     loop
 24         Root := Old_Root + (Num/Root**2 - Root)/3.0;  -- compute new approx
 25         exit when abs (Root-Old_Root) < Tolerance;    -- exit when close
 26         Old_Root := Root;
 27     end loop;
 28 
 29     -- Print out the Root
 30     Text_IO.Put ("The cube root is ");
 31     Float_Text_IO.Put (Root, 7, 4, 0);
 32     Text_IO.New_Line;
 33 
 34 end Cubic;
