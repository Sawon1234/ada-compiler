The Ada Program: complex.adb

  1 -- complex.adb:  package for complex numbers
  2 
  3 with Ada.Numerics.Elementary_Functions;
  4 use  Ada.Numerics.Elementary_Functions;
  5 
  6 package body Complex is
  7 
  8    function "+" (Left, Right: Complex_Number) return Complex_Number is
  9    begin
 10       return (Complex_Number'(
 11          Real      => Left.Real      + Right.Real,
 12          Imaginary => Left.Imaginary + Right.Imaginary
 13       ));
 14    end "+";
 15 
 16    -- Magnitude of a complex number is its distance from the origin.
 17    -- This function is used by the comparison operators.
 18    function Magnitude (X: Complex_Number) return Float is
 19    begin
 20       return (Sqrt (X.Real**2 + X.Imaginary**2));
 21    end Magnitude;
 22 
 23    function "<" (Left, Right: Complex_Number) return Boolean is
 24    begin
 25       return (Magnitude(Left) < Magnitude(Right));
 26    end "<";
 27 
 28 end Complex;
