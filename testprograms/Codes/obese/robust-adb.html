The Ada Program: robust.adb

  1 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Float_Text_IO;
  2 use Ada;
  3 
  4 procedure Robust is
  5 
  6    type Gender_Type is (Male, Female);
  7    package Gender_IO is new Text_IO.Enumeration_IO (Gender_Type);
  8 
  9    subtype Height_Type is Integer range 20 .. 120;  -- inches
 10    subtype Weight_Type is Integer range 10 .. 500;  -- pounds
 11 
 12    procedure Get_Input
 13       (G: out Gender_Type; H: out Height_Type; W: out Weight_Type) is
 14    begin
 15       Input:  loop
 16          Validation: begin
 17             -- get gender, height, and weight
 18             Text_IO.Put_Line ("Enter gender (male/female), height " &
 19                               "(in inches), and weight (in pounds)");
 20             Text_IO.Put ("=> ");
 21             Gender_IO.Get (G);
 22             Integer_Text_IO.Get (H);
 23             Integer_Text_IO.Get (W);
 24 
 25             exit Input;  -- all is good
 26 
 27          exception
 28             when Constraint_Error =>
 29                Text_IO.Put_Line ("Unreasonable height or weight, " &
 30                                  "please try again.");
 31                Text_IO.Skip_Line;    -- Skip over bad data
 32 
 33             when Text_IO.Data_Error =>
 34                Text_IO.Put_Line ("Invalid input, please try again.");
 35                Text_IO.Skip_Line;    -- Skip over bad data
 36 
 37             when IO_Exceptions.End_Error =>
 38                Text_IO.Put_Line ("Not enough input, giving up!");
 39                raise;
 40          end Validation;
 41       end loop Input;
 42    end;
 43 
 44    Gender   : Gender_Type;
 45    Height_Inches : Height_Type;
 46    Weight_Pounds : Weight_Type;
 47    Height_Meters, Weight_Kilos, Body_Mass_Index : Float;
 48    type Cutoff_Type is array (Gender_Type) of Float;
 49    Cutoff : Constant Cutoff_Type := (27.8, 27.3);
 50 
 51 begin
 52 
 53    Get_Input (Gender, Height_Inches, Weight_Pounds);
 54 
 55    -- compute body mass
 56    Height_Meters := Float(Height_Inches)/39.37;
 57    Weight_Kilos  := Float(Weight_Pounds) / 2.2;
 58    Body_Mass_Index := Weight_Kilos / (Height_Meters**2);
 59 
 60    -- print result
 61    Text_IO.Put ("Body mass index of ");
 62    Float_Text_IO.Put (Body_Mass_Index, Fore=>2, Aft=>2, Exp=>0);
 63    Text_IO.Put (" kg/m**2 is ");
 64    if (Body_Mass_Index < Cutoff (Gender)) then
 65      Text_IO.Put ("not ");
 66    end if;
 67    Text_IO.Put_Line ("considered obese for ");
 68    if Gender=Female then
 69       Text_IO.Put ("nonpregnant ");
 70    end if;
 71    Gender_IO.Put (Gender, Set=>Text_IO.Lower_Case);
 72    Text_IO.Put_Line ("s by the Centers of Disease Control.");
 73 
 74 end Robust;
