  3 with Ada.Text_IO;
  4 use Ada;
  5 
  6 procedure Binary_Search is
  7 
  8    type Array_Type is array (Positive range <>) of Integer;
  9 
 10    procedure Search (List    : in Array_Type;
 11                      Value   : in Integer;
 12                      Location: out Natural) is
 13       Midpoint: Positive;
 14    begin
 15       if List'Length = 0 then
 16          Location := 0;
 17       else
 18          Midpoint := (List'First + List'Last) / 2;
 19          if List (Midpoint) = Value then
 20             Location := Midpoint;
 21          elsif List (Midpoint) > Value then
 22             Search (List (List'First..Midpoint-1), Value, Location);
 23          else
 24             Search (List (Midpoint+1..List'Last), Value, Location);
 25          end if;
 26       end if;
 27    end Search;
 28 
 29    Data: constant Array_Type := (1,2,3,4,6,7,8,9);
 30    I   : Integer;
 31 
 32 begin
 33    Search (List=>Data, Value=>6, Location=>I);
 34    if I = 0 then
 35       Text_IO.Put_Line ("Didn't find it.");
 36    else
 37       Text_IO.Put_Line ("Found it at location " & Integer'Image(I));
 38    end if;
 39 end Binary_Search;
