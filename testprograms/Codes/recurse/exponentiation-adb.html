  3 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Float_Text_IO;
  4 use Ada;

  6 procedure Exponentiation is
  7 
  8    function Power (X: Float; N: Positive) return Float is
  9    begin
 10       if N=1 then
 11          return (X);
 12       else
 13          return (X * Power(X,N-1));
 14       end if;
 15    end Power;
 16 
 17    Number: Float;
 18    Exponent : Positive;
 19 
 20 begin
 21 
 22    Text_IO.Put_Line ("Enter a float number and positive exponent");
 23    Float_Text_IO.Get (Number);
 24    Integer_Text_IO.Get (Exponent);
 25    Text_IO.Put ("The answer is ");
 26    Float_Text_IO.Put (Power (Number, Exponent), Exp=>0);
 27    Text_IO.New_Line;
 28 
 29 end Exponentiation;
