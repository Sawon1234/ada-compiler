  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;

  6 procedure fib is
  7 
  8    function Fibonacci (N: in Positive) return Positive is
  9    begin
 10       if N <= 2 then
 11          return (1);
 12       else
 13          return (Fibonacci (N-1) + Fibonacci (N-2));
 14       end if;
 15    end Fibonacci;
 16 
 17    Number: Positive;
 18 
 19 begin
 20 
 21    Text_IO.Put_Line ("Enter a positive number");
 22    Integer_Text_IO.Get (Number);
 23    Text_IO.Put ("Fibonacci(");
 24    Integer_Text_IO.Put (Number, Width=>1);
 25    Text_IO.Put (") = ");
 26    Integer_Text_IO.Put (Fibonacci(Number), Width=>1);
 27    Text_IO.New_Line;
 28 
 29 end fib;
