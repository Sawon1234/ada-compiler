  1 with Ada.Text_IO;
  2 use Ada;
  3 
  4 procedure Display (N: Integer) is
  5    Limit: Integer := 20;
  6 begin
  7    for I in -Limit..-1 loop
  8      if I<N then
  9         Text_IO.Put (' ');
 10      else
 11         Text_IO.Put ('*');
 12      end if;
 13    end loop;
 14    Text_IO.Put ('|');
 15    for I in 1..Limit loop
 16      if I>N then
 17         Text_IO.Put (' ');
 18      else
 19         Text_IO.Put ('*');
 20      end if;
 21    end loop;
 22    Text_IO.New_Line;
 23 end Display;
