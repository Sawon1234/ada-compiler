The Ada Program: hanoi.adb

  1 -- hanoi.adb:  the classic Towers of Hanoi recursive program
  2 
  3 with Ada.Text_IO;
  4 use Ada;
  5 
  6 procedure Hanoi is
  7 
  8    type Pegs is ('A', 'B', 'C');
  9    package Peg_IO is new Text_IO.Enumeration_IO (Pegs);
 10    type Disks is (Very_Small, Small, Medium, Large, Very_Large);
 11    package Disk_IO is new Text_IO.Enumeration_IO (Disks);
 12 
 13    -- Move disk "D" from tower "From" to tower "To".  We assume
 14    -- "D" is the topmost disk on the "From" tower and it is
 15    -- smaller than the topmost disk (if any) on the "To" tower.
 16    procedure MoveDisk (D: Disks; From, To: Pegs) is
 17    begin
 18       Text_IO.Put ("Move ");              Disk_IO.Put (D, Width=>10);
 19       Text_IO.Put (" disk from tower ");  Peg_IO.Put (From);
 20       Text_IO.Put (" to tower ");         Peg_IO.Put (To);
 21       Text_IO.New_Line;
 22    end MoveDisk;
 23 
 24    -- Move the top "N" through "Very_Small" disks on the "From" tower
 25    -- to the "To" tower, making use of the auxiliary "Aux" tower.
 26    procedure Tower (From, To, Aux: Pegs; N: Disks) is
 27    begin
 28       if N=Disks'First then
 29          MoveDisk (Disks'First, From, To);
 30       else
 31          Tower (From, Aux, To, Disks'Pred(N));
 32          MoveDisk (N, From, To);
 33          Tower (Aux, To, From, Disks'Pred(N));
 34       end if;
 35    end Tower;
 36 
 37 begin
 38 
 39    -- Print the instructions for moving all the disks
 40    Tower (From=>'A', To=>'B', Aux=>'C', N=>Disks'Last);
 41 
 42 end Hanoi;
