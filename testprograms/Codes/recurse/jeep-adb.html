The Ada Program: jeep.adb

  1 -- jeep.adb:  library unit (function) to find distance from desert oasis
  2 
  3 function Jeep (
  4    N: Positive;  -- number of cans of gas, each containing C liters
  5    C: Float;     -- capacity of can and tank of jeep
  6    R: Float      -- rate of consumption in liters per 100 kilometers
  7 ) return Float is
  8 
  9    -- the Jeep starts out empty.
 10 
 11    -- distance in kilometers the jeep can travel on one tank of gas
 12    Distance: Float := (C*Float(100)/R);
 13 
 14    -- journeys needed to move N-1 cans of gas (jeep can carry only 1)
 15    Journeys: Positive;
 16 
 17 begin
 18 
 19    if N=1 then
 20       return (Distance);
 21    else
 22       -- back and forth for N-2 cans; 1 outbound journey for last can
 23       Journeys := 2*(N-2) + 1;
 24       return (Distance/Float(Journeys) + Jeep (N-1, C, R));
 25    end if;
 26 
 27 end Jeep;
