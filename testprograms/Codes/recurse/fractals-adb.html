  9 procedure Fractals is
 10 
 11    Seed: Float_Random.Generator;
 12 
 13    -- Compute a random number between "Lower" and "Upper"
 14    function Random (Lower, Upper: Float) return Float is
 15       R: Float := Lower;
 16    begin
 17       R := R + (Upper - Lower) * Float_Random.Random (Seed);
 18       return R;
 19    end Random;
 20 
 21    Epsilon: constant Float := 1.0;
 22 
 23    procedure Random_Fractal (Left, Right, Width: Float) is
 24       Mid: Float := (Left + Right) / 2.0;
 25    begin
 26       if Width < Epsilon then
 27          Display (Integer(Mid));
 28       else
 29          Mid := Mid + Random (-Width, Width);
 30          Random_Fractal (Left, Mid, Width/2.0);
 31          Random_Fractal (Mid, Right, Width/2.0);
 32       end if;
 33    end Random_Fractal;
 34 
 35 begin
 36    Float_Random.Reset (Seed);
 37    Random_Fractal (10.0, 10.0, 16.0);
 38 end Fractals;
