  3 -- Computes the number of combinations (committees) of size "Members"
  4 -- that can be made from a collection of size "Total".
  5 
  6 function Combinations (Total  : Positive;
  7                        Members: Positive) return Positive is
  8 begin
  9 
 10    if Members = 1 then
 11       return (Total);          -- There are "Total" committees of size 1
 12    elsif Members = Total then
 13       return (1);              -- There is 1 committee of size "Total"
 14    else
 15       -- pick an individual 'A' from the committee
 16       return (
 17          Combinations (Total-1, Members-1) +  -- committees with 'A'
 18          Combinations (Total-1, Members)      -- committees without 'A'
 19       );
 20    end if;
 21 
 22 end Combinations;
