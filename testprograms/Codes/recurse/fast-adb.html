The Ada Program: fast.adb

  1 -- fast.adb: print a table of fast growing functions
  2 
  3 with Ack;
  4 with Ada.Text_IO, Ada.Long_Long_Integer_Text_IO;
  5 use Ada;
  6 
  7 
  8 procedure Fast is
  9    subtype Natural is Long_Long_Integer range 0 .. Long_Long_Integer'Last;
 10 
 11    LLN : Natural;
 12 
 13    function Factorial (N: Natural) return Natural is
 14    begin
 15       if N=0 then
 16          return (1);
 17       else
 18          return (N * Factorial (N-1));
 19       end if;
 20    end Factorial;
 21 
 22    function Pow (Base: Natural; Exp: Natural) return Natural is
 23       Semi: Natural;
 24    begin
 25       if Exp=0 then
 26          return (1);
 27       else
 28          Semi := Pow (Base, Exp/2);
 29          if (Exp mod 2 = 0) then
 30             -- Exp = Exp/2 + Exp/2
 31             return Semi*Semi;
 32          else
 33             -- Exp = 1 + Floor(Exp/2) + Floor(Exp/2)
 34             return Base*Semi*Semi;
 35          end if;
 36       end if;
 37    end Pow;
 38 
 39    --  Fib(N) <= (5/3)**N < 2**N
 40    --  N! < N**N
 41 
 42 begin
 43 
 44    Text_IO.Put_Line ("  n    n**2     n**3       2**N            3**N"&
 45                      "                  N!                 N*N");
 46    Text_IO.Put_Line ("---   -----  -------  ---------  --------------"&
 47                      "   -----------------  ------------------");
 48 
 49    for N in 1 .. 30 loop
 50       LLN := Long_Long_Integer (N);
 51       Long_Long_Integer_Text_IO.Put (LLN, Width=>3);
 52       Long_Long_Integer_Text_IO.Put (LLN**2, Width=>8);
 53       Long_Long_Integer_Text_IO.Put (LLN**3, Width=>9);
 54       -- NB. Implicit convertion of Integer to Long_Long_Integer.
 55       Long_Long_Integer_Text_IO.Put (2**N, Width=>11);
 56       Long_Long_Integer_Text_IO.Put (3**N, Width=>16);
 57       if N<20 then
 58          Long_Long_Integer_Text_IO.Put (Factorial (LLN), Width=>20);
 59          if N<16 then
 60             Long_Long_Integer_Text_IO.Put (Pow (LLN, LLN), Width=>20);
 61          end if;
 62       end if;
 63          Text_IO.New_Line;
 64    end loop;
 65 
 66 end Fast;
