  3 with Ada.Text_IO;
  4 use Ada;

  6 procedure Linear_Search is
  7 
  8    type Array_Type is array (Positive range <>) of Integer;
  9 
 10    function Search (List: Array_Type; Value: Integer) return Boolean is
 11    begin
 12       if List'Length = 0 then
 13          return (False);
 14       elsif Value = List(List'First) then
 15          return (True);
 16       else
 17          return (Search (List(List'First+1..List'Last), Value));
 18       end if;
 19    end Search;
 20 
 21    Data : constant Array_Type := (1,2,3,4,6,7,8,9);
 22 
 23 begin
 24    if (Search (List=>Data, Value=>5)) then
 25       Text_IO.Put_Line ("Found it!");
 26    else
 27       Text_IO.Put_Line ("Didn't find it.");
 28    end if;
 29 end Linear_Search;
