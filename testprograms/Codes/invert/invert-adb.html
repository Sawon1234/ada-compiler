The Ada Program: invert.adb

  1 with Matrix_Package;
  2 with Matrix_IO, Text_IO, Ada.Float_Text_IO;
  3 
  4 package body Invert is
  5 
  6   procedure Invert_Matrix (Matrix : in out Matrix_Package.Matrix_Type) is
  7      Inverse : Matrix_Package.Matrix_Type (Matrix'Range(1), Matrix'Range(2));
  8      Pivot_Row : Integer range Matrix'First(1)..Matrix'Last(1)+1;
  9      Pivot_Factor, Multiple: Float;
 10   begin
 11      -- Initialize "Inverse" to the identity matrix
 12      for Row in Inverse'Range(1) loop
 13         for Col in Inverse'Range(2) loop
 14           if Row=Col then
 15              Inverse(Row,Col) := 1.0;
 16           else
 17              Inverse(Row,Col) := 0.0;
 18           end if;
 19         end loop;
 20      end loop;
 21 
 22 
 23 Matrix_IO.Put_Matrix (Matrix);
 24 Text_IO.New_Line (2);
 25 
 26      -- Apply identical row operations to Matrix and Inverse until Matrix
 27      -- is found to be singular or transformed into the identity matrix
 28 
 29      for I in Matrix'Range(2) loop
 30 Text_IO.Put_Line ("for I = " & Integer'Image(I));
 31 
 32         -- Search for a non-zero element in column i of Matrix,
 33         -- at or below row i
 34         Pivot_Row := I;
 35         while Pivot_Row <=Matrix'Last(1) and then Matrix (Pivot_Row,I) =0.0 loop
 36            Pivot_Row := Pivot_Row + 1;
 37         end loop;
 38 
 39         -- If no non-zero element found, Matrix is singular
 40         if Pivot_Row > Matrix'Last(1) then
 41            raise Singular;
 42         end if;
 43 
 44         -- Exchange rows if necessary to place non-zero element in Matrix(i,i)
 45         if Pivot_Row /= I then
 46            Matrix_Package.Exchange_Rows (Matrix,I,Pivot_Row);
 47            Matrix_Package.Exchange_Rows (Inverse,I,Pivot_Row);
 48 
 49 Text_IO.Put_Line ("exch with Pivot_Row = " & Integer'Image(Pivot_Row));
 50 Matrix_IO.Put_Matrix (Inverse);
 51 Matrix_IO.Put_Matrix (Matrix);
 52 Text_IO.New_Line (2);
 53         end if;
 54 
 55         -- Multiply each elemnet in row i by the same value,
 56         -- so that matrix(i,i) = 1.0
 57         Pivot_Factor := 1.0 / Matrix(I,I);
 58 Text_IO.Put ("Pivot_factor = ");
 59 Ada.Float_Text_IO.Put (Pivot_Factor);
 60 Text_IO.New_Line;
 61         Matrix_Package.Multiply_Row (Matrix, Pivot_Factor, I);
 62         Matrix_Package.Multiply_Row (Inverse, Pivot_Factor, I);
 63 Matrix_IO.Put_Matrix (Inverse);
 64 Matrix_IO.Put_Matrix (Matrix);
 65 Text_IO.New_Line (2);
 66         -- Add multiples of row i to each other row so that Matrix(i,i)
 67         -- becomes the only non-zero element in column i of Matrix
 68         for J in Matrix'Range(1) loop
 69            if J /= I then
 70               Multiple := - Matrix (J,I);
 71 Text_IO.Put ("For J = " & Integer'Image(J) & ".  Multiple = ");
 72 Ada.Float_Text_IO.Put (Multiple);
 73 Text_IO.New_Line;
 74               Matrix_Package.Add_Row_Multiple (Matrix, Multiple, I, J);
 75               Matrix_Package.Add_Row_Multiple (Inverse, Multiple, I, J);
 76 Matrix_IO.Put_Matrix (Inverse);
 77 Matrix_IO.Put_Matrix (Matrix);
 78 Text_IO.New_Line (2);
 79            end if;
 80         end loop;
 81 
 82      end loop;
 83      Matrix := Inverse;
 84   end Invert_Matrix;
 85 
 86 end Invert;
