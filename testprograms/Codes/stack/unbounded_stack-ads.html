The Ada Program: unbounded_stack.ads

  1 -- unbounded_stack.ads:  generic package for stack data structure
  2 
  3 generic
  4 
  5    type Element_Type is private;  -- The type of stack elements
  6 
  7 package Unbounded_Stack is
  8 
  9    type Stack_Type is limited private;
 10 
 11    Underflow, Overflow: exception;
 12 
 13    -- Test whether a stack is empty (contains no elements).
 14    function Empty (Stack:  Stack_Type) return Boolean;
 15 
 16    -- Whether or not the stack is full, depends on the OS.  In any case
 17    -- it is hard to detect reliably and should be extremely rare.
 18 
 19    -- Remove all elements from the stack.
 20    procedure Clear (Stack: in out Stack_Type);
 21 
 22    -- Add a new element to the top of the stack; if already full,
 23    -- then the exception "Overflow" is raised and stack is unchanged.
 24    procedure Push (Stack       : in out Stack_Type;
 25                    New_Element : in     Element_Type);
 26 
 27 
 28    -- Remove the top element from stack and return it in "Top_Element".
 29    -- If stack is empty, then the exception "Underflow" is raised.
 30    procedure Pop (Stack       : in out Stack_Type;
 31                   Top_Element :    out Element_Type);
 32 
 33 private
 34 
 35    type Node_Type;                       -- Incomplete type
 36 
 37    type Stack_Type is access Node_Type;
 38 
 39    type Node_Type is                     -- Full type
 40       record
 41          Info : Element_Type;    -- One item
 42          Next : Stack_Type;      -- The next node in the stack
 43       end record;
 44 
 45 end Unbounded_Stack;
