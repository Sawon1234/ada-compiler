  3 package body Bounded_Stack is
  4 
  5    function Empty (Stack: Stack_Type) return Boolean is
  6    begin
  7       return (Stack.Top = 0);
  8    end Empty;
  9 
 10    function Full (Stack: Stack_Type) return Boolean is
 11    begin
 12       return (Stack.Top = Stack.Max_Size);
 13    end Full;
 14 
 15    procedure Clear (Stack: in out Stack_Type) is
 16    begin
 17       Stack.Top := 0;
 18    end Clear;
 19 
 20    procedure Push (Stack       : in out Stack_Type;
 21                    New_Element : in     Element_Type) is
 22    begin
 23       if not Full (Stack) then
 24          Stack.Top := Stack.Top + 1;
 25          Stack.Elements (Stack.Top) := New_Element;
 26       else
 27          raise Overflow;
 28       end if;
 29    end Push;
 30 
 31    procedure Pop (Stack         : in out Stack_Type;
 32                   Top_Element   :    out Element_Type) is
 33    begin
 34       if not Empty (Stack) then
 35          Top_Element := Stack.Elements (Stack.Top);
 36          Stack.Top := Stack.Top - 1;
 37       else
 38          raise Underflow;
 39       end if;
 40    end Pop;
 41 
 42 end Bounded_Stack;
