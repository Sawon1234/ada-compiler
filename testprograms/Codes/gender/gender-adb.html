The Ada Program: gender.adb

  1 -- gender.adb: simple dialog using "Put" for characters
  2 
  3 with Ada.Text_IO;
  4 use Ada;
  5 
  6 procedure Gender is
  7    Gender: Character;
  8 begin
  9    Text_IO.Put ("Enter the gender of the person (""m"" or ""f"") ==> ");
 10    Text_IO.Get (Gender);
 11    if (Gender = 'm') then
 12       Text_IO.Put_Line ("Male");
 13    else
 14       Text_IO.Put_Line ("Female");
 15    end if;
 16 end Gender;
