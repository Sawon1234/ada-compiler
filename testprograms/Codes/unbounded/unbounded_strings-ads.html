The Ada Program: unbounded_strings.ads

  1 -- unbounded_strings.ads: generic package implementing unbounded-length strings
  2 
  3 with Ada.Text_IO;
  4 use Ada;
  5 
  6 package Unbounded_Strings is
  7 
  8    type Unbounded_String is limited private;         -- Unbounded-length string
  9 
 10    Null_Unbounded_String: constant Unbounded_String; -- A string with no characters
 11 
 12    ------------------------------------------------------------------------------
 13    -- Relational operators
 14    ------------------------------------------------------------------------------
 15    function "<"   (Left, Right : Unbounded_String) return Boolean;
 16    function "<="  (Left, Right : Unbounded_String) return Boolean;
 17    function ">"   (Left, Right : Unbounded_String) return Boolean;
 18    function ">="  (Left, Right : Unbounded_String) return Boolean;
 19    function Equal (Left, Right : Unbounded_String) return Boolean;
 20 
 21    ------------------------------------------------------------------------------
 22    -- Concatenation operators
 23    -- Exceptions:  CONSTRAINT_ERROR  is raised if the combined length of the two
 24    --              parameters exceeds the value of Max_Length
 25     -----------------------------------------------------------------------------
 26    function "&" (Left, Right: Unbounded_String) return Unbounded_String;
 27    function "&" (Left: Unbounded_String; Right: Character) return Unbounded_String;
 28    function "&" (Left: Character; Right: Unbounded_String) return Unbounded_String;
 29    function "&" (Left: Unbounded_String; Right: String) return Unbounded_String;
 30    function "&" (Left: String; Right: Unbounded_String) return Unbounded_String;
 31 
 32    ------------------------------------------------------------------------------
 33    -- Number of characters in a string
 34    function Length (Source: Unbounded_String) return Natural;
 35 
 36    ------------------------------------------------------------------------------
 37    procedure Assign (Target: in out Unbounded_String;
 38                      Source: in     Unbounded_String);
 39 
 40    ------------------------------------------------------------------------------
 41    -- The substring of "Source" from "Low" to "High".
 42    -- Exceptions:  CONSTRAINT_ERROR is raised if Low <= High or
 43    -- High is greater than Length (Source)
 44    function Slice (Source : Unbounded_String;
 45                    Low    : Positive;
 46                    High   : Natural) return Unbounded_String;
 47 
 48    ------------------------------------------------------------------------------
 49    -- Return the character in the string at the position given by Index.
 50    -- Exceptions: CONSTRAINT_ERROR is raised if Index is greater than
 51    --             Length(Source).
 52    function Element (Source : in Unbounded_String;
 53                      Index  : in Positive) return Character;
 54 
 55    ------------------------------------------------------------------------------
 56    -- Search for "Pattern" in "Source". If the pattern is found, "Index"
 57    -- returns the smallest index I such that the slice of "Source" starting
 58    -- at I matches the pattern.  Otherwise, 0 is returned.
 59    function Index (Source  : in Unbounded_String;
 60                    Pattern : in Unbounded_String) return Natural;
 61 
 62 
 63    --------------
 64    -- Conversions
 65    --------------
 66 
 67    ------------------------------------------------------------------------------
 68    -- This function converts an unbounded string to a fixed-length string
 69    function To_String (Source: Unbounded_String) return String;
 70 
 71    ------------------------------------------------------------------------------
 72    -- This function converts a fixed-length string to a Unbounded-length string
 73    -- Exceptions:  CONSTRAINT_ERROR is raised if Source'Last > Max_Length
 74    function To_Unbounded_String (Source: String) return Unbounded_String;
 75 
 76    ------------------------------------------------------------------------------
 77    -- Input and Output Operations
 78    ------------------------------------------------------------------------------
 79 
 80    ------------------------------------------------------------------------------
 81    -- Unbounded-length string equivalent to Text_IO.Get_Line
 82    procedure Get_Line (File: in  Text_IO.File_Type := Text_IO.Current_Input;
 83                        Item: out Unbounded_String);
 84 
 85 
 86    ------------------------------------------------------------------------------
 87    -- Unbounded-length string equivalent to Text_IO.Put
 88    procedure Put (File: in Text_IO.File_Type := Text_IO.Current_Output;
 89                   Item: in Unbounded_String);
 90 
 91 private
 92 
 93    type Unbounded_String is access String;
 94 
 95    Null_Unbounded_String : constant Unbounded_String := new String'("");
 96 
 97 end Unbounded_Strings;
