The Ada Program: nice.adb

  1 -- nice.adb:  compute "score" of words, print ever improving ones
  2 
  3 -- NB. "nice" is a comman in Unix; invoke this program as "./nice"
  4 
  5 with Text_IO, Ada.Integer_Text_IO;
  6 use Ada;
  7 
  8 procedure Nice is
  9 
 10    Line: String (1..100);   -- buffer for input line
 11    Length: Integer;         -- lenght of line in buffer "Line"
 12    Best : Natural := 0;     -- best score >0 so far
 13    Score: Integer;          -- score of current word
 14    Max_Length : constant := 25;
 15 
 16 begin
 17 
 18    Text_IO.Put_Line ("word                  score");
 19    Text_IO.Put_Line ("============================================");
 20    Text_IO.Flush;
 21 
 22    while not (Text_IO.End_Of_File) loop
 23       Text_IO.Get_Line (Line, Length);
 24       Score := 0;
 25       Add: for i in 1..Length loop
 26          case Line(I) is
 27             when 'n'|'i'|'c'|'e' => Score := Score + 1;
 28             -- duplicates not allowed in case statement!
 29             when 'a'..'b'|'d'|'f'..'h'|'j'..'m'|'o'..'z' => Score := Score - 1;
 30             when others => Score := 0; exit Add;  -- reject
 31          end case;
 32       end loop Add;
 33 
 34       if (Score > Best) then
 35          Best := Score;
 36          Text_IO.Put (Item=>Line (1..Length));
 37          Text_IO.Set_Col (Max_Length);
 38          Integer_Text_IO.Put (Item=>Score,  Width=>7);
 39          Text_IO.New_Line;
 40       end if;
 41    end loop;
 42 
 43 end Nice;
