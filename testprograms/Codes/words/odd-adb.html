The Ada Program: odd.adb

  1 -- odd.adb:  compute "score" of words of all odd characters
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Odd is
  7 
  8    Line: String (1..100);
  9    Length: Integer;
 10    LC_A: constant Integer := character'Pos('a');
 11    UC_A: constant Integer := character'Pos('A');
 12    All_Odd: Boolean;
 13    Max_Length_So_Far: Integer := 0;
 14 
 15 begin
 16    while not (Text_IO.End_Of_File) loop
 17       Text_IO.Get_Line (Line, Length);
 18       if Length>3 then
 19          All_Odd := False;
 20          Add: for i in 1..Length loop
 21             if Line(i) in 'a'..'z' then
 22                All_Odd := (Character'Pos(Line(i))-LC_A) mod 2 = 0;
 23             elsif Line(i) in 'A'..'Z' then
 24                All_Odd := (Character'Pos(Line(i))-UC_A) mod 2 = 0;
 25             else
 26                All_Odd := true;
 27             end if;
 28             exit Add when not All_Odd;
 29          end loop Add;
 30          if All_Odd and then Max_Length_So_Far < Length then
 31             Max_Length_So_Far := Length;
 32             Text_IO.Put (Item=>Line (1..Length));
 33             Text_IO.Set_Col (20);
 34             Integer_Text_IO.Put (Length);
 35             Text_IO.Put_Line ("  all odd");
 36          end if;
 37       end if;
 38    end loop;
 39 end Odd;
