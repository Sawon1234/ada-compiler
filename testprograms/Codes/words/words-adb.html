The Ada Program: words.adb

  1 -- words.adb:  compute "score" of words on standard input
  2 
  3 with Text_IO, Ada.Integer_Text_IO, Ada.Float_Text_IO;
  4 use Ada;
  5 
  6 procedure Words is
  7 
  8    Line: String (1..100);
  9    Length: Integer;
 10    LC_A: constant Integer := character'Pos('a')-1;
 11    UC_A: constant Integer := character'Pos('A')-1;
 12    Score: Integer;
 13    Avg: Float;
 14 
 15 begin
 16 
 17    while not (Text_IO.End_Of_File) loop
 18       Text_IO.Get_Line (Line, Length);
 19       if Length>3 then     -- ignore short words
 20          Score := 0;
 21          Add: for i in 1..Length loop
 22             if Line(i) in 'a'..'z' then
 23                Score := Score + Character'Pos(Line(i))-LC_A;
 24             elsif Line(i) in 'A'..'Z' then
 25                Score := Score + Character'Pos(Line(i))-UC_A;
 26             else
 27                Score := 0;
 28                exit Add;
 29             end if;
 30          end loop Add;
 31          Text_IO.Put (Item=>Line (1..Length));
 32          Text_IO.Set_Col (20);
 33          Integer_Text_IO.Put (Item=>Length, Width=>4);
 34          Integer_Text_IO.Put (Item=>Score,  Width=>4);
 35          Float_Text_IO.Put (Item=>Float(Score)/Float(Length),
 36                             Fore=>6, Aft=>2, Exp=>0);
 37          Text_IO.New_Line;
 38       end if;
 39    end loop;
 40 
 41 end Words;
