The Ada Program: score.adb

  1 -- score.adb:  compute "score" of words, print ever improving ones
  2 
  3 with Text_IO, Ada.Integer_Text_IO, Ada.Float_Text_IO;
  4 use Ada;
  5 
  6 procedure Score is
  7 
  8    Line: String (1..100);
  9    Length: Integer;
 10    LC_A: constant Integer := Character'Pos('a')-1;
 11    UC_A: constant Integer := Character'Pos('A')-1;
 12    Score  : Integer;
 13    Max_Length : constant := 25;
 14    Lowest: array(1..Max_Length) of Positive := (others=>Positive'Last);
 15 
 16 begin
 17 
 18    Text_IO.Put_Line ("word                  length  score  average");
 19    Text_IO.Put_Line ("============================================");
 20 
 21    while not (Text_IO.End_Of_File) loop
 22       Text_IO.Get_Line (Line, Length);
 23       Score := 0;
 24       Add: for i in 1..Length loop
 25          if Line(i) in 'a'..'z' then
 26             Score := Score + Character'Pos(Line(i))-LC_A;
 27          elsif Line(i) in 'A'..'Z' then
 28             Score := Score + Character'Pos(Line(i))-UC_A;
 29          else
 30             Score := Positive'Last;
 31             exit Add;
 32          end if;
 33       end loop Add;
 34 
 35       if (Score < Lowest (Length)) then
 36          Lowest (Length) := Score;
 37          Text_IO.Put (Item=>Line (1..Length));
 38          Text_IO.Set_Col (Max_Length);
 39          Integer_Text_IO.Put (Item=>Length, Width=>4);
 40          Integer_Text_IO.Put (Item=>Score,  Width=>7);
 41          Float_Text_IO.Put (Item=>Float(Score)/Float(Length),
 42                             Fore=>6, Aft=>2, Exp=>0);
 43          Text_IO.New_Line;
 44       end if;
 45    end loop;
 46 
 47 end Score;
