The Ada Program: longup.adb

  1 -- From Gries, "Science of Programming", §20.2
  2 
  3 -- If one deletes i (not necessarily adjacent) values from a list
  4 -- of length n, one has a subsequence of length n-i.  The longest,
  5 -- non-decreasing such subsequence is called an upsequence.
  6 
  7 with Ada.Text_IO; use Ada.Text_IO;
  8 with Ada.Integer_Text_IO; use Ada.Integer_Text_IO;
  9 
 10 procedure Longup is
 11    C: Character;
 12    I: Integer := 1;
 13    Length: Integer;
 14    A:  array (1..40) of Integer;
 15    M:  array (1..40) of Integer;
 16 
 17 begin
 18 
 19    Get (C);        -- ignore opening '('
 20    loop
 21       Get (A(I));  -- get an integer
 22       Get (C);     -- get a ',' or a ')'
 23 
 24       -- M[K] for 1<=K<=Length is the smallest value ending an
 25       -- upsequence of length K in A[1:I-1]
 26 
 27       if I=1 then
 28          Length := 1;  M(1):=A(1);
 29 
 30       elsif A(I)>=M(Length) then
 31          Length := Length+1; -- a longer upsequence found
 32          M(Length) := A(I);
 33 
 34       elsif A(I)<M(1) then
 35          M(1) := A(I);      -- smallest value found so far
 36 
 37       else
 38          -- find J st M[J-1]<=A[I]<M[J]
 39          declare
 40             H:Integer:=1; J:Integer:=Length; E:Integer;
 41          begin
 42             while (H /= (J-1)) loop
 43                E := (H+J)/2;
 44                if M(E)<=A(I) then
 45                   H:=E;
 46                else
 47                   J:=E;
 48                end if;
 49             end loop;
 50             M(J) := A(I);
 51          end;
 52       end if;
 53 
 54       exit when C=')';
 55       I:=I+1;
 56 
 57    end loop;
 58 
 59    Put ("Length of longest Upsequence = ");
 60    Put (Length, Width=>0);
 61    New_Line;
 62 
 63 end Longup;
