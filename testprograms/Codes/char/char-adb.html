The Ada Program: char.adb

  1 -- char.adb:  get a character, write the character and its position
  2 
  3 with Ada.Text_IO;
  4 use Ada;
  5 
  6 procedure Char is
  7    C: Character;
  8 begin
  9    Text_IO.Put ("Enter a character -- ");
 10    Text_IO.Get (C);
 11    Text_IO.Put_Line ("Your character is: " & C);
 12    Text_IO.Put (C);
 13    Text_IO.Put (" is at position" & Integer'Image(Character'Pos(C)));
 14    Text_IO.New_Line (2);
 15 end Char;
