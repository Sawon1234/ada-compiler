  1 -- ret.adb:  Passing subprograms up
  2 
  3 procedure Ret is
  4    procedure P (X: Integer) is begin null; end P;
  5    procedure N is
  6       procedure T (X: Integer) is begin null; end T;
  7 
  8       type F is access procedure (X: Integer);
  9 
 10       function R (B: Boolean) return F is
 11       begin
 12          if B then
 13             return P'Access;  -- no problem
 14          else
 15             return T'Access;  -- no problem
 16          end if;
 17       end R;
 18    begin
 19       R (True) (3);   R (False) (3);
 20    end N;
 21 begin
 22    N;
 23 end Ret;
