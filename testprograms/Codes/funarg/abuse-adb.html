The Ada Program: abuse.adb

  1 -- abuse.adb:  try to slip closure out of its proper context
  2 
  3 procedure Abuse is
  4 
  5    type F is access procedure (X: Integer);
  6    Z: F;
  7 
  8    procedure Q (FP: F) is
  9    begin
 10       FP(5);  -- downward funarg is quite safe
 11 
 12       -- assigning to non-local is much like a return;
 13       -- impossible to check if ep is escaping
 14 
 15       Z := FP;  -- put closure in non-local variable
 16    end Q;
 17 
 18    procedure P (X: Integer) is begin null; end P;
 19 
 20    procedure R is
 21       procedure T (X: Integer) is begin null; end T;
 22    begin
 23       Q (P'Access);  -- no problem
 24       Q (T'Access);  -- subprogram "T" deeper than access type "F"
 25    end R;
 26 
 27 begin
 28 
 29    R;
 30    Z(6);  -- could call "T" without "R" on stack!
 31 
 32 end Abuse;
