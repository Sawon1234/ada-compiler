  1 -- up.adb:  Passing subprograms up
  2 
  3 procedure Up is
  4 
  5    type F is access procedure (X: Integer);
  6 
  7    procedure P (X: Integer) is begin null; end P;
  8 
  9    function R (B: Boolean) return F is
 10       N: Integer;
 11       procedure T (X: Integer) is begin N:=X; end T;
 12    begin
 13       if B then
 14          return P'Access;  -- no problem
 15       else
 16          return T'Access;  -- subprogram "T" deeper than access type "F"
 17       end if;
 18 
 19    end R;
 20 
 21    FP: F;
 22 
 23 begin
 24 
 25    FP := R (True);   FP (3);
 26    FP := R (False);  FP (3);
 27 
 28    -- the same again without using a temporary variable

 31 end Up;
