The Ada Program: square.adb

  1 -- square.adb:  compute square of input; check for valid input
  2 
  3 with Ada.Text_IO;
  4 use Ada;
  5 
  6 procedure Square is
  7    package Natural_IO is new Text_IO.Integer_IO (Num => Natural);
  8    N: Natural;
  9 
 10 -- input from 0 to 2,147,483,647=2**31 - 1
 11 -- answers for input from 0 to 46340=floor(sqrt(2**31 - 1))
 12 
 13 -- Under GNAT you need the -gnato option: "gnatmake square -cargs -gnato"
 14 -- To get the (expensive) check for integer overflow
 15 
 16 -- This won't work:
 17 --   Input: while not (Text_IO.End_Of_File) loop
 18 -- I tried Get_Immediate (Item,Available) too
 19 
 20 begin
 21    Input: loop
 22       begin
 23          Text_IO.Put ("Enter a non-negative integer -- ");
 24          Natural_IO.Get (N);
 25          Natural_IO.Put (N*N);
 26          Text_IO.New_Line;
 27       exception
 28          when Text_IO.End_Error => exit Input;
 29          when Text_IO.Data_Error =>
 30             Text_IO.Put_Line ("Bad input. Either negative, too big, garbage, etc.");
 31             -- Note:  input not consumed
 32          when Constraint_Error =>
 33             Text_IO.Put_Line ("Input too large; overflow.");
 34       end;
 35       Text_IO.Skip_Line;  -- discard line
 36    end loop Input;
 37 end Square;
