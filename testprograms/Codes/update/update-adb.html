The Ada Program: update.adb

  1 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Float_Text_IO, Ada.Sequential_IO;
  2 use Ada;
  3 
  4 procedure Update is
  5 
  6    -- This program performs an update of a sequential master file

  7    -- Assumption:  No customer has an account number equal to Positive'last

  8 
  9    type Transaction_Type is (Payment, Charge, Limit);
 10 
 11    subtype Name_String    is String (1..20);
 12    subtype Address_String is String (1..40);
 13    subtype State_String   is String (1..2);
 14 
 15    type Name_Rec is                    -- For customer name

 16       record
 17          First : Name_String;
 18          Last  : Name_String;
 19       end record;
 20 
 21    type Address_Rec is                 -- For customer address

 22       record
 23          Street : Address_String;
 24          City   : Address_String;
 25          State  : State_String;
 26          Zip    : Positive;
 27       end record;
 28 
 29    type Customer_Rec is                -- Customer Account information

 30       record
 31          Account_Number : Positive;
 32          Name           : Name_Rec;
 33          Address        : Address_Rec;
 34          Credit_Limit   : Float;
 35          Balance        : Float;
 36       end record;
 37 
 38    -- Instantiate I/O Packages

 39    package Customer_IO is new Sequential_IO (Element_Type => Customer_Rec);
 40    package Choice_IO   is new Text_IO.Enumeration_IO (Enum => Transaction_Type);
 41 
 42    New_Master  : Customer_IO.File_Type;  -- Updated account records

 43    Old_Master  : Customer_IO.FIle_Type;  -- Old account records

 44    Transaction : Text_IO.File_Type;      -- Changes to accounts

 45    Report      : Text_IO.File_Type;      -- Error and Control report

 46 
 47    Account_Number : Positive;     -- The account number of the transaction

 48    Customer       : Customer_Rec; -- The customer record for the current account

 49 
 50    -----------------------------------------------------------------------------

 51    procedure Prepare_Files
 52          (Old_Master : in out Customer_IO.File_Type;  -- Accounts before update

 53           New_Master : in out Customer_IO.File_Type;  -- Accounts after update

 54           Transaction : in out Text_IO.File_Type;     -- Changes to accounts

 55           Report      : in out Text_IO.File_Type) is  -- Update report

 56 
 57    -- This procedure prepares all of the files for the sequential file update

 58    begin   -- Prepare Files

 59       Customer_IO.Open (File => Old_Master,
 60                         Mode => Customer_IO.In_File,
 61                         Name => "Old_Master.Dat");
 62       Customer_IO.Create (File => New_Master,
 63                           Mode => Customer_IO.Out_File,
 64                           Name => "New_Master.Dat");
 65       Text_IO.Open (File => Transaction,
 66                     Mode => Text_IO.In_File,
 67                     Name => "Transaction.Dat");
 68       Text_IO.Create (File => Report,
 69                       Mode => Text_IO.Out_File,
 70                       Name => "Report.Dat");
 71    end Prepare_Files;
 72 
 73 
 74    -----------------------------------------------------------------------------

 75    procedure Check_Account_Balance
 76          (Customer     : Customer_Rec;           -- Customer account information

 77           Control_File : Text_IO.File_Type) is   -- File for messages

 78 
 79    -- This procedure writes a message to the Error and Control file

 80    -- if the customer has exceeded their credit limit.

 81 
 82    begin   -- Check Account Balance

 83       if Customer.Balance > Customer.Credit_Limit then
 84          Integer_Text_IO.Put (File  => Control_File,
 85                          Item  => Customer.Account_Number,
 86                          Width => 10);
 87          Text_IO.Put (File => Control_File,
 88                       Item => ' ');
 89          Text_IO.Put (File => Control_File,
 90                       Item => Customer.Name.First);
 91          Text_IO.Put (File => Control_File,
 92                       Item => Customer.Name.Last);
 93          Float_Text_IO.Put (File => Control_File,
 94                        Item => Customer.Balance - Customer.Credit_Limit,
 95                        Fore => 8,
 96                        Aft  => 2,
 97                        Exp  => 0);
 98          Text_IO.New_Line (Control_File);
 99       end if;
100    end Check_Account_Balance;
101 
102 
103    -----------------------------------------------------------------------------

104    procedure Process_All_Transactions
105          (Account_Number : in out Positive;          -- The account number being processed

106           Transact_File  : in out Text_IO.File_Type; -- The file of transactions

107           Customer       : in out Customer_Rec) is   -- The account record

108 
109    -- This procedure processes all the transactions for one customer account

110 
111       Choice : Transaction_Type;  -- What kind of transaction

112       Amount : Float;             -- The amount of the transaction

113 
114    begin   -- Process all transactions

115       Transact_Loop:  -- Each iteration, one transaction is processed

116       loop
117          exit Transact_Loop when Account_Number /= Customer.Account_Number;
118          -- Get the kind and amount of transaction

119          Choice_IO.Get (File => Transact_File,  Item => Choice);
120          Float_Text_IO.Get (File => Transact_File, Item => Amount);
121          Text_IO.Skip_Line (Transact_File);                   -- Finish the line

122 
123          -- Modify the customer record

124          case Choice is
125             when Payment =>
126                Customer.Balance := Customer.Balance - Amount;
127             when Charge =>
128                Customer.Balance := Customer.Balance + Amount;
129             when Limit =>
130                Customer.Credit_Limit := Amount;
131          end case;
132 
133          -- Get the account number of the next transaction

134          if not Text_IO.End_Of_File (Transact_File) then
135             Integer_Text_IO.Get (File => Transact_File, Item => Account_Number);
136          else                                -- No more transactions so assign

137             Account_Number := Positive'last; -- an impossible account number

138          end if;
139       end loop Transact_Loop;
140    end Process_All_Transactions;
141 
142 
143 
144 -----------------

145 begin   -- Program Update

146    Prepare_Files (Old_Master  => Old_Master,
147                   New_Master  => New_Master,
148                   Transaction => Transaction,
149                   Report      => Report);
150    -- Priming read of first transaction

151    if not Text_IO.End_Of_File (Transaction) then
152       Integer_Text_IO.Get (File => Transaction, Item => Account_Number);
153    else                                    -- No more transactions so assign

154       Account_Number := Positive'last;     -- an impossible account number

155    end if;
156 
157    -- Process all the accounts in the Old Master file

158    Acct_Loop:  -- Each iteration, one old master account is processed

159    loop
160       exit Acct_Loop when Customer_IO.End_Of_File (Old_Master);
161       -- Get the old account information for the next customer

162       Customer_IO.Read (File => Old_Master, Item => Customer);
163       -- Process all the transactions for this customer

164       Process_All_Transactions (Account_Number => Account_Number,
165                                 Transact_File  => Transaction,
166                                 Customer       => Customer);
167       -- Check to make sure the customer is not over their credit limit

168       Check_Account_Balance (Customer     => Customer,
169                              Control_File => Report);
170       -- Put the updated account information into the new master file

171       Customer_IO.Write (File => New_Master, Item => Customer);
172    end loop Acct_Loop;
173 
174    -- Close all files

175   Customer_IO.Close (Old_Master);
176   Customer_IO.Close (New_Master);
177   Text_IO.Close (Transaction);
178   Text_IO.Close (Report);
179 end Update;
