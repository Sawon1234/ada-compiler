 11    subtype Name_String    is String (1..20);
 12    subtype Address_String is String (1..40);

 25          State  : State_String;

 27       end record;

 29    type Customer_Rec is                -- Customer Account information

 30       record
 31          Account_Number : Positive;
 32          Name           : Name_Rec;
 33          Address        : Address_Rec;
 34          Credit_Limit   : Float;
 35          Balance        : Float;
 36       end record;
 37 
 38    X: array (1..100) of Customer_Rec := (others=>
 39      (1, ("12345678901234567890", "12345678901234567890"), ((others=>'S'),
 40   (others=>'C'), "FL", 1), 1000.0, 5000.0));
 41 
 42    -- Instantiate I/O Packages

 43    package Customer_IO is new Sequential_IO (Element_Type => Customer_Rec);
 44    package Choice_IO   is new Text_IO.Enumeration_IO (Enum => Transaction_Type);
 45 
 46    Old_Master  : Customer_IO.FIle_Type;  -- Old account records

 47 
 48    Account_Number : Positive;     -- The account number of the transaction

 49    Customer       : Customer_Rec; -- The customer record for the current account

 50 
 51 begin
 52    Customer_IO.Create (File => Old_Master,
 53                        Mode => Customer_IO.Out_File,
 54                        Name => "Old_Master.Dat");
 55 
 56    Account_Number := 1_000;
 57    Input: loop
 58       Text_IO.Put ("Enter the first name => ");
 59       -- We must test for EOF after prompt, because it blocks

 60       -- waiting for input, if the input comes from the user.

 61       exit when Text_IO.End_Of_File;
 62       Customer.Account_Number := Account_Number;
 63       Customer_IO.Write (File=>Old_Master, Item=>Customer);
 64       Account_Number := Account_Number + 10;
 65    end loop Input;
 66 
 67    -- Close all files

 68   Customer_IO.Close (Old_Master);
 69 
 70 end Create;
