The Ada Program: question_10a.adb

  1 -- question_10a.adb: understand program without exception handling first
  2 
  3 -- This program is the same as program "Question_10" with all the
  4 -- exception handling removed.  When one first considers a program
  5 -- one should (mentally) ignore the exception handling to understand
  6 -- the normal execution of the program.
  7 
  8 with Ada.Text_IO; use Ada;
  9 
 10 procedure Question_10a is
 11 
 12    procedure One (N: Integer) is
 13    begin
 14       case N is
 15          when 1 => Text_IO.Put_Line ("Procedure 'One': N=1");
 16          when 2 => Text_IO.Put_Line ("Procedure 'One': N=2");
 17          when 3 => Text_IO.Put_Line ("Procedure 'One': N=3");
 18          when others => null;
 19       end case;
 20    end One;
 21 
 22    procedure Two (N: Integer) is
 23    begin
 24       One (N);
 25       Text_IO.Put_Line ("Procedure 'Two' executing normally.");
 26    end Two;
 27 
 28    procedure Three (N: Integer) is
 29    begin
 30       Two (N);
 31       Text_IO.Put_Line ("Procedure 'Three' executing normally.");
 32    end Three;
 33 
 34 begin
 35 
 36    Three (1);  Three (2);  Three (3);
 37    Text_IO.Put_Line ("All done!");
 38 
 39 end Question_10a;
