  1 -- stack.adb: underflow, overflow in stacks make good exceptions
  2 
  3 with Ada.Text_IO; use Ada;
  4 
  5 procedure Stack is
  6 
  7    Underflow, Overflow : exception;
  8 
  9    type Stack_Array is array (Positive range <>) of Character;
 10    type Stack_Type (Max_Size : Positive) is
 11       record
 12          Top      : Natural := 0;
 13          Elements : Stack_Array (1..Max_Size);
 14       end record;
 15 
 16    procedure Push (Stack       : in out Stack_Type;
 17                    New_Element : in     Character   ) is
 18    begin
 19       if Stack.Top = Stack.Max_Size then
 20          raise Overflow;
 21       else
 22          Stack.Top := Stack.Top + 1;
 23          Stack.Elements (Stack.Top) := New_Element;
 24       end if;
 25    end Push;
 26 
 27    procedure Pop (Stack          : in out Stack_Type;
 28                   Popped_Element :    out Character   ) is
 29    begin
 30       if Stack.Top = 0 then
 31          raise Underflow;
 32       else
 33          Popped_Element := Stack.Elements (Stack.Top);
 34          Stack.Top := Stack.Top - 1;
 35       end if;
 36    end Pop;
 37 
 38    The_Stack : Stack_Type (15);
 39    C         : Character;
 40 
 41 begin
 42    Push (The_Stack, 'A');  Push (The_Stack, 'B');
 43    Pop (The_Stack, C); Pop (The_Stack, C); Pop (The_Stack, C);
 44 exception
 45    when Underflow =>
 46       Text_IO.Put_Line ("Program execution interrupted by underflow.");
 47    when Overflow =>
 48       Text_IO.Put_Line ("Program execution interrupted by underflow.");
 49    when others =>
 50       Text_IO.Put_Line ("Program execution interrupted by some exception.");
 51 end Stack;
