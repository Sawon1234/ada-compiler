The Ada Program: over.adb

  1 -- over.adb: catch overflow in multiplication function
  2 
  3 -- With GNAT you need the -gnato option: "gnatmake over -cargs -gnato"
  4 -- to get the (expensive) check for integer overflow, contrary to the
  5 -- philosophy of the Ada programming language and ARM 3.5.4 §20.
  6 
  7 with Ada.Text_IO, Ada.Integer_Text_IO;
  8 use Ada;
  9 
 10 procedure Over is
 11 
 12    -- "fix" multiplication so it does not overflow
 13    function "*" (Left, Right: Integer) return Integer is
 14    begin
 15       return (Standard."*" (Left, Right));
 16    exception
 17       when Constraint_Error => return (Integer'Last);
 18    end "*";
 19 
 20    A, B, Product : Integer := 0;
 21 
 22 begin
 23 
 24    Text_IO.Put ("Enter two integers:  ");
 25    Integer_Text_IO.Get (A);
 26    Integer_Text_IO.Get (B);
 27 
 28    Product := A * B;
 29 
 30    Text_IO.Put ("The product is ");
 31    Integer_Text_IO.Put (Product, Width=>1);
 32    Text_IO.Put_Line (".");
 33 
 34 end Over;
