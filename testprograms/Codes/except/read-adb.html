The Ada Program: read.adb

  1 -- read.adb:  use exception handling in an input validation loop
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Read is
  7 
  8    Min : constant Integer := -10;
  9    Max : constant Integer :=  10;
 10    subtype Small_Integer is Integer range Min .. Max;
 11 
 12    procedure Get (Item: out Small_Integer) is
 13    begin
 14       Try: loop
 15          Read: begin
 16             Text_IO.Put ("Enter an integer between ");
 17             Integer_Text_IO.Put (Small_Integer'First, Width=>1);
 18             Text_IO.Put (" and ");
 19             Integer_Text_IO.Put (Small_Integer'Last, Width=>1);
 20             Text_IO.Put (" ... ");
 21             Integer_Text_IO.Get (Item);
 22 
 23             exit Try;   -- input is valid, so leave
 24 
 25          exception
 26             when Constraint_Error =>
 27                Text_IO.Put_Line ("Value entered is out of range.");
 28             when Text_IO.Data_Error =>
 29                Text_IO.Put_Line ("Value entered is not an integer.");
 30                Text_IO.Skip_Line;  -- discard invalid input
 31          end Read;
 32       end loop Try;
 33    end Get;
 34 
 35    Input_Value : Small_Integer;
 36    Sum         : Integer := 0;
 37 
 38 begin
 39    for Count in 1..5 loop
 40       Get (Input_Value);
 41       Sum := Sum + Input_Value;
 42    end loop;
 43    Text_IO.Put ("The sum is ");
 44    Integer_Text_IO.Put (Sum, Width=>1);
 45    Text_IO.New_Line;
 46 end Read;
