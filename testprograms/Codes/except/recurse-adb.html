  1 -- recurse.adb:  Local exceptions are not local in recursive programs
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Exceptions;
  4 use Ada;
  5 
  6 procedure Recurse is
  7 
  8    procedure P (N: Integer) is
  9       Zero : exception;
 10    begin
 11       if N>0 then
 12          P (N-1);    -- recursive call
 13       else
 14          raise Zero;
 15       end if;
 16    exception
 17       when Zero =>
 18          Text_IO.Put ("Exception 'Zero' caught in procedure P (N=");
 19          Integer_Text_IO.Put (N, Width=>1);
 20          Text_IO.Put_Line (") and propagated.");
 21          raise;
 22       when E: others =>
 23          Text_IO.Put (Exceptions.Exception_Name (E));
 24          Text_IO.Put (" caught in procedure P (N=");
 25          Integer_Text_IO.Put (N, Width=>1);
 26          Text_IO.Put_Line (") and propagated.");
 27          raise;
 28    end P;
 29 
 30 begin
 31 
 32    P (5);
 33 
 34 exception
 35 
 36    when E: others =>
 37       Text_IO.Put (Exceptions.Exception_Name (E));
 38       Text_IO.Put_Line (" raised; giving up.");
 39 
 40 end Recurse;
