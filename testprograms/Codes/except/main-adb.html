  4 use Ada;

  6 procedure Main is
  7 
  8    Bad_Args: exception;
  9 
 10    function Binomial (N,R: Natural) return Positive is
 11       B: Positive := 1;
 12    begin
 13       if (R>N) then
 14          raise Bad_Args;
 15       end if;
 16       for I in R+1 .. N loop
 17          B := B*I;
 18       end loop;
 19       for I in 2 .. N-R loop
 20          B := B/I;
 21       end loop;
 22       return B;
 23    end Binomial;
 24 
 25    X,Y: Integer;
 26 
 27 begin
 28 
 29    Integer_Text_IO.Get (X);
 30    Integer_Text_IO.Get (Y);
 31    Integer_Text_IO.Put (Binomial (X,Y));

 41    when Bad_Args =>
 42       Text_IO.Put_Line ("First number must be greater than second.");

 44 end Main;
