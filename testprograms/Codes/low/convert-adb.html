The Ada Program: convert.adb

  1 -- convert.adb:  convert command line string in hex to number
  2 
  3 with
  4   Ada.Command_Line,     -- access to external execution env (Ada95 A.15)
  5   Ada.Text_IO,          -- usual text (Latin 1) input/output (A.10.1)
  6   Ada.Integer_Text_IO,  -- preinstantiated generic for Integer IO
  7   Ada.Unchecked_Conversion,
  8   Interfaces;
  9 use Ada;
 10 
 11 procedure Convert is
 12    subtype Word is Interfaces.Unsigned_32;
 13    package Word_Text_IO is new Text_IO.Modular_IO (Num=>Word);
 14    Hex : constant String := Command_Line.Argument(1);
 15    U   : Word := Word'Value ("16#" & Hex & "#");
 16 
 17    type Sign_Bit is mod 2**1;
 18    for Sign_Bit'Size use 1;
 19    type Exponent is mod 2*12;
 20    for Exponent'Size use 12;
 21    type Mantissa is mod 2**19;
 22    for Mantissa'Size use 19;
 23 
 24    type IBM_Floating_Point_Representation is
 25       record
 26          Sign : Sign_Bit;
 27          E    : Exponent;
 28          M    : Mantissa;
 29       end record;
 30 
 31    for IBM_Floating_Point_Representation use
 32       record
 33          Sign at 0 range  0 .. 0;
 34          E    at 0 range  1 .. 12;
 35          M    at 0 range 13 .. 31;
 36       end record;
 37 
 38    function To_Record is new Unchecked_Conversion (
 39       Source=>Word, Target=>IBM_Floating_Point_Representation);
 40 
 41    R : IBM_Floating_Point_Representation := To_Record (U);
 42 
 43 begin
 44    Word_Text_IO.Put (Item=>U);
 45    Text_IO.New_Line;
 46 
 47    Integer_Text_IO.Put (Integer(R.E));
 48    Text_IO.New_Line;
 49 
 50    Integer_Text_IO.Put (Integer(R.M));
 51    Text_IO.New_Line;
 52 end Convert;
