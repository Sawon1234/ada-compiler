The Ada Program: picture.adb

  1 -- "Get the Picture", decrypt a binary image from a pair of images
  2 -- ACM Southeast Regional Programming Contest 1999 at Auburn University
  3 
  4 with Ada.Text_IO, Ada.Integer_Text_IO;
  5 use Ada;
  6 
  7 procedure Picture (Rows, Cols: Positive) is
  8    A_Pos: constant Integer := Character'Pos ('A');
  9    Z_Pos: constant Integer := Character'Pos ('0');
 10 
 11    function Hex (C: Character) return Natural is
 12    begin
 13       case C is
 14          when '0'..'9' => return (Character'Pos(C)-Z_Pos);
 15          when 'A'..'F' => return (10+Character'Pos(C)-A_Pos);
 16          when others => raise Constraint_Error;
 17       end case;
 18    end Hex;
 19 
 20    type Image_Type is array (1..Rows,1..Cols) of Boolean;
 21 
 22    procedure Read (Image: out Image_Type) is
 23       Char: Character;
 24       N: Natural;
 25    begin
 26       for R in Image'Range(1) loop
 27          for C in 1..Cols/2 loop
 28             Text_IO.Get (Char);
 29             N := Hex(Char);
 30             Image (R,2*C-1) := N>7;           -- 1 _ X X
 31             Image (R,2*C)   := (N mod 4) > 1; -- X X 1 _
 32          end loop;
 33       end loop;
 34    end Read;
 35 
 36    Image1, Image2:  Image_Type;
 37 
 38 begin
 39    pragma Assert (Cols mod 2 = 0);  -- no of cols will be even
 40    Read (Image1);  Read (Image2);
 41    for R in Image1'Range(1) loop
 42       for C in Image1'Range(2) loop
 43          if Image1(R,C)=Image2(R,C) then
 44             Text_IO.Put ('.');
 45          else
 46             Text_IO.Put ('X');
 47          end if;
 48       end loop;
 49       Text_IO.New_Line;
 50    end loop;
 51 end Picture;
