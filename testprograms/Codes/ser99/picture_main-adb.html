The Ada Program: picture_main.adb

  1 -- Main input loop for "Get the Picture"
  2 -- ACM Southeast Regional Programming Contest 1999 at Auburn University
  3 
  4 with Ada.Text_IO, Ada.Integer_Text_IO, Picture;
  5 use Ada;
  6 
  7 procedure Picture_Main is
  8 
  9    Count    : Natural:=0; -- keep track of the number of input sets
 10    Rows,Cols: Integer;    -- image size
 11 
 12 begin
 13 
 14    Input: loop
 15 
 16       Integer_Text_IO.Get (Rows);
 17       Integer_Text_IO.Get (Cols);
 18       exit when Rows<=0 or else Cols<=0;
 19       Text_IO.Skip_Line;
 20 
 21       Count := Count + 1;
 22 
 23       -- separate the output of each data set with a new line
 24       if Count>1 then Text_IO.New_Line; end if;
 25 
 26       -- each data set gets a heading
 27       Text_IO.Put ("Data set ");
 28       Integer_Text_IO.Put (Count, Width=>0);
 29       Text_IO.Put_Line (":");
 30 
 31       Picture (Rows, Cols);
 32 
 33    end loop Input;
 34 
 35 end Picture_Main;
