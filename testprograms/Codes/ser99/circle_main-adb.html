  1 -- Main input loop for "Crop Circles"
  2 -- ACM Southeast Regional Programming Contest 1999 at Auburn University
  3 
  4 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Float_Text_IO, Circle;
  5 use Ada;
  6 
  7 procedure Circle_Main is
  8 
  9    Count  : Natural:=0;  -- keep track of the number of input sets
 10    Number : Natural;     -- number of circles in an input set
 11 
 12 begin
 13 
 14    Input: loop
 15 
 16       Integer_Text_IO.Get (Number);
 17       exit when Number<=0;
 18       Text_IO.Skip_Line;
 19 
 20       Count := Count + 1;
 21 
 22       -- separate the output of each data set with a new line
 23       if Count>1 then Text_IO.New_Line; end if;
 24 
 25       -- each data set gets a heading
 26       Text_IO.Put ("Data set #");
 27       Integer_Text_IO.Put (Count, Width=>0);
 28       Text_IO.Put (":  Total area = ");
 29       Float_Text_IO.Put (Circle (Number), Exp=>0, Aft=>4, Fore=>1);
 30       Text_IO.Put_Line (" square meters.");
 31 
 32    end loop Input;
 33 
 34 end Circle_Main;
