The Ada Program: circle.adb

  1 -- "Crop Circles", area of interesecting circles
  2 -- ACM Southeast Regional Programming Contest 1999 at Auburn University
  3 
  4 with Ada.Text_IO, Ada.Integer_Text_IO,
  5      Ada.Numerics, Ada.Numerics.Elementary_Functions;
  6 use Ada, Ada.Numerics, Ada.Numerics.Elementary_Functions;
  7 
  8 function Circle (N: Positive) return Float is
  9 
 10    Pi : constant Float := 3.1415926535;  -- Must use this value for pi
 11    Sqrt3 : constant Float := Sqrt (3.0);
 12 
 13    DataX, DataY: array (1..N) of Integer;
 14 
 15    type Dir is (North, South, East, West, NE, SE, NW, SW);
 16 
 17    Adj: array (1..N, Dir) of Boolean := (others => (others => False));
 18 
 19    procedure Read is
 20    begin
 21       for I in DataX'Range loop
 22          Integer_Text_IO.Get (DataX(I));
 23          Integer_Text_IO.Get (DataY(I));
 24          Text_IO.Skip_Line;
 25       end loop;
 26    end Read;
 27 
 28    procedure FillAdj is
 29    begin
 30       for I in DataX'Range loop
 31          for J in DataX'Range loop
 32             if DataX(I)=DataX(J)-1 and then DataY(I)=DataY(J) then
 33                Adj(I,East) := True;
 34                Adj(J,West) := True;
 35             end if;
 36             if DataX(I)=DataX(J) and then DataY(I)=DataY(J)-1 then
 37                Adj(I,North) := True;
 38                Adj(J,South) := True;
 39             end if;
 40             if DataX(I)=DataX(J)-1 and then DataY(I)=DataY(J)-1 then
 41                Adj(I,NE) := True;
 42                Adj(J,SW) := True;
 43             end if;
 44             if DataX(I)=DataX(J)+1 and then DataY(I)=DataY(J)-1 then
 45                Adj(I,SE) := True;
 46                Adj(J,NW) := True;
 47             end if;
 48          end loop;
 49       end loop;
 50    end FillAdj;
 51 
 52    function AreaInter (Q1,Q2,Q3: Boolean := false) return Float is
 53    begin
 54       if (not Q1 and not Q2 and not Q3) then return Pi/4.0;
 55       elsif ((Q1 xor Q3) and (not Q2)) then return Pi/6.0 + Sqrt3/4.0 ;
 56       else return 1.0;
 57       end if;
 58    end AreaInter;
 59 
 60    function AreaInter (C: Integer; Q1,Q2,Q3: Dir) return Float is
 61    begin
 62       return AreaInter (Adj(C,Q1), Adj(C,Q2), Adj(C,Q3));
 63    end AreaInter;
 64 
 65    Area : Float := 0.0;
 66 
 67 begin
 68 
 69    Read;
 70    FillAdj;
 71 
 72    -- Each circle is responsible for contributing its area in each quadrant
 73    -- as long as an adjacent cricle does have priority for that quadrant.
 74    -- The circle with the lowest quadrant number take priority for a quadrant.
 75 
 76    for C in DataX'Range loop
 77       -- compute area in quadrant I
 78       Area := Area + AreaInter (C, East, NE, North);
 79       -- compute area in quadrant II
 80       if (not Adj(C,West)) then
 81          Area := Area + AreaInter (C, North, NW, West);
 82       end if;
 83       -- compute area in quadrant III
 84       if (not Adj(C,SW) and not Adj(C,South)) then
 85          Area := Area + AreaInter (C, West, SW, South);
 86       end if;
 87       -- compute area in quadrant IV
 88       if (not Adj(C,South) and not Adj(C,SE) and not Adj(C,East)) then
 89          Area := Area + Pi/4.0;
 90       end if;
 91    end loop;
 92    return (Area);
 93 end Circle;
