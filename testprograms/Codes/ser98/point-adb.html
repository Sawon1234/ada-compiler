The Ada Program: point.adb

  1 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Float_Text_IO;
  2 use Ada;
  3 
  4 procedure Point (Points: Positive) is
  5    Unit: constant Float := 360.0 / Float(Points);
  6    A_Pos: constant Integer := Character'Pos ('A');
  7 
  8    function P (C: Character) return Float is
  9    begin
 10       return (Float (Character'Pos(C)-A_Pos)*Unit);
 11    end P;
 12 
 13    -- Since we are assured that the compass points are specified in
 14    -- clockwise (CW) order, i.e,
 15    --          Character'Succ (X(X'Last-1)) = X(X'Last)
 16    -- a point equal to X(X'Last) is a positive (CW) contribution and
 17    -- a point equal to X(X'Last-1) is a negative (CCW) contribution.
 18 
 19    function Direction (X: String; N: Positive) return Float is
 20    begin
 21       if N=X'Last-1 and then X(X'Last)='A' then
 22          -- The angle is CCW of 0°, so to avoid negative angles
 23          -- start with 360.0° and don't get 0° from recursive call
 24          return (360.0 - Unit/2.0);
 25       elsif N=X'Last then
 26          return P(X(N));
 27       elsif X(N)=X(X'Last) then
 28          return Direction (X,N+1) + Unit/Float(2**(X'Last-N));
 29       else -- X(N)=X(X'Last-1).
 30          return Direction (X,N+1) - Unit/Float(2**(X'Last-N));
 31       end if;
 32    end Direction;
 33 
 34    Number, Line_Length: Integer;
 35    Buffer: String (1..10);
 36 
 37 begin
 38    Integer_Text_IO.Get (Number);
 39    Text_IO.Skip_Line;
 40    for i in 1..Number loop
 41       Text_IO.Get_Line (Buffer, Line_Length);
 42       Text_IO.Put (Buffer (1..Line_Length) & " : ");
 43       Float_Text_IO.Put (Direction (Buffer(1..Line_Length), 1),
 44                          Exp=>0, Aft=>2);
 45       Text_IO.Put_Line (" degrees");
 46    end loop;
 47 end Point;
