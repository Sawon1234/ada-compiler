The Ada Program: xml.adb

  1 -- xml.adb:  simple xml parser
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Strings.Fixed;
  4 use Ada, Ada.Strings;
  5 
  6 procedure XML is
  7 
  8    Indent: Natural := 0;
  9    procedure Tab (S: String) is
 10    begin
 11       if S(S'Last-1)='/' then
 12          for I in 1..Indent loop Text_IO.Put ('.'); end loop;
 13       elsif S(S'First+1)='/' then
 14          Indent := Indent-3;
 15          for I in 1..Indent loop Text_IO.Put ('.'); end loop;
 16       else
 17          for I in 1..Indent loop Text_IO.Put ('.'); end loop;
 18          Indent := Indent+3;
 19       end if;
 20       Text_IO.Put_Line (S);
 21    end Tab;
 22 
 23    procedure Find (S: String) is
 24       B,E: Natural;
 25    begin
 26       loop
 27          B := Fixed.Index (S, "<");
 28          exit when B=0;
 29          E := Fixed.Index (S, "<");
 30          Tab (S (B..E));
 31       end loop;
 32    end Find;
 33 
 34    Line: String (1..100);
 35    Length: Integer;
 36 
 37 begin
 38 
 39    while not (Text_IO.End_Of_File) loop
 40       Text_IO.Get_Line (Line, Length);
 41       Find (Line (1..Length));
 42    end loop;
 43 
 44 end XML;
