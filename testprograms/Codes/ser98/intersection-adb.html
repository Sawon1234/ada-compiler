  1 -- intersection.adb:  intersection of two line segments
  2 with Ada.Text_IO, Ada.Integer_Text_IO;
  3 use Ada, Ada.Text_IO, Ada.Integer_Text_IO;
  4 
  5 package body Intersection is
  6 
  7    type Line_Equation is record A,B,C: Integer; end record;
  8 
  9    function Equation (L: Segment) return Line_Equation is
 10       C: constant Integer := L.Stop.X*L.Start.Y - L.Start.X*L.Stop.Y;
 11    begin
 12       return (A=>L.Stop.Y-L.Start.Y,B=>L.Start.X-L.Stop.X,C=>C);
 13    end Equation;
 14 
 15    function Residue (P: Point; E: Line_Equation) return Integer is
 16    begin
 17       return (E.A*P.X + E.B*P.Y + E.C);
 18    end;
 19 
 20    function Collinear (EL,EM: Line_Equation) return Boolean is
 21    begin
 22       return (EL.A*EM.B = EM.A*EL.B);
 23    end Collinear;
 24 
 25    function Disjoint (L,M: Segment) return Boolean is
 26       function Disjoint (X,Y,Z,W: Integer) return Boolean is
 27       begin
 28          return Integer'Min(Z,W) >= Integer'Max(X,Y) or else
 29            Integer'Max(Z,W) <= Integer'Min(X,Y);
 30       end Disjoint;
 31    begin
 32       return Disjoint(L.Start.X, L.Stop.X, M.Start.X, M.Stop.X) and then
 33         Disjoint (L.Start.Y, L.Stop.Y, M.Start.Y, M.Stop.Y);
 34    end Disjoint;
 35 
 36    function Intersect (L,M: Segment) return Kind is
 37       EL: constant Line_Equation := Equation (L);
 38       EM: constant Line_Equation := Equation (M);
 39       R1: constant Integer := Residue (M.Start, EL);
 40       R2: constant Integer := Residue (M.Stop, EL);
 41       R3: constant Integer := Residue (L.Start, EM);
 42       R4: constant Integer := Residue (L.Stop, EM);
 43    begin
 44       if L.Start=L.Stop or else M.Start=M.Stop then
 45          return Degenerate;
 46       end if;
 47 
 48       if L.Start=M.Start or else L.Start=M.Stop or else
 49         L.Stop=M.Start or else L.Stop=M.Stop then
 50          return Endpoint;
 51       end if;
 52 
 53       if (R1<0 and then R2<0) or else (R1>0 and then R2>0) then
 54          -- line segment M on same half plane formed by line L
 55          return No_Intersection;
 56       elsif (R3<0 and then R4<0) or else (R3>0 and then R4>0) then
 57          -- line segment L on same half plane formed by line M
 58          return No_Intersection;
 59       end if;
 60 
 61       -- L, M must now intersect (unless collinear and non-overlapping).
 62 
 63       -- L, M are collinear if both endpoints have 0 residue
 64       if (R1=0 and then R2=0) then
 65          pragma Assert (R3=0 and then R4=0);
 66          if not Disjoint (L,M) then
 67             return Overlaps;    -- collinear and intersecting
 68          else
 69             return Collinear;   -- but not overlapping
 70          end if;
 71       end if;
 72 
 73       if (R1=0 or else R2=0 or else R3=0 or else R4=0) then
 74          return Abuts;
 75       end if;
 76 
 77       return Intersects;
 78 
 79    end Intersect;
 80 end Intersection;
