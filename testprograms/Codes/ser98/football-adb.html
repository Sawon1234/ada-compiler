The Ada Program: football.adb

  1 -- football.adb:  simulated game of simplified US football
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Football is
  7    subtype Team is Boolean;  -- true iff home
  8    Home   : constant Boolean := True;
  9    Visitor: constant Boolean := False;
 10    Score : array (Team, 1..4) of Natural := (others=>(0,0,0,0));
 11 
 12    -- These values persist over the end of the 1 and 3 quarters
 13    Ball: Team;
 14    Position: Integer;  -- from the perspective of team with ball
 15    Down, FDM: Natural;
 16 
 17    procedure First_Down (B: Team:= Ball; P: Natural := Position) is
 18    begin
 19       Ball := B; Down:= 1; Position:=P; FDM:=P+10;
 20    end First_Down;
 21 
 22    procedure Quarter (Q: Natural) is
 23       Yardage: Integer;
 24    begin
 25 
 26       if Q=1 then
 27          First_Down (Visitor, 20);
 28       elsif Q=3 then
 29          First_Down (Home, 20);
 30       end if;
 31 
 32       loop
 33          Integer_Text_IO.Get (Yardage);
 34          exit when Yardage>100;
 35          Position := Position + Yardage;
 36          if (Position >= 100) then
 37             Score(Ball, Q) := Score(Ball, Q) + 6;  -- touchdown
 38             Integer_Text_IO.Get (Yardage);
 39             if (Yardage > 2) then
 40                Score(Ball, Q) := Score(Ball, Q) + 1;  -- PAT
 41             end if;
 42             First_Down (not Ball, 20);
 43          elsif Position <= 0 then
 44             First_Down (not Ball, 50);
 45             Score(Ball, Q) := Score(Ball, Q) + 2;  -- safety
 46 
 47          elsif Position >= FDM then     -- first down
 48             First_Down;
 49          elsif Down=4 then    -- change of possesion
 50             First_Down (not Ball, 100-Position);
 51          else
 52             Down := Down + 1;
 53          end if;
 54       end loop;
 55    end Quarter;
 56 
 57    Total: Natural;
 58 
 59 begin
 60 
 61    for Q in 1..4 loop
 62       Quarter (Q);
 63    end loop;
 64 
 65    Text_IO.Set_Col (16);  Text_IO.Put_Line ("Quarter");
 66    Text_IO.Set_Col (12);  Text_IO.Put_Line ("1   2   3   4   Final");
 67 
 68    Text_IO.Put ("Home    ");
 69    Total := 0;
 70    for Q in 1..4 loop
 71       Total := Total + Score (True, Q);
 72       Integer_Text_IO.Put (Score (Home, Q), Width=>4);
 73    end loop;
 74    Integer_Text_IO.Put (Total, Width=>8);
 75    Text_IO.New_Line;
 76    Text_IO.Put ("Visitor ");
 77    Total := 0;
 78    for Q in 1..4 loop
 79       Total := Total + Score (False, Q);
 80       Integer_Text_IO.Put (Score (Visitor, Q), Width=>4);
 81    end loop;
 82    Integer_Text_IO.Put (Total, Width=>8);
 83    Text_IO.New_Line;
 84 
 85 end Football;
