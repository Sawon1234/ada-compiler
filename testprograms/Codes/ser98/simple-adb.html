The Ada Program: simple.adb

  1 -- simple.adb: determine if input sets are simple polygons
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO, Intersection;
  4 use Ada, Intersection;
  5 
  6 procedure Simple is
  7 
  8    procedure Get (P: out Point) is
  9    begin
 10       Integer_Text_IO.Get (P.X);
 11       Integer_Text_IO.Get (P.Y);
 12       Text_IO.Skip_Line;
 13    end Get;
 14 
 15    type Point_Array is array (Positive range <>) of Point;
 16 
 17    -- We assume V(V'Last)=V(V'First), so the number of vertices,
 18    -- N, in the polygon is V'Length-1.  The edges of the polygon are:
 19    --    1.  V'First     V(V'First)   .. V(V'First+1)
 20    --    2.  V'First+1   V(V'First+1) .. V(V'First+2)
 21    --   ...
 22    --  N-1.  V'Last-2    V(V'Last-2)  .. V(V'Last-1
 23    --    N.  V'Last-1    V(V'Last-1)  .. V(V'Last)=V(V'First)
 24 
 25    function Simple_Polygon (V: Point_Array) return Boolean is
 26 
 27       function Edge (I: Positive) return Segment is
 28       begin
 29          return Segment' (V(I), V(I+1));
 30       end Edge;
 31 
 32    begin
 33       for I in V'First .. V'Last-3 loop
 34          -- To avoid duplication compare against higher numbered edges.
 35          -- Also, avoid comparing against the two adjacent edges which
 36          -- naturally coincide at one endpoint.
 37          for J in I+2 .. Integer'Min(V'Last-1,I+V'Length-3) loop
 38             case Intersect (Edge(I), Edge(J)) is
 39                when Intersects => return False;
 40                when Abuts => null;
 41                when others => null;
 42             end case;
 43          end loop;
 44       end loop;
 45       return (True);
 46    end Simple_Polygon;
 47 
 48    Count   : Natural:=0;      -- keep track of the number of input sets
 49    Number  : Integer;         -- number of vertices in the polygon
 50 
 51    -- Maximum number of vertices, equals the maximum number of edges
 52    Max     : constant Positive := 100;
 53 
 54    Vertices: Point_Array(1..Max+1);  -- one extra point needed
 55 
 56 begin
 57 
 58    Input: loop
 59 
 60       Number := 1;
 61 
 62       V: loop
 63 
 64          Get (Vertices(Number));
 65          exit Input when Vertices(Number)=Point'(0,0);
 66          exit V when Number>1 and then Vertices(Number)=Vertices(1);
 67          Number := Number + 1;
 68 
 69       end loop V;
 70 
 71       Count := Count + 1;
 72 
 73       Text_IO.Put ("Polygon #");
 74       Integer_Text_IO.Put (Count, Width=>0);
 75 
 76       if Simple_Polygon (Vertices(1..Number)) then
 77          Text_IO.Put_Line (":  SIMPLE");
 78       else
 79          Text_IO.Put_Line (":  COMPLEX");
 80       end if;
 81 
 82    end loop Input;
 83 
 84 end Simple;
