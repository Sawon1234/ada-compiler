  3 procedure Find_Next (Next: in out Integer; Data: Data_Array) is
  4    Curr: Integer := Next;
  5    Short: Float := 1.0E15;  -- Shortest distance so far
  6 begin
  7    for I in Data'Range loop
  8       if I/=Curr and not Data(I).Drilled  then
  9          declare
 10             D: constant  Float := Distance (Data(I), Data(Curr));
 11          begin
 12             if (D<Short) then
 13                Short := D;
 14                Next := I;
 15             end if;
 16          end;
 17       end if;
 18    end loop;
 19 end Find_Next;
