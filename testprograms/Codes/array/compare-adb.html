The Ada Program: compare.adb

  1 -- compare.adb:  comparing arrays and array slices
  2 
  3 with Ada.Text_IO;
  4 use Ada.Text_IO;
  5 
  6 procedure Compare is
  7 begin
  8    declare
  9       type Integer_Array is array (Positive range <>) of Integer;
 10       S: Integer_Array := (31, 32, 33, 34, 35);
 11       L: Integer_Array := (21, 22, 23, 24, 25, 31, 32, 33, 34, 35, 41);
 12    begin
 13       Put_Line (Boolean'Image (  S       = L         ));   -- false
 14       Put_Line (Boolean'Image (  S       < L         ));   -- false
 15       Put_Line (Boolean'Image (  S       = L(1..5)   ));   -- false
 16       Put_Line (Boolean'Image (  S       = L(6..10)  ));   -- true
 17       Put_Line (Boolean'Image (  S(1..4) < L(6..10)  ));   -- true
 18       Put_Line (Boolean'Image (  S(5..1) < L         ));   -- true
 19    end;
 20 
 21    declare
 22       type Month is (Jan,Feb,Mar,Apr,May,Jun,Jul,Aug,Sep,Oct,Nov,Dec);
 23       type Month_Array is array (Positive range <>) of Month;
 24       S: Month_Array := (Jun,Jul,Aug,Sep,Oct);
 25       L: Month_Array := (Jan,Feb,Mar,Apr,May,Jun,Jul,Aug,Sep,Oct,Dec);
 26    begin
 27       Put_Line (Boolean'Image (  S       = L         ));   -- false
 28       Put_Line (Boolean'Image (  S       < L         ));   -- false
 29       Put_Line (Boolean'Image (  S       = L(1..5)   ));   -- false
 30       Put_Line (Boolean'Image (  S       = L(6..10)  ));   -- true
 31       Put_Line (Boolean'Image (  S(1..4) < L(6..10)  ));   -- true
 32       Put_Line (Boolean'Image (  S(5..1) < L         ));   -- true
 33    end;
 34 
 35    declare
 36       S: String := "PQRST";
 37       L: String := "ABCDEPQRSTZ";
 38    begin
 39       Put_Line (Boolean'Image (  S       = L         ));   -- false
 40       Put_Line (Boolean'Image (  S       < L         ));   -- false
 41       Put_Line (Boolean'Image (  S       = L(1..5)   ));   -- false
 42       Put_Line (Boolean'Image (  S       = L(6..10)  ));   -- true
 43       Put_Line (Boolean'Image (  S(1..4) < L(6..10)  ));   -- true
 44       Put_Line (Boolean'Image (  S(5..1) < L         ));   -- true
 45    end;
 46 end Compare;
