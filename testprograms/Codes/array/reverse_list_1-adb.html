  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;

  6 procedure Reverse_List_1 is
  7 
  8    type Value_Array_Type is array (1..10) of Integer;
  9 
 10    Value : Value_Array_Type;

 13
