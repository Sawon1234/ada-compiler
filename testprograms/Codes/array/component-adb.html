The Ada Program: component.adb

  1 -- component.adb:  componentwise operations on boolean arrays
  2 
  3 -- Only if the component type is boolean are there any
  4 -- componentwise operations predefined (ARM 3.6.2).
  5 
  6 procedure Component is
  7 
  8    type T is array (1..100) of Boolean;
  9 
 10    A, B, C: T;
 11 
 12 begin
 13 
 14    A := B xor (C and not (A or B));
 15 
 16 end Component;
