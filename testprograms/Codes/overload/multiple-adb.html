The Ada Program: multiple.adb

  1 -- multiple.adb: illustrate overloaded subprocedures
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Multiple is
  7 
  8    procedure Put (Item: Integer; By: in Natural:=1) is
  9    begin
 10       for I in 1..By loop
 11          Integer_Text_IO.Put (Item, Width=>7);
 12       end loop;
 13    end Put;
 14 
 15    procedure Put (Item_1, Item_2: Integer; By: in Natural:=1) is
 16    begin
 17       for I in 1..By loop
 18          Integer_Text_IO.Put (Item_1, Width=>7);
 19          Integer_Text_IO.Put (Item_2, Width=>7);
 20       end loop;
 21    end Put;
 22 
 23    procedure Put (Item_1, Item_2, Item_3: Integer; By: in Natural:=1) is
 24    begin
 25       for I in 1..By loop
 26          Integer_Text_IO.Put (Item_1, Width=>7);
 27          Integer_Text_IO.Put (Item_2, Width=>7);
 28          Integer_Text_IO.Put (Item_3, Width=>7);
 29       end loop;
 30    end Put;
 31 
 32    procedure Put (Item_1, Item_2: String; By: in Natural:=1) is
 33    begin
 34       for I in 1..By loop
 35          Text_IO.Put (Item_1);
 36          Text_IO.Put (Item_2);
 37       end loop;
 38    end Put;
 39 
 40 begin
 41    Put (" ******", " ------", 3);  Text_IO.New_Line;
 42    Put (1,2,Item_3=>3); Put (4,5,Item_3=>6); Text_IO.New_Line;
 43    Put (1998, By=>6);              Text_IO.New_Line;
 44    Put (1998, 1999, By=>3);        Text_IO.New_Line;
 45    Put (1998, 1999, 2000, By=>2);  Text_IO.New_Line;
 46    Put (" ******", " ------", 3);  Text_IO.New_Line;
 47 end Multiple;
