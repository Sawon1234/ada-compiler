  1 --  overload.adb:  demonstrate overloading in Ada
  2 
  3 procedure Overload is
  4 
  5    Integer_Result : Integer;
  6    Float_Result   : Float;
  7 
  8    --- The float average of two integers
  9    procedure Calc_Average (Value_1, Value_2: in Integer;
 10                            Result: out Float) is
 11    begin
 12       Result := Float (Value_1 + Value_2) / 2.0;
 13    end Calc_Average;
 14 
 15    --- The integer average of two integers
 16    procedure Calc_Average (Value_1, Value_2: in Integer;
 17                            Result: out Integer) is
 18    begin
 19       Result := (Value_1 + Value_2) / 2;
 20    end Calc_Average;
 21 
 22    --- The float average of three integers
 23    procedure Calc_Average (Value_1, Value_2, Value_3: in Integer;
 24                            Result: out Float) is
 25    begin
 26       Result := Float (Value_1 + Value_2 + Value_3) / 3.0;
 27    end Calc_Average;
 28 
 29 begin -- Overload
 30 
 31    Calc_Average (33, 86, Integer_Result);
 32    Calc_Average (33, 86, Float_Result);
 33    Calc_Average (33, 54, 63, Float_Result);
 34 
 35 end Overload;
