  3 procedure Largest is
  4 
  5    --  version 1:  Integer, Integer --> Integer
  6    function Magnitude (I1, I2: Integer) return Integer is
  7    begin
  8       if abs I1 > abs I2 then
  9          return I1;
 10       else
 11          return I2;
 12       end if;
 13    end Magnitude;
 14 
 15    --  version 2:  Integer, Float --> Float
 16    function Magnitude (I1: Integer;  F2: Float) return Float is
 17       F1: constant Float := Float(I1);
 18    begin
 19       if abs F1 > abs F2 then
 20          return F1;
 21       else
 22          return F2;
 23       end if;
 24    end Magnitude;
 25 
 26    --  version 3:  Float, Integer --> Float
 27    function Magnitude (F1: Float; I2: Integer) return Float is
 28       F2: constant Float := Float(I2);
 29    begin
 30       if abs F1 > abs F2 then
 31          return F1;
 32       else
 33          return F2;
 34       end if;
 35    end Magnitude;
 36 
 37    --  version 4:  Float, Float --> Float
 38    function Magnitude (F1, F2: Float) return Float is
 39    begin
 40       if abs F1 > abs F2 then
 41          return F1;
 42       else
 43          return F2;
 44       end if;
 45    end Magnitude;
 46 
 47    --  version 5:  Integer, Integer --> Float
 48    function Magnitude (I1, I2: Integer) return Float is
 49    begin
 50       if abs I1 > abs I2 then
 51          return Float(I1);
 52       else
 53          return Float(I2);
 54       end if;
 55    end Magnitude;
 56 
 57    --  version 6:  Float, Float, Float --> Float
 58    function Magnitude (F1, F2, F3: Float) return Float is
 59    begin
 60       return Magnitude (F1, Magnitude (F2, F3));
 61    end Magnitude;
 62 
 63    N1, N2, N3: Integer := 0;
 64    X1, X2, X3: Float   := 0.0;
 65 
 66 begin
 67 
 68    X1 := Magnitude (N2, N3);
 69    X1 := Magnitude (N1, Magnitude (X2, N2));
 70    X1 := Magnitude (X2, Magnitude (N1, N2), X3);
 71 
 72    -----------------------------------------------
 73    --  four different ways of disambiguating:
 74    --    X1 := Magnitude (X2, Magnitude (N1, N2));
 75 
 76    X1 := Magnitude (X2, Float'(Magnitude (N1, N2)));
 77    X1 := Magnitude (X2, Integer'(Magnitude (N1, N2)));
 78    X1 := Magnitude (X2, I2 => Magnitude (N1, N2));
 79    X1 := Magnitude (X2, F2 => Magnitude (N1, N2));
 80 
 81 end Largest;
