The Ada Program: line.adb

  1 -- line.adb:  compute equation for line given two points
  2 
  3 with Ada, Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Line is
  7 
  8    type Point is record X,Y: Integer; end record;
  9 
 10    type Segment is record Start, Stop: Point; end record;
 11 
 12    -- A line in 2D is described by the equation Ax+By+C=0.
 13    type Line_Equation is record A,B,C: Integer; end record;
 14 
 15    function Equation (L: Segment) return Line_Equation is
 16       A: constant Integer := L.Stop.Y-L.Start.Y;
 17       B: constant Integer := L.Start.X-L.Stop.X;
 18       C: constant Integer := L.Stop.X*L.Start.Y - L.Start.X*L.Stop.Y;
 19    begin
 20       return Line_Equation'(A,B,C);
 21    end Equation;
 22 
 23    L: Segment := Segment' (Point' (1,3), Point' (5, -3));
 24    E: Line_Equation := Equation (L);
 25 
 26 begin
 27 
 28    Integer_Text_IO.Put (E.A, Width=>0);
 29    Text_IO.Put ("x + ");
 30    Integer_Text_IO.Put (E.B, Width=>0);
 31    Text_IO.Put ("y + ");
 32    Integer_Text_IO.Put (E.C, Width=>0);
 33    Text_IO.Put_Line (" = 0");
 34 
 35 end Line;
