  1 -- records.adb:  examples of hierarchical records, arrays of records
  2 
  3 procedure Records is
  4 
  5    type Month_Name is (Jan, Feb, Mar, Apr, May, Jun,
  6                        Jul, Aug, Sep, Oct, Nov, Dec);
  7 
  8    type Date is
  9       record
 10          Day  : Integer range 1 .. 31;
 11          Month: Month_Name;
 12          Year : Integer;
 13       end record;
 14 
 15    D: Date;
 16 
 17    type Person is
 18       record
 19          Birth: Date;
 20          Name : String (1..16) := (others => ' ');
 21       end record;
 22 
 23    John  : Person;
 24    People: array (1..100) of Person;
 25 
 26 begin
 27 
 28    D.Day := 4;
 29    D.Month := Jul;
 30    D.Year := 1776;
 31 
 32    D := (4, Jul, 1776);
 33    D := Date'(4, Jul, 1776);
 34    D := Date'(Month=>Jul, Day=>4, Year=>1776);
 35 
 36    John := Person'(Date'(19, Aug, 1937), "John            ");
 37    John := Person'(Birth=>Date'(19, Aug, 1937), others=>"John            ");
 38 
 39    People(6).Birth.Day := 19;
 40    People(8).Name(3)   := 'H';
 41 
 42 end Records;
