  1 -- simple_records.adb:  examples of records for complex numbers and dates
  2 
  3 procedure Simple_Records is
  4 
  5    type Complex is
  6       record
  7          Real, Imaginary : Float := 0.0;
  8       end record;
  9 
 10    type Month_Name is (Jan, Feb, Mar, Apr, May, Jun,
 11                        Jul, Aug, Sep, Oct, Nov, Dec);
 12    type Date is
 13       record
 14          Day: Integer range 1 .. 31;
 15          Month: Month_Name;
 16          Year: Integer;
 17       end record;
 18 
 19    C: Complex;   -- gets default value (0.0, 0.0)
 20    D: Date;
 21 
 22 begin
 23 
 24    if C.Real<0.0 then
 25       C.Real := -C.Real;
 26    end if;
 27 
 28    if C = Complex'(Real=>1.0, Imaginary=>0.0) then
 29       C := Complex'(Real=>0.0, Imaginary=>1.0);
 30    end if;
 31 
 32    D.Day := 4;
 33    D.Month := Jul;
 34    D.Year := 1776;
 35 
 36    D := (4, Jul, 1776);
 37    D := Date'(4, Jul, 1776);
 38    D := Date'(Month=>Jul, Day=>4, Year=>1776);
 39 
 40 end Simple_Records;
