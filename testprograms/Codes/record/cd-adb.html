  3 with Ada.Text_IO;
  4 use Ada;
  5 
  6 procedure CD is
  7 
  8    subtype String20 is String (1..20);
  9    subtype Year_Range is Integer range 1900..2990;
 10 
 11    type Disc is record
 12       Artist, Album: String20;
 13       Year: Year_Range;
 14    end record;
 15 
 16    CD1, CD2: Disc;
 17 
 18 begin
 19 
 20    CD1 := Disc'(
 21       Album =>"Freewheelin'        ",
 22       Artist=>"Bob Dylan           ",
 23       Year  =>1963);
 24 
 25    CD2 := Disc'(
 26       Album =>"Highway 61 Revisited",
 27       Artist=>"Bob Dylan           ",
 28       Year  =>1965);
 29 
 30    if CD1.Artist=CD2.Artist then
 31       Text_IO.Put_Line ("Yes");
 32    else
 33       Text_IO.Put_Line ("No");
 34    end if;
 35 
 36 end CD;
