The Ada Program: hobo.adb

  1 -- hobo.adb: "Lazy Hobo Riddle" from Cohoon & Davidson
  2 --           C++ Programming Design, page 184.
  3 
  4 with Ada.Text_IO, Ada.Integer_Text_IO;
  5 use Ada;
  6 
  7 procedure Hobo is
  8 
  9   Sum   : constant Integer := 200;
 10   Limit : constant Integer := 14;  --  Limit**2+1+1+1<=Sum
 11 
 12 begin
 13 
 14    -- Find I, J, K, L all different
 15    -- such that I**2 + J**2 + K**2 + L**2 = 200
 16 
 17    -- Without loss of generality assume: I < J < K < L
 18 
 19    for I in 1..Limit loop
 20       for J in I+1..Limit loop
 21          for K in J+1..Limit loop
 22             for L in K+1..Limit loop
 23                if I*I+J*J+K*K+L*L = Sum then
 24                   Integer_Text_IO.Put (I);
 25                   Integer_Text_IO.Put (J);
 26                   Integer_Text_IO.Put (K);
 27                   Integer_Text_IO.Put (L);
 28                   Text_IO.New_Line;
 29                end if;
 30             end loop;
 31          end loop;
 32       end loop;
 33    end loop;
 34 
 35 end Hobo;
