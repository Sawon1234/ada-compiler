  1 -- overload.adb:  an example of overloaded procedures
  2 
  3 procedure Overload is
  4    Integer_Result: Integer;
  5    Float_Result:   Float;
  6 
  7    --- The float average of two integers
  8    procedure Calc_Average (Value_1, Value_2: in Integer;
  9                            Result: out Float) is
 10    begin
 11        Result := Float (Value_1 + Value_2) / 2.0;
 12    end Calc_Average;
 13 
 14    --- The integer average of two integers
 15    procedure Calc_Average (Value_1, Value_2: in Integer;
 16                            Result: out Integer) is
 17    begin
 18        Result := (Value_1 + Value_2) / 2;
 19    end Calc_Average;
 20 
 21    --- The float average of three integers
 22    procedure Calc_Average (Value_1, Value_2, Value_3: in Integer;
 23                            Result: out Float) is
 24    begin
 25        Result := Float (Value_1 + Value_2 + Value_3) / 3.0;
 26    end Calc_Average;
 27 
 28 begin
 29 
 30    Calc_Average (33, 86, Integer_Result);
 31    Calc_Average (33, 86, Float_Result);
 32    Calc_Average (33, 54, 63, Float_Result);
 33 
 34 end Overload;
