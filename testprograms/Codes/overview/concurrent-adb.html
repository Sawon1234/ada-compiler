  3 with Ada.Text_IO; use Ada.Text_IO;

  5 procedure Concurrent is
  6     task T1;

  8     task body T1 is begin Put ('A'); Put ('B'); end T1;

 11     Put ('E'); Put ('F');

 13         task T3;
 14         task body T3 is begin Put ('G'); Put ('H'); end T3;
 15     begin
 16         Put ('I'); Put ('J');

 18     Put ('K'); Put ('L');
 19 end Concurrent;
