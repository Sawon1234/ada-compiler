The Ada Program: rend.adb

  1 -- rend.adb:  dynamic task creation, subprograms, and rendezvous
  2 
  3 with Ada.Text_IO; use Ada.Text_IO;
  4 
  5 procedure Rend is
  6     task type TT;
  7     type PtrTT is access TT;
  8     X : PtrTT;      -- pointer to task type TT
  9     Y : PtrTT;      -- pointer to task type TT
 10     procedure S is  -- declaration of subprog S
 11         task T is entry Z; end T;
 12         task body T is
 13         begin
 14             Put ('1');
 15             accept Z;
 16         end T;
 17     begin
 18         T.Z;        -- entry call
 19     end S;
 20     task body TT is begin Put ('2'); Put ('3'); end TT;
 21 begin
 22     X := new TT;    -- task of type TT begins
 23     Rend.S;         -- call to subprogram S
 24     Put ('4');
 25     Y := new TT;    -- task of type TT begins
 26     Rend.S;         -- call to subprogram S
 27 end Rend;
