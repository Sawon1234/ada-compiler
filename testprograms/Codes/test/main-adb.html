The Ada Program: main.adb

  1 -- main.adb:  a while loop that divides an interval in thirds
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Main is
  7     Low : Integer := -10;
  8     High: Integer :=   8;
  9     Step: Integer;
 10 begin
 11     while (Low <= High) loop
 12         Step := (High-Low)/3 + 1;
 13         -- since High-Low>=0, Step>0
 14         Low  := Low  + Step;
 15         High := High - Step;
 16         Integer_Text_IO.Put (Low);
 17         Integer_Text_IO.Put (High);
 18         Integer_Text_IO.Put (Step);
 19         Text_IO.New_Line;
 20     end loop;
 21 end Main;
