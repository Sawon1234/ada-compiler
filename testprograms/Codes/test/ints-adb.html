The Ada Program: ints.adb

  1 -- ints.adb:  read and write integers
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Ints is
  7    I: Integer;
  8 begin
  9 
 10    Integer_Text_IO.Put (Item=>4);            -- print int 4
 11    Integer_Text_IO.Put (Item=>7, Width=>4);  -- print int 7
 12    Integer_Text_IO.Put (Item=>17);           -- print int 17
 13    Text_IO.New_Line;                         -- print a new line
 14 
 15    Integer_Text_IO.Get (Item=>I);       -- get int from keyboard
 16    Integer_Text_IO.Put (Item=>I);       -- print the int
 17    Text_IO.New_Line;
 18 
 19 end Ints;
