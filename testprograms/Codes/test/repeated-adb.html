The Ada Program: repeated.adb

  1 -- repeated.adb:  get integers until one of them equals the previous one
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Repeated (Duplicate: out Integer) is
  7    Previous:  Integer;
  8 begin
  9    Integer_Text_IO.Get (Previous);
 10    loop
 11       Integer_Text_IO.Get (Duplicate);
 12       exit when Previous = Duplicate;
 13       Previous := Duplicate;
 14    end loop;
 15 end Repeated;
