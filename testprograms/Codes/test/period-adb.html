  3 with Ada.Numerics, Ada.Numerics.Elementary_Functions;
  4 use Ada, Ada.Numerics;

  6 function Period (L, G: Float) return Float is
  7    Two_Pi   : constant       := 2 * Pi;   -- universal real
  8    Quotient : constant Float := L / G;
  9 begin
 10    return (Two_Pi * Elementary_Functions.Sqrt (Quotient));
 11 end Period;
