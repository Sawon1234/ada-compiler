The Ada Program: random.adb

  1 -- A crude source of random numbers 0 .. 2**12 - 1
  2 package body Random is
  3 
  4    type T is mod 2 ** 12;
  5 
  6    Seed : T;
  7    C: constant := 5 ** 5;
  8 
  9    procedure Init (N: Integer) is
 10    begin
 11      if N mod 2 = 0 then
 12         raise Bad;
 13      end if;
 14      Seed := T(N);
 15    end Init;
 16 
 17 
 18    function Next return Integer is
 19    begin
 20       Seed := Seed * C;
 21       return (Integer (Seed));
 22    end Next;
 23 
 24 end Random;
