  1 -- queues.adb:  queue package using pointers
  2 
  3 with Ada.Unchecked_Deallocation;
  4 use Ada;
  5 
  6 package body Queues is
  7 
  8    procedure Free_Queue_Element is
  9      new Unchecked_Deallocation(Queue_Element_Type,Queue_Element_Pointer);
 10 
 11    procedure Enqueue (Item: in Integer; Queue: in out Queue_Type) is
 12       New_Element : Queue_Element_Pointer;
 13    begin
 14       New_Element := new Queue_Element_Type' (Item, null);
 15       if Queue.Length = 0 then
 16          Queue.Front := New_Element;
 17       else
 18          Queue.Back.Link := New_Element;
 19       end if;
 20       Queue.Back := New_Element;
 21       Queue.Length := Queue.Length + 1;
 22    end Enqueue;
 23 
 24    procedure Dequeue (Item: out Integer; Queue: in out Queue_Type) is
 25       Old_Front: Queue_Element_Pointer;
 26    begin
 27       if Queue.Length > 0 then
 28          Old_Front := Queue.Front;
 29          Queue.Front := Old_Front.Link;
 30          Item := Old_Front.Data;
 31          Free_Queue_Element (Old_Front);  -- deallocate storage!
 32          Queue.Length := Queue.Length - 1;
 33       else
 34          raise Empty_Queue_Error;
 35       end if;
 36    end Dequeue;
 37 
 38    function Length (Queue: Queue_Type) return Natural is
 39    begin
 40       return Queue.Length;
 41    end Length;
 42 
 43 end Queues;
