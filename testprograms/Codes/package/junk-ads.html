The Ada Program: junk.ads

  1 -- junk.ads:  the specification of the "Junk" package
  2 
  3 -- One can have types, subtypes, objects, constants,
  4 -- exceptions, subprograms, and functions in a package.
  5 
  6 package Junk is
  7 
  8    type T is (Red, Green, Blue);     -- a type declaration
  9 
 10    subtype S is T range Red..Green;  -- a subtype declaration
 11 
 12    N: Integer;                       -- an object declaration
 13 
 14    C: constant Float := 3.14159;     -- a constant declaration
 15 
 16    E: exception;                     -- an exception declaration
 17 
 18    procedure Sub;                    -- a subprocedure declaration
 19 
 20    function Fun return Boolean;      -- a function declaration
 21 
 22 end Junk;
