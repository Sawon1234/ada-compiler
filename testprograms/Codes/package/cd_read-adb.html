  1 -- cd_read.adb:  read records from binary file
  2 
  3 with CD_Package;
  4 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Sequential_IO;
  5 use Ada;
  6 
  7 procedure CD_Read is
  8 
  9    package CD_Bin_IO is new Sequential_IO (CD_Package.Track_Record);
 10    File      : CD_Bin_IO.File_Type;
 11    File_Name : constant String := "disk.data";
 12    Song      : CD_Package.Track_Record;
 13 
 14 begin
 15 
 16    CD_Bin_IO.Open (File, Mode=>CD_Bin_IO.In_File, Name=>File_Name);
 17    while not (CD_Bin_IO.End_Of_File (File)) loop
 18       CD_Bin_IO.Read (File=>File, Item=>Song);
 19       Text_IO.Put_Line (Item=>Song.Title);
 20    end loop;
 21    CD_Bin_IO.Close (File=>File);
 22 
 23 end CD_Read;
