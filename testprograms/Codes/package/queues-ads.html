  1 -- queues.ads:  queue package using pointers
  2 
  3 package Queues is
  4 
  5    -- Queue is limited because
  6    --    1)  Predefined "=" operator makes little sense.
  7    --    2)  Assignment copies pointers, so is dangerous.
  8    --    3)  Guarantees all initial values are consistent.
  9    type Queue_Type is limited private;
 10 
 11    procedure Enqueue (Item: in Integer; Queue: in out Queue_Type);
 12    procedure Dequeue (Item: out Integer; Queue: in out Queue_Type);
 13    function Length (Queue: Queue_Type) return Natural;
 14    Empty_Queue_Error: exception;
 15 
 16 private
 17 
 18    type Queue_Element_Type;  -- forward declaration
 19    type Queue_Element_Pointer is access Queue_Element_Type;
 20    type Queue_Element_Type is record
 21       Data : Integer;
 22       Link : Queue_Element_Pointer;
 23    end record;
 24    type Queue_Type is record
 25       Length : Natural := 0;
 26       Front  : Queue_Element_Pointer := null;
 27       Back   : Queue_Element_Pointer := null;
 28    end record;
 29 
 30 end Queues;
