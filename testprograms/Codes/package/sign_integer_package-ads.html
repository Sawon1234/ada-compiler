The Ada Program: sign_integer_package.ads

  1 -- sign_Integer_Package.ads:  abstract type of sign/magnitude integers
  2 
  3 -- This package creates an abstract data type for signed integers using
  4 -- sign/magnitude represenation.  Of course, this is completely unnessary
  5 -- as Ada has a predefined integer type already.
  6 
  7 package Sign_Integer_Package is
  8 
  9    type Integer is private;
 10 
 11    Zero, One, Minus_One : constant Integer;
 12 
 13    -- Operations on integers:
 14    -- function "+" (Left, Right: Integer) return Integer;
 15    -- function "-" (Left, Right: Integer) return Integer;
 16    -- function "*" (Left, Right: Integer) return Integer;
 17    -- function "/" (Left, Right: Integer) return Integer;
 18 
 19 private
 20 
 21    type Integer is array (0 .. 31) of Boolean;
 22    for Integer'Component_Size use 1;
 23 
 24    Zero      : constant Integer := Integer'(others=>False);
 25    One       : constant Integer := Integer'(0=>True, others=>False);
 26    Minus_One : constant Integer := Integer'(0|31=>True, others=>False);
 27 
 28 end Sign_Integer_Package;
