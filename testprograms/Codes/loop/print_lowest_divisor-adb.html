  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Print_Lowest_Divisor is
  7 
  8    Dividend, Divisor:  Positive;
  9 
 10 begin
 11    Integer_Text_IO.Get (Dividend);
 12 
 13    if Dividend = 1 then
 14       Text_IO.Put_Line ("1 has no divisors greater than 1.");
 15    else
 16       Divisor := 2;
 17       while (Dividend mod Divisor) /= 0 loop
 18          Divisor := Divisor + 1;
 19       end loop;
 20       Integer_Text_IO.Put (Divisor, Width=>0);
 21       Text_IO.Put (" is the lowest divisor of ");
 22       Integer_Text_IO.Put (Dividend, Width=>0);
 23       Text_IO.Put_Line (" greater than 1.");
 24    end if;
 25 
 26 end Print_Lowest_Divisor;
