The Ada Program: reach_goal.adb

  1 -- read_goal.adb:  a procedure illustrating "loop"/"exit"
  2 
  3 -- From Cohen, "Ada as a Second Language," page 69
  4 
  5 with Ada.Text_IO, Ada.Integer_Text_IO;
  6 use Ada;
  7 
  8 procedure Reach_Goal is
  9 
 10    Goal       : constant Integer := 100;
 11    Left_To_Go : Integer := Goal;
 12    Progress   : Integer;
 13 
 14 begin
 15 
 16    loop
 17       Text_IO.Put_Line ("Enter progress toward goal.");
 18       Integer_Text_IO.Get (Progress);
 19       Left_To_Go := Left_To_Go - Progress;
 20       exit when Left_To_Go <= 0;
 21       Integer_Text_IO.Put (Left_To_Go);
 22       Text_IO.Put_Line (" left to go.");
 23    end loop;
 24 
 25    Text_IO.Put_Line ("You have reached your goal.");
 26 
 27 end Reach_Goal;
