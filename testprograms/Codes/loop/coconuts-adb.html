  2 with Ada.Text_IO, Ada.Integer_Text_IO;
  3 use Ada;
  4 
  5 -- Four people are stranded on a tropical island.  They gather a large
  6 -- pile of coconuts before going to sleep.  Several times during the
  7 -- night someone gets up and reduces the pile of coconuts.  Each time
  8 -- the person, divides the pile in to 4 equal shares and finds one
  9 -- coconut left over.  This coconut is given to a monkey.  The person
 10 -- takes one of the shares and pushes the rest of the coconuts together.
 11 -- In the morning the people divided the pile into four equal shares
 12 -- and again one coconut is left over.  How many coconuts were gathered
 13 -- to begin with?
 14 
 15 procedure Coconuts is
 16    Nuts : Natural := 1;
 17 
 18    function Even (N: Natural) return Boolean is
 19    begin
 20       return (N mod 4 = 1);
 21    end Even;
 22 
 23    function Reduce (N: Natural) return Natural is
 24    begin
 25       return (3 * ((N-1) / 4));
 26    end Reduce;
 27 
 28    function Try (N, S: Natural) return Boolean is
 29    begin
 30       return (Even (N) and then (
 31          S=0 or else Try (Reduce (N), S-1)));
 32    end Try;
 33 
 34 begin
 35 
 36    Step: loop
 37       exit when Try (Nuts, 5);
 38       Nuts := Nuts + 1;
 39    end loop Step;
 40 
 41    Integer_Text_IO.Put (Nuts);
 42    Text_IO.New_Line;
 43 
 44 end Coconuts;
