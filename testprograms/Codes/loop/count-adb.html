The Ada Program: count.adb

  1 -- count.adb: count the characters up to '#'
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO; use Ada;
  4 
  5 procedure Count is
  6 
  7    End_Marker: constant Character := '#';
  8 
  9    Ch: Character;
 10    N : Integer := 0;
 11 
 12 begin
 13 
 14    Input: loop
 15 
 16       -- NB.  End of line indications (CR&LF, LF, or whatever)
 17       -- are skipped by "Get"; but not by "Text_IO.Get_Immediate".
 18 
 19       Text_IO.Get (Ch);
 20       exit Input when Ch=End_Marker;
 21       N := N + 1;
 22 
 23    end loop Input;
 24 
 25    Integer_Text_IO.Put (N, Width=>0);
 26 
 27 end Count;
