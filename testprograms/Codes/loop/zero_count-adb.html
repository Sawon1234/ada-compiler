The Ada Program: zero_count.adb

  1 -- zero_count.adb:  an Ada program fragment to count zero characters
  2 
  3 -- A block with a loop that counts the number of zero characters ('0')
  4 -- in the standard input until a period is read.
  5 
  6 Zero_Count: declare
  7    Ch    : Character;
  8    Count : Integer := 0;
  9 begin
 10    loop
 11       Text_IO.Get (Ch);
 12       exit when (Ch='.');
 13       if Ch='0' then
 14          Count := Count + 1;
 15       end if;
 16    end loop;
 17 end Zero_Count;
