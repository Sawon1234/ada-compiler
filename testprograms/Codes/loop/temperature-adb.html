The Ada Program: temperature.adb

  1 -- Temperature.adb:  high and low temperatures over 24 hours
  2 with Ada.Text_IO, Ada.Integer_Text_IO; use Ada;
  3 procedure Temperature is
  4 
  5    Num_Hours : constant Integer := 24; -- Number of hours in time period
  6 
  7    Temp : Integer;                     -- An hourly temperature reading
  8    High : Integer := Integer'First;    -- Highest temperature so far
  9    Low  : Integer := Integer'Last;     -- Lowest temperature so far
 10    Hour : Integer;                     -- Loop control variable
 11 
 12 begin
 13    -- Prompt user for all input once
 14    Text_IO.Put (Item => "Enter 24 temperatures");
 15    Text_IO.New_Line;
 16 
 17    Hour := 1;
 18    Input_Loop: loop
 19       exit Input_Loop when Hour > Num_Hours;
 20 
 21       -- Get and echo an hourly temperature
 22       Integer_Text_IO.Get (Item => Temp);
 23       Integer_Text_IO.Put (Item => Temp, Width => 4);
 24 
 25       -- Lowest temperature so far?
 26       if Temp < Low then
 27          Low := Temp;
 28       end if;
 29 
 30       -- Highest temperature so far?
 31       if Temp > High then
 32          High := Temp;
 33       end if;
 34 
 35       -- Update loop ending condition
 36       Hour := Hour + 1;
 37    end loop Input_Loop;
 38 
 39       -- Print high and low temperatures
 40    Text_IO.New_Line;
 41    Text_IO.Put (Item => "High temperature is ");
 42    Integer_Text_IO.Put (Item => High, Width => 1);
 43    Text_IO.New_Line;
 44    Text_IO.Put (Item => "Low temperature is ");
 45    Integer_Text_IO.Put (Item => Low, Width => 1);
 46    Text_IO.New_Line;
 47 end Temperature;
