  1 -- previous.adb:  an Ada program fragment to count same values in input
  2 
  3 -- A block with a loop that counts the number of integer values in the
  4 -- the standard input that are equal to the previous value.  Stop when
  5 -- any value over 10,000 is read.
  6 
  7 
  8 Previous: declare
  9 
 10    Max   : constant Integer := 10_000;
 11    Count : Integer := 0;               -- initially, none the same
 12    Value : Integer;
 13    Previous_Value : Integer := Max+1;  -- initially, some illegal value
 14 
 15 begin
 16 
 17    loop
 18       Integer_Text_IO.Get (Value);
 19       exit when Value > Max;
 20       if Value = Previous_Value then
 21          Count := Count + 1;
 22       end if;
 23       Previous_Value := Value;
 24    end loop;
 25 
 26 end Previous;
