  7 procedure Riemann is
  8 
  9    function F (X: Float) return Float is
 10    begin
 11        return (Sin (X));
 12    end F;
 13 
 14    -- left endpoint approximations
 15    function Integrate (Lower, Upper: Float; N: Integer) return Float is
 16       DX : constant Float := (Upper-Lower)/Float(N);
 17       Sum, XI: Float;
 18    begin
 19       Sum := F (Lower);
 20       for I in 1 .. N-1 loop
 21          XI := Float(I) * DX;
 22          Sum := Sum + F (Xi);
 23       end loop;
 24       return (DX * Sum);
 25    end Integrate;
