The Ada Program: nonzero.adb

  1 -- nonzero.adb:  find last nonzero digit of N factorial
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Nonzero is
  7 
  8    N, Twos : Natural;
  9    P, J : Positive;
 10 
 11 begin
 12 
 13    while not (Text_IO.End_Of_File) loop
 14 
 15       Integer_Text_IO.Get (N);
 16       P := 1;  Twos := 0;
 17 
 18       for I in 2..N loop
 19 
 20          -- discard factors of 2 and 5 from I
 21          J := I;
 22          while J mod 2 = 0 loop J := J / 2; Twos := Twos + 1; end loop;
 23          while J mod 5 = 0 loop J := J / 5; Twos := Twos - 1; end loop;
 24 
 25          -- compute the last digit of the factorial of I without
 26          -- any factors of 2 or 5 included
 27          P := (J*P) mod 10;
 28 
 29       end loop;
 30 
 31       -- restore factors of 2 excepted those canceled by a 5
 32       for I in 1..Twos loop P := (2*P) mod 10; end loop;
 33 
 34       Integer_Text_IO.Put (N, Width=>5);
 35       Text_IO.Put (" ->");
 36       Integer_Text_IO.Put (P, Width=>1);
 37       Text_IO.New_Line;
 38 
 39    end loop;
 40 
 41 end Nonzero;
