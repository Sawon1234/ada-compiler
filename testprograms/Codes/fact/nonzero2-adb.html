The Ada Program: nonzero2.adb

  1 -- nonzero.adb:  find last nonzero digit of N factorial
  2 
  3 with
  4   Ada.Command_Line,     -- access to external execution env (Ada95 A.15)
  5   Ada.Text_IO,          -- usual text (Latin 1) input/output (A.10.1)
  6   Ada.Integer_Text_IO;  -- preinstantiated generic for Integer IO
  7 use Ada;
  8 
  9 procedure Nonzero is
 10 
 11    S : constant String  := Command_Line.Argument (1);
 12    N : constant Natural := Natural'Value (S);
 13    P : Positive := 1;
 14    J : Positive;
 15    Twos : Natural := 0;
 16 
 17 begin
 18 
 19    for I in 2..N loop
 20 
 21       -- discard factors of 2 and 5 from I
 22       J := I;
 23       while J mod 2 = 0 loop J := J / 2; Twos := Twos + 1; end loop;
 24       while J mod 5 = 0 loop J := J / 5; Twos := Twos - 1; end loop;
 25 
 26       -- compute the last digit of the factorial of I without
 27       -- any factors of 2 or 5 included
 28       P := (J*P) mod 10;
 29 
 30    end loop;
 31 
 32    -- restore factors of 2 excepted those canceled by a 5
 33    for I in 1..Twos loop P := (2*P) mod 10; end loop;
 34 
 35    Integer_Text_IO.Put (P);
 36    Text_IO.New_Line;
 37 
 38 end Nonzero;
