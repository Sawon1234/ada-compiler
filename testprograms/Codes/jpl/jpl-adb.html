The Ada Program: jpl.adb

  1 with Ada.Text_IO, Ada.Command_Line;
  2 use Ada;
  3 
  4 procedure JPL is
  5 
  6    subtype Intensity is Natural range 0 .. 20;
  7    type Star_Array is
  8       array (Positive range <>, Positive range <>) of Intensity;
  9 
 10    Arg: constant String := Command_Line.Argument(1);
 11 
 12    Astro : Star_Array :=
 13       (( 0, 3, 4, 0, 0, 0, 6, 8),
 14        ( 5,13, 6, 0, 0, 0, 2, 3),
 15        ( 2, 6, 2, 7, 3, 0,10, 0),
 16        ( 0, 0, 4,15, 4, 1, 6, 0),
 17        ( 0, 0, 7,12, 6, 9,10, 4),
 18        ( 5, 0, 6,10, 6, 4, 8, 0));
 19 
 20    Stellar : Star_Array :=
 21       (( 0, 3, 4, 0,14, 0, 6, 8),
 22        ( 5,13, 6, 6, 9,14, 2, 3),
 23        ( 2, 6, 2, 7, 3, 0,10, 0),
 24        ( 0, 0, 4,15, 4, 1, 6, 0),
 25        ( 0, 0, 7,12, 6, 9,10, 4),
 26        ( 5, 0, 6,10, 6, 4, 8, 0),
 27        (11, 2, 0, 0, 1, 2, 3, 0));
 28 
 29 
 30    procedure Map (Sky: Star_Array) is
 31       I: Integer;
 32    begin
 33 
 34       -- Print top border
 35       Text_IO.Put ("-");
 36       for Col in Sky'Range(2) loop
 37          Text_IO.Put (" -");
 38       end loop;
 39       Text_IO.Put (" -");
 40       Text_IO.New_Line;
 41 
 42 
 43       for Row in Sky'Range(1) loop
 44          Text_IO.Put (":");     -- Print left border
 45          for Col in Sky'Range(2) loop
 46             if (Row in Sky'First(1)+1 .. Sky'Last(1)-1) and then
 47                (Col in Sky'First(2)+1 .. Sky'Last(2)-1) then
 48                I := Sky(Row,Col)+Sky(Row+1,Col)+Sky(Row-1,Col)+
 49                   Sky(Row,Col+1)+Sky(Row,Col-1);
 50             else
 51                I := 0;   -- no star
 52             end if;
 53             if (I>30) then
 54                Text_IO.Put (" *");
 55             else
 56                Text_IO.Put ("  ");
 57             end if;
 58          end loop;
 59          Text_IO.Put (" :");    -- Print right border
 60          Text_IO.New_Line;
 61       end loop;
 62 
 63       -- Print bottom border
 64       Text_IO.Put ("-");
 65       for Col in Sky'Range(2) loop
 66          Text_IO.Put (" -");
 67       end loop;
 68       Text_IO.Put (" -");
 69       Text_IO.New_Line;
 70 
 71    end Map;
 72 
 73 begin
 74    if Arg = "Astro" then
 75       Text_IO.Put_Line ("The constellation Astro");
 76       Map (Astro);
 77    elsif Arg = "Stellar" then
 78       Text_IO.Put_Line ("The constellation Stellar");
 79       Map (Stellar);
 80    end if;
 81 end JPL;
