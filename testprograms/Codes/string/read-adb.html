The Ada Program: read.adb

  1 -- read.adb:  read in array of bounded strings from standard input
  2 
  3 with
  4   Ada.Strings.Bounded,  -- bounded-string type (Ada95 A.4.4)
  5   Ada.Text_IO;
  6 use Ada;
  7 
  8 procedure Read is
  9 
 10    package String_25 is new Strings.Bounded.Generic_Bounded_Length (25);
 11 
 12    subtype Word_Type is String_25.Bounded_String;  -- renaming
 13 
 14    Max_Number_Of_Words : constant := 100;
 15    List : array (1..Max_Number_Of_Words) of Word_Type;
 16    Number_Of_Words : Integer range 0 .. Max_Number_Of_Words := 0;
 17 
 18    Line: String (1..100);
 19    Length: Integer;
 20 
 21 begin
 22 
 23    while not (Text_IO.End_Of_File) loop
 24       Text_IO.Get_Line (Line, Length);
 25       Number_Of_Words := Number_Of_Words + 1;
 26       List (Number_Of_Words) :=
 27          String_25.To_Bounded_String (Line (1..Length));
 28    end loop;
 29 
 30 end Read;
