The Ada Program: simple.adb

  1 -- simple.adb:  illustrate string slices and character concatenation
  2 
  3 with Ada.Text_IO, Ada.Characters.Latin_1;
  4 use Ada, Ada.Characters;
  5 
  6 procedure Simple is
  7 
  8    C: Character;
  9    S_1: constant String := "This is a pretty long string.";
 10 
 11    -- Index constraints are necessary for strings in order to specify
 12    -- how many characters these fixed-sized strings contain.  If the
 13    -- string is initialized, then the constraint is not needed.
 14    S_2: String (1..5);
 15    S_3: String (6..10);
 16    S_4: String (2..4);
 17 
 18 begin
 19 
 20    C := S_1 (1);           -- Select a single character from the string.
 21    S_2 := S_1 (1..5);      -- Assign array slice to a string of the same
 22    S_2 := S_1 (6..10);     -- size--only size matters, not the bounds.
 23    S_3 := S_2;
 24    S_3(7..8) := S_1(26..27);
 25    S_4 := S_1 (10..12);
 26 
 27    Text_IO.Put (C);  Text_IO.New_Line;                  -- T
 28    Text_IO.Put_Line (S_1);                              -- This ...
 29    Text_IO.Put_Line (S_1(10..12));                      --  pr
 30    Text_IO.Put_Line (S_1(1..1));                        -- T
 31    Text_IO.Put_Line (S_1(5..13) & 'A');                 -- is a preA
 32    Text_IO.Put_Line (S_1(5..13) & '¿');                 -- is a pre¿
 33    Text_IO.Put_Line (S_1(5..13) & Latin_1.UC_A_Grave);  -- is a pre¿
 34    Text_IO.Put_Line (S_2);                              -- is a
 35    Text_IO.Put_Line (S_3);                              -- in a
 36    Text_IO.Put_Line (S_4);                              --  pr
 37 
 38 end Simple;
