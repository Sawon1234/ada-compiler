The Ada Program: opaque.ads

  1 -- opaque.ads:  fraction package with data representation hidden
  2 
  3 package Opaque is
  4 
  5    -- Clients can use (declare objects of) type "Fraction", but
  6    -- the actually definition (representation) is not revealed.
  7    type Fraction is private;
  8 
  9    function Rat (Num: Integer; Den: Positive:=1) return Fraction;
 10    function Is_Zero (F: Fraction) return Boolean;
 11    function "/" (Left: Integer; Right: Integer) return Fraction;
 12    function "/" (Left, Right: Fraction) return Fraction;
 13    function "*" (Left, Right: Fraction) return Fraction;
 14    function "-" (Right: Fraction) return Fraction;
 15    function "-" (Left, Right: Fraction) return Fraction;
 16    function "+" (Left, Right: Fraction) return Fraction;
 17 
 18 private
 19 
 20    -- The "private" section of a package body reveals information
 21    -- needed by the compiler, but the client cannot make use of
 22    -- this infomation in the program.
 23 
 24    type Fraction is
 25       record
 26          Num : Integer;
 27          Den : Positive := 1;
 28       end record;
 29 
 30 end Opaque;
