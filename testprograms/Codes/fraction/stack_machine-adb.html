  1 with Ada.Integer_Text_IO, Ada.Text_IO, Ada.Strings.Fixed,
  2      Ada.Strings.Maps.Constants, Fraction_Package;
  3 use Ada, Ada.Strings, Fraction_Package;
  4 
  5 
  6 procedure Stack_Machine is
  7 
  8    type Stack_Array is array (1..10) of Fraction;
  9    Stack: Stack_Array;
 10    Top: Integer range 1..11 := 1;
 11 
 12    Overflow, Underflow: exception;
 13 
 14    procedure Push (X: Fraction) is
 15    begin
 16       if Top > 10 then
 17          raise Overflow;
 18       else
 19          Stack(Top) := X;
 20          Top := Top + 1;
 21       end if;
 22    end Push;
 23 
 24    function Pop return Fraction is
 25    begin
 26       if Top = 1 then
 27          raise Underflow;
 28       else
 29          Top := Top - 1;
 30          return (Stack(Top));
 31       end if;
 32    end Pop;
 33 
 34    function Is_Operand (X: String) return Boolean is
 35       Digit_Set: Maps.Character_Set renames Maps.Constants.Decimal_Digit_Set;
 36    begin
 37       return (Fixed.Index (X, Digit_Set) /= 0);
 38    end Is_Operand;
 39 
 40    Zero_Divide: exception;
 41 
 42    function Get (X: String) return Fraction is
 43       Last_Position: Integer;
 44       Slash: Integer;
 45       Num, Den: Integer;
 46    begin
 47       Integer_Text_IO.Get (From=>X, Item=>Num, Last=>Last_Position);
 48       Slash := Fixed.Index (X, Maps.To_Set("/"));
 49       if Slash = 0 then
 50          return (Rat (Num));
 51       else
 52          Integer_Text_IO.Get (From=>X(Slash+1..X'Last),
 53                               Item=>Den, Last=>Last_Position);
 54          if Den = 0 then
 55             raise Zero_Divide;
 56          else
 57             return (Num/Den);
 58          end if;
 59       end if;
 60    end Get;
 61 
 62 
 63    procedure Put_Line (X: Fraction) is
 64    begin
 65       Integer_Text_IO.Put (X.Num, Width=>0);
 66       Text_IO.Put ("/");
 67       Integer_Text_IO.Put (X.Den, Width=>0);
 68       Text_IO.New_Line;
 69    end Put_Line;
 70 
 71 begin
 72 
 73    while not (Text_IO.End_Of_File) loop
 74 
 75       declare
 76          Line: String(1..100);
 77          Line_Length: Integer;
 78          Left, Right, Result: Fraction;
 79       begin
 80          Text_IO.Get_Line(Line, Line_Length);
 81          if Is_Operand (Line (1..Line_Length)) then
 82             Push (Get (Line (1..Line_Length)));
 83          else -- must be an operator
 84             Fixed.Trim (Line, Strings.Left);  -- garbage on the right?!
 85             case Line (1) is
 86                when '+' =>
 87                   Right := Pop;
 88                   Left := Pop;
 89                   Result := Left + Right;
 90                when '-' =>
 91                   Right := Pop;
 92                   Left := Pop;
 93                   Result := Left - Right;
 94                when '*' =>
 95                   Right := Pop;
 96                   Left := Pop;
 97                   Result := Left * Right;
 98                when '/' =>
 99                   Right := Pop;
100                   Left := Pop;
101                   if Is_Zero (Right) then raise Zero_Divide; end if;
102                   Result := Left / Right;
103                when '~' =>   -- unary negation
104                   Right := Pop;
105                   Result := -Right;
106                when others =>
107                   Text_IO.Put_Line ("Error -- bad input");
108             end case;
109 
110             Put_Line (Result);
111             Push (Result);
112 
113          end if;
114       exception
115          when Constraint_Error => null;
116          when Underflow | Overflow => null;
117          when Zero_Divide => null;
118       end;
119 
120    end loop;
121 
122 end Stack_Machine;
