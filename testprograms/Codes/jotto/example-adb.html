  1 -- example.adb:  example program using Screen package
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO, Screen;
  4 use Ada;
  5 
  6 procedure Example is
  7 
  8    Buffer : String (1..80);
  9    Length : Natural;
 10    Entries: Natural := 0;
 11 
 12 begin
 13 
 14    Screen.Set_Foreground_Color (Screen.Magenta);
 15    Screen.Clear_Screen;
 16    Screen.Move_Cursor (Row=>2, Column=>35);
 17    Text_IO.Put ("Hello!");
 18    loop
 19       Screen.Set_Foreground_Color (Screen.Red);
 20       Screen.Move_Cursor (Row=>23, Column=>3);
 21       Screen.Set_Attribute (Screen.Bold);
 22       Text_IO.Put ("Enter a string => ");
 23       Screen.Erase_Line;
 24       Screen.Set_Attribute (Screen.Normal);
 25       Text_IO.Flush;
 26       Text_IO.Get_Line (Buffer, Length);
 27       exit when Buffer(1..Length)="quit";
 28       Screen.Set_Foreground_Color (Screen.Green);
 29       Screen.Move_Cursor (Row=>5+(Entries mod 16), Column=>10);
 30       Screen.Erase_Line;
 31       Entries := Entries+1;
 32       Integer_Text_IO.Put (Entries, Width=>2);
 33       Text_IO.Put (".  " & Buffer(1..Length));
 34    end loop;
 35    Screen.Set_Foreground_Color (Screen.White);
 36    Screen.Clear_Screen;
 37 
 38 exception
 39    when others => Screen.Set_Foreground_Color (Screen.White);
 40 end Example;
