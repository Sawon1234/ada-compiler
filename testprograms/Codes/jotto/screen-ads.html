The Ada Program: screen.ads

  1 -- screen.ads:  View ASCII terminal as 2D window using escape sequences
  2 
  3 -- The output device must recognized ANSI X3.64 1979 terminal-control
  4 -- escape sequences.  ECMA-48 is another such standard.
  5 
  6 --   X Window System:  use xterm (old versions don't support color)
  7 --   DOS:  check for ANSI.sys
  8 
  9 -- Ryan Stansifer (ryan@cs.fit.edu) at Mon Mar 24 18:09:39 1997
 10 -- Written for CSE1001/CSE1002 at Florida Tech.
 11 
 12 package Screen is
 13 
 14    type Attribute is (Normal, Bold, Inverse, Blink);
 15    type Color is (Black,Red,Green,Yellow,Blue,Magenta,Cyan,White);
 16 
 17    procedure Beep;
 18    procedure Move_Cursor (Row, Column: Positive:=1);
 19    procedure Set_Attribute (A: Attribute);
 20    procedure Set_Foreground_Color (C: Color);
 21    procedure Set_Background_Color (C: Color);
 22    procedure Set_Color (Fg, Bg: Color);
 23    procedure Clear_Screen;
 24    procedure Erase_Line;
 25 
 26 end Screen;
