The Ada Program: factorial.adb

  1 -- factorial.adb:  compute factorial illustrating loop invariants
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Factorial is
  7 
  8    N,I,P: Integer;
  9 
 10 begin -- Inv
 11 
 12    Read: loop
 13       -- Get input interactively from user
 14       Text_IO.Put ("Enter an integer N:  ");
 15       Integer_Text_IO.Get (N);
 16 
 17       exit when N < 0;
 18       -- assertion 1:  0<=N
 19 
 20       I := 0; P := 1;
 21 
 22       -- Loop invariant:  P=I! and I<=N
 23       while I<N loop
 24          -- Loop invariant:  P=I! and I<=N
 25          I:=I+1; P:=P*I;
 26          -- Observe that:  if I<N,  then (I+1)<=N
 27          -- Observe that:  if P=I!, then P*(I+1)=(I+1)!
 28          -- Loop invariant:  P=I! and I<=N
 29       end loop;
 30       -- assertion 2: I=N
 31       -- assertion 3: P=N!
 32 
 33       -- Output results:  "N! = P"
 34       Integer_Text_IO.Put (Item=>N, Width=>1);
 35       Text_IO.Put ("! = ");
 36       Integer_Text_IO.Put (Item=>P, Width=>1);
 37       Text_IO.New_Line;
 38    end loop Read;
 39 
 40 end Factorial;
