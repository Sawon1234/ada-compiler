  1 -- multiply.adb:  multiplying integers illustrating loop invariants
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Multiply is
  7 
  8    N,X,Y,Z: Integer;
  9 
 10 begin
 11 
 12    Read: loop
 13       -- Get input interactively from user
 14       Text_IO.Put ("Enter integers X and Y:  ");
 15       Integer_Text_IO.Get (X);
 16       Integer_Text_IO.Get (Y);
 17 
 18       exit when Y<0;
 19 
 20       -- assertion 1:  Y>=0
 21 
 22       Z:=0; N:=Y;
 23 
 24       -- Loop invariant:  Z=X*(Y-N) and N>=0
 25       while N>0 loop
 26          -- Loop invariant:  Z=X*(Y-N) and N>=0
 27          Z:=Z+X; N:=N-1;
 28          -- Observe that:  if N>0, then (N-1)>=0
 29          -- Observe that:  if Z=X*(Y-N), then (Z+X)=X*(Y-(N-1))
 30          -- Loop invariant:  Z=X*(Y-N) and N>=0
 31       end loop;
 32       -- assertion 2: N=0
 33       -- assertion 3: Z=X*Y
 34 
 35       -- Output results:  "X*Y=Z"
 36       Integer_Text_IO.Put (Item=>X, Width=>1);
 37       Text_IO.Put ("*");
 38       Integer_Text_IO.Put (Item=>Y, Width=>1);
 39       Text_IO.Put ("=");
 40       Integer_Text_IO.Put (Item=>Z, Width=>1);
 41       Text_IO.New_Line;
 42    end loop Read;
 43 
 44 end Multiply;
