The Ada Program: exp.adb

  1 -- exp.adb:  compute exponentiation illustrating loop invariants
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Exp is
  7    N,I,X,Y: Integer;
  8 begin
  9 
 10    Read: loop
 11       -- Get input interactively from user
 12       Text_IO.Put ("Enter integers X and N:  ");
 13       Integer_Text_IO.Get (X);
 14       Integer_Text_IO.Get (N);
 15       exit when N < 0;
 16 
 17       -- assertion 1: N>=0
 18       -- GNAT compiler requires "-gnata"; ow, no check
 19       pragma Assert (N>=0, "N<0 !?!");
 20 
 21       I := 0;  Y := 1;
 22 
 23       -- Loop invariant:  Y=X**I and I<=N
 24       pragma Assert (Y=X**I and then I<=N, "invariant false!");
 25       while I<N loop
 26          -- Loop invariant:  Y=X**I and I<=N
 27          pragma Assert (Y=X**I and then I<=N, "invariant false!");
 28          Y:=Y*X; I:=I+1;
 29          -- Observe that:  if I<N, then (I+1)<=N
 30          -- Observe that:  if Y=X**I, then Y*X=X**(I+1)
 31          -- Loop invariant:  Y=X**I and I<=N
 32          pragma Assert (Y=X**I and then I<=N, "invariant false!");
 33       end loop;
 34       -- assertion 2: I=N
 35       -- assertion 3: Y=X**N
 36       pragma Assert (I=N and then Y=X**N, "invariant false!");
 37 
 38       -- Output results
 39       Integer_Text_IO.Put (Item=>X, Width=>1);
 40       Text_IO.Put ("**");
 41       Integer_Text_IO.Put (Item=>N, Width=>1);
 42       Text_IO.Put ("=");
 43       Integer_Text_IO.Put (Item=>Y, Width=>1);
 44       Text_IO.New_Line;
 45    end loop Read;
 46 
 47 end Exp;
