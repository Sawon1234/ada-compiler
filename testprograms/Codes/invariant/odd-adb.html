The Ada Program: odd.adb

  1 -- odd.adb:  multiplying integers illustrating loop invariants
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Odd is
  7 
  8    A,B,X,Y,Z: Integer;
  9 
 10 begin
 11    Read: loop
 12       -- Get input interactively from user
 13       Text_IO.Put ("Enter integers A and B:  ");
 14       Integer_Text_IO.Get (A);
 15       Integer_Text_IO.Get (B);
 16       exit when B<0;
 17       -- assertion 1:  B>=0
 18       X := A;  Y:= B;  Z:= 0;
 19       -- Loop invariant:  Z+X*Y=A*B and Y>=0
 20       while Y>0 loop
 21          -- Loop invariant:  Z+X*Y=A*B and Y>=0
 22 
 23          if Y mod 2 = 0 then
 24             Y := Y/2;  X := X + X;
 25             -- Observe that:  if Y>0, then (Y/2)>=0
 26             -- Z+(X+X)*(Y/2) = Z+X*Y = A*B
 27          else
 28             Y := Y-1;  Z := Z + X;
 29             -- Observe that:  if Y>0, then (Y-1)>=0
 30             -- Z+X+X*(Y-1) = Z+X*Y = A*B
 31          end if;
 32 
 33          -- Loop invariant:  Z+X*Y=A*B and Y>=0
 34       end loop;
 35       -- assertion 2: Y=0
 36       -- assertion 3: Z+X*Y=A*B
 37 
 38       -- Output results:  "A*B=Z"
 39       Integer_Text_IO.Put (Item=>A, Width=>1);
 40       Text_IO.Put ("*");
 41       Integer_Text_IO.Put (Item=>B, Width=>1);
 42       Text_IO.Put ("=");
 43       Integer_Text_IO.Put (Item=>Z, Width=>1);
 44       Text_IO.New_Line;
 45    end loop Read;
 46 end Odd;
