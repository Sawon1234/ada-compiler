The Ada Program: shapes.adb

  1 --  shapes.adb: extent "Object" to point, circle, rectangle
  2 
  3 with Ada.Numerics; use Ada.Numerics;
  4 
  5 package body Shapes is
  6 
  7    function Area (C: Circle) return Float is
  8    begin
  9       return (Pi * C.Radius**2);
 10    end Area;
 11 
 12    function Area (R: Rectangle) return Float is
 13       Length: constant Float := R.X - R.Opposite_Corner.X;
 14       Width : constant Float := R.Y - R.Opposite_Corner.Y;
 15    begin
 16       return (abs (Length*Width));
 17    end Area;
 18 
 19    function Circumference (C: Circle) return Float is
 20    begin
 21       return (2.0 * Pi * C.Radius);
 22    end Circumference;
 23 
 24 end Shapes;
