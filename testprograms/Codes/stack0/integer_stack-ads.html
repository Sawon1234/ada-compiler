  1 -- integer_stack.ads:  data structure for stack of integer elements
  2 
  3 package Integer_Stack is
  4 
  5    type Stack_Array is array (Positive range <>) of Integer;
  6 
  7    -- Record type with discrimiant for array index constraint
  8    type Stack_Type (Max_Size: Positive) is
  9       record
 10          Top      : Natural := 0;
 11          Elements : Stack_Array (1..Max_Size);
 12       end record;
 13 
 14    Underflow, Overflow: exception;
 15 
 16    -- Test whether a stack is empty (contains no elements).
 17    function Empty (Stack:  Stack_Type) return Boolean;
 18 
 19    -- Test whether a stack is full (no room for more elements).
 20    function Full (Stack: Stack_Type) return Boolean;
 21 
 22    -- Remove all elements from the stack.
 23    procedure Clear (Stack: in out Stack_Type);
 24 
 25    -- Add a new element to the top of the stack; if already full,
 26    -- then the exception "Overflow" is raised and stack is unchanged.
 27    procedure Push (Stack       : in out Stack_Type;
 28                    New_Element : in     Integer);
 29 
 30 
 31    -- Remove the top element from stack and return it in "Top_Element".
 32    -- If stack is empty, then the exception "Underflow" is raised.
 33    procedure Pop (Stack       : in out Stack_Type;
 34                   Top_Element :    out Integer);
 35 
 36 end Integer_Stack;
