  6    type String_Compare_Type is access function (Left, Right: String) return Boolean;
  7 
  8    -- unfortunately `Standard."<"'Access` is not legal
  9    function Standard_Lex (Left, Right: String) return Boolean is
 10    begin
 11       return (Left < Right);
 12    end Standard_Lex;
 13 
 14    Lex : String_Compare_Type := Standard_Lex'Access;
 15 
 16    type String_Pointer is access String;
 17 
 18    Max_Number_Of_Names : constant := 100;
 19    Number_Of_Names     : Integer range 0 .. Max_Number_Of_Names := 0;
 20 
 21    type Array_Names_Type is array (Natural range <>) of String_Pointer;
 22    Names: Array_Names_Type (0..Max_Number_Of_Names);
 23 
 24    -- sort in non descreasing order
 25    procedure Sort (N: in out Array_Names_Type; Compare: String_Compare_Type := Lex) is
 26       Temp: String_Pointer;
 27       No_Exchanges: Boolean := True;
 28    begin
 29       -- Bubble sort
 30       loop
 31          No_Exchanges := True;
 32          for I in N'First .. N'Last-1 loop
 33             if Compare (N(I+1).all, N(I).all) then
 34                Temp := N(I+1);
 35                N(I+1) := N(I) ;
 36                N(I) := Temp;
 37                No_Exchanges := False;
 38             end if;
 39          end loop;
 40       exit when No_Exchanges;
 41       end loop;
 42    end Sort;

 46    declare
 47       Line: String (1..100);
 48       Length: Integer;
 49    begin
 50       while not (Text_IO.End_Of_File) loop
 51          Text_IO.Get_Line (Line, Length);
 52          Names (Number_Of_Names) := new String'(Line(1..Length));
 53          Number_Of_Names := Number_Of_Names + 1;
 54       end loop;
 55    end;
 56 
 57    Sort (Names(0..Number_Of_Names-1));
 58 
 59    for I in 0..Number_Of_Names-1 loop

 61    end loop;
