  3 -- sort in non decreasing order
  4 procedure Sort (N: in out Array_Words_Type;
  5                 Compare: String_Compare_Type := Standard_Lex'Access) is
  6    Temp: String_Pointer;
  7    No_Exchanges: Boolean := True;
  8 begin
  9    -- Bubble sort
 10    loop
 11       No_Exchanges := True;
 12       for I in N'First .. N'Last-1 loop
 13          if Compare (N(I+1).all, N(I).all) then
 14             Temp := N(I+1);
 15             N(I+1) := N(I) ;
 16             N(I) := Temp;
 17             No_Exchanges := False;
 18          end if;
 19       end loop;
 20    exit when No_Exchanges;
 21    end loop;
 22 end Sort;
