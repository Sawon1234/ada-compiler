  3 -- Fill language "L", and array "Words" declared in the parent
  4 -- with values from user.
  5 
  6 -- This procedure could raise CONSTRAINT_ERROR, if more words
  7 -- are entered than the size of the array "Words".
  8 
  9 procedure Get_Input is
 10 
 11       Line: String (1..100);
 12       Length: Integer;
 13 
 14 begin
 15 
 16    Text_IO.Put ("Enter language to sort by => ");
 17    Language_IO.Get (L);
 18    Text_IO.Skip_Line;
 19 
 20    while not (Text_IO.End_Of_File) loop
 21       Text_IO.Get_Line (Line, Length);
 22       Number_Of_Words := Number_Of_Words + 1;
 23       Words (Number_Of_Words) := new String'(Line(1..Length));
 24    end loop;
 25 
 26 end Get_Input;
