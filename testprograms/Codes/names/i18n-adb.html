  1 with Ada.Text_IO, Ada.Integer_Text_IO;  use Ada;
  2 with Spanish_Ordering, German_Ordering;
  3 
  4 procedure I18n is
  5 
  6    type Language is (Lex, Spanish, German);
  7    package Language_IO is new Text_IO.Enumeration_IO (Language);
  8 
  9    -- Declare type of pointers to functions that compare strings.
 10    type String_Compare_Type is access function (Left, Right: String) return Boolean;
 11 
 12    -- Unfortunately `Standard."<"'Access` is not legal since "<" is presumably
 13    -- compiled directly to machine code and so there is no code to point to.
 14    -- Simply renaming "<" does not avoid the problem.  So we define our own
 15    -- subpropram so that we can point to it, ie., `Standard_Lex'Access`.
 16    function Standard_Lex (Left, Right: String) return Boolean is
 17    begin
 18       return (Left < Right);
 19    end Standard_Lex;
 20 
 21    type String_Pointer is access String;
 22    type Array_Words_Type is array (Natural range <>) of String_Pointer;
 23    Max_Number_Of_Words : constant := 100;
 24    Number_Of_Words     : Integer range 0 .. Max_Number_Of_Words := 0;
 25    Words: Array_Words_Type (1..Max_Number_Of_Words);
 26    L : Language;
 27 
 28    procedure Get_Input is separate;
 29    procedure Sort (N: in out Array_Words_Type;
 30       Compare: String_Compare_Type := Standard_Lex'Access) is separate;
 31 
 32 begin
 33 
 34    Get_Input;
 35 
 36    case L is
 37       when Lex =>     Sort (Words(1..Number_Of_Words));
 38       when Spanish => Sort (Words(1..Number_Of_Words), Spanish_Ordering'Access);
 39       when German =>  Sort (Words(1..Number_Of_Words), German_Ordering'Access);
 40    end case;
 41 
 42    for I in 1..Number_Of_Words loop
 43       Text_IO.Put_Line (Words(I).all);
 44    end loop;
 45 
 46 end I18n;
