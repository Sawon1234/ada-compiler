  3 with Ada.Numerics.Elementary_Functions; use Ada.Numerics;

  5 package body Vectors is

  7    function "*" (Left, Right: Vector_Type) return Vector_Type is
  8    begin
  9       return (
 10          Left(2)*Right(3) - Left(3)*Right(2),
 11          Left(3)*Right(1) - Left(1)*Right(3),
 12          Left(1)*Right(2) - Left(2)*Right(1)
 13       );
 14    end "*";
 15 
 16    function Magnitude (V: Vector_Type) return Float is
 17       Sum_Of_Squares: Float := 0.0;
 18    begin
 19       for I in V'Range loop
 20          Sum_Of_Squares := Sum_Of_Squares + V(I)**2;
 21       end loop;
 22       return Elementary_Functions.Sqrt (Sum_Of_Squares);
 23    end Magnitude;
 24 
 25 end Vectors;
