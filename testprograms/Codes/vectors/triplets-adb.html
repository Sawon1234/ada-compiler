The Ada Program: triplets.adb

  1 -- Triplets.adb:  type for triples; used as base type of vectors
  2 
  3 -- From Cohen, 12.2 Derived types, page 503
  4 
  5 package body Triplets is
  6 
  7    function "+" (Left, Right: Triplet_Type) return Triplet_Type is
  8       Result: Triplet_Type;
  9    begin
 10       for I in Result'Range loop
 11          Result(I) := Left(I) + Right(I);
 12       end loop;
 13       return Result;
 14    end "+";
 15 
 16    function "*" (Left, Right: Triplet_Type) return Triplet_Type is
 17       Result: Triplet_Type;
 18    begin
 19       for I in Result'Range loop
 20          Result(I) := Left(I) * Right(I);
 21       end loop;
 22       return Result;
 23    end "*";
 24 
 25    procedure Rescale (Triplet: in out Triplet_Type; Factor: in Float) is
 26    begin
 27       for I in Triplet'Range loop
 28          Triplet(I) := Triplet(I) * Factor;
 29       end loop;
 30    end Rescale;
 31 
 32 end Triplets;
