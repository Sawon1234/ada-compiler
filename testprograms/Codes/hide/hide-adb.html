  1 -- hide.adb: Ada program using a package for information hiding

  3 with Pack, Text_IO;
  4 procedure Hide is
  5    Max : constant := 200;
  6    Len : Natural;
  7    Input: String (1..Max);
  8 begin
  9    loop
 10       Text_IO.Put ("Enter a string -- ");
 11       Text_IO.Get_Line (Input, Len);
 12       Text_IO.Put_Line ("OK: " & Natural'Image (Pack.Sub (Input (1..Len))));
 13    end loop;
 14 end Hide;
