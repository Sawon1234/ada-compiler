The Ada Program: loan.adb

  1 -- loan.adb:  program to print amortization table for given loan
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Float_Text_IO;
  4 with Ada.Text_IO.Editing;
  5 use Ada, Ada.Text_IO;
  6 
  7 procedure Loan is
  8 
  9    type Money is delta 0.01 digits 10;   -- decimal fixed-point type
 10    package Money_IO is new Editing.Decimal_Output (Num=>Money);
 11    package Decimal_Text_IO is new Text_IO.Decimal_IO (Num=>Money);
 12    Format:constant Editing.Picture:=Editing.To_Picture("BBB$ZZZ_ZZ9.99");
 13 
 14    Loan, Payment     : Money;
 15    Rate, X           : Float;
 16    Number_Of_Payments: Natural;
 17    Index             : Natural := 1;
 18 
 19    Balance, Total_Interest, Principle       : Money;
 20    This_Payment, New_Balance, Interest_Owed : Money;
 21 
 22 begin
 23 
 24    -- Get amount of loan (Money), interest rate (Float),
 25    -- and number of payments (Natural)
 26    Decimal_Text_IO.Get (Loan, Width=>10);
 27    Float_Text_IO.Get (Rate);
 28    Integer_Text_IO.Get (Number_Of_Payments);
 29 
 30    -- Convert interest rate from yearly percent
 31    Rate := Rate / 1200.0;
 32 
 33    -- Calculate amount of monthly payment (as a floating point number)
 34    X := (Rate + 1.0) ** Number_Of_Payments;
 35    X := 1.0 + (1.0 / (X - 1.0));
 36    X := Float(Loan) * Rate * X;
 37 
 38    -- Convert to type "Money" by rounding up to the next cent
 39    Payment := Money (X + 0.005);
 40 
 41    -- Print as money the amount of the loan and payment just calculated
 42    Text_IO.Put ("Amount of loan     = ");
 43    Money_IO.Put (Loan, Format);
 44    Text_IO.New_Line;
 45    Text_IO.Put ("Interest rate      = ");
 46    Float_Text_IO.Put (Rate, Fore=>11, Aft=>5, Exp=>0);
 47    Text_IO.New_Line;
 48    Text_IO.Put ("Number of payments = ");
 49    Integer_Text_IO.Put (Number_Of_Payments,  Width=>11);
 50    Text_IO.New_Line;
 51    Text_IO.Put ("Amount of payment  = ");
 52    Money_IO.Put (Payment, Format);
 53    Text_IO.New_Line(2);
 54 
 55    -- Print header of table
 56    Text_IO.Put_Line ("Month     Payment     Principle      Interest       Balance");
 57    Text_IO.Set_Col (46);
 58    Money_IO.Put (Loan, Format);
 59    Text_IO.New_Line;
 60 
 61    -- Print each monthly payment
 62    Balance := Loan;
 63    Total_Interest := 0.00;
 64 
 65    for I in 1..Number_Of_Payments loop
 66       Interest_Owed := Money (Rate*Float(Balance));-- round to nearest cent
 67       Total_Interest := Total_Interest + Interest_owed;
 68       if (I=Number_Of_Payments) then
 69          This_Payment := Interest_Owed + Balance;
 70       else
 71          This_Payment := Payment;
 72       end if;
 73       Principle   := This_Payment - Interest_Owed;
 74       New_Balance := Balance - Principle;
 75 
 76       Integer_Text_IO.Put (Index, Width=>3);
 77       Money_IO.Put (This_Payment, Format);
 78       Money_IO.Put (Principle, Format);
 79       Money_IO.Put (Interest_Owed, Format);
 80       Money_IO.Put (New_Balance, Format);
 81       Text_IO.New_Line;
 82 
 83       Balance := New_Balance;
 84       Index := Index + 1;
 85    end loop;
 86 
 87    -- Print the sum of all the interest paid during the life of the loan
 88    Text_IO.New_Line;
 89    Text_IO.Set_Col (15);
 90    Text_IO.Put ("Total interest = ");
 91    Money_IO.Put (Total_Interest, Format);
 92    Text_IO.New_Line;
 93 
 94 end Loan;
