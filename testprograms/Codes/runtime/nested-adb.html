  1 -- nested.adb:  the dynamic structure of pending calls
  2 
  3 procedure Nested is
  4 
  5    procedure Q is
  6       procedure R is begin null; end R;
  7    begin
  8       R;
  9    end Q;
 10 
 11    procedure P is begin Q; end P;
 12 
 13 begin
 14 
 15    P;  -- call subprocedure "P"
 16    Q;  -- call subprocedure "Q"
 17 
 18 end Nested;
