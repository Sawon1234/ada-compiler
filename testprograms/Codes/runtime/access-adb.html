  1 -- access.adb:  the dynamic structure of pending calls
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Access is
  7 
  8    G: Integer := 2;
  9 
 10    procedure Q is
 11       N: Integer := 3;
 12 
 13       procedure R is
 14          L: Integer := 5;
 15       begin
 16          G := G+L;
 17          Integer_Text_IO.Put (G);  -- access G
 18          Integer_Text_IO.Put (N);  -- access N
 19          Integer_Text_IO.Put (L);  -- access L
 20          Text_IO.New_Line;
 21       end R;
 22 
 23    begin
 24       R;
 25    end Q;
 26 
 27    procedure P is
 28    begin
 29       Q;
 30    end P;
 31 
 32 begin
 33 
 34    P;  -- call subprocedure "P"
 35    Q;  -- call subprocedure "Q"
 36 
 37 end Access;
