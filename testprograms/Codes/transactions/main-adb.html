  1 -- main.adb:  tagged records types and inheritence
  2 
  3 with Transactions, Transactions.Deposits, Transactions.Checks;
  4 use Transactions, Transactions.Deposits, Transactions.Checks;
  5 
  6 with Ada.Strings.Unbounded;
  7 use Ada.Strings;
  8 
  9 procedure Main is
 10 
 11    type Transaction_List_Cell_Type;  -- forward declaration
 12    type Transaction_List_Type is access Transaction_List_Cell_Type;
 13    type Transaction_List_Cell_Type is
 14       record
 15          Transaction_Part : Transaction_Pointer_Type;
 16          Link_Part        : Transaction_List_Type;
 17       end record;
 18 
 19    List        : Transaction_List_Type;
 20    Transaction : Transaction_Pointer_Type;
 21 
 22 begin
 23 
 24    Transaction := new Transaction_Type'(Account_Part=>"123-4567-5432-FH",
 25       Amount_Part=>123.45);
 26    List := new Transaction_List_Cell_Type'(Transaction_Part=>Transaction,
 27       Link_Part=>null);
 28 
 29    Transaction := new Deposit_Type'(Account_Part=>"832-7331-5432-FH",
 30       Amount_Part=>123_456.78,
 31       Branch_Part=>Unbounded.To_Unbounded_String ("Brevard County"));
 32    List := new Transaction_List_Cell_Type'(Transaction_Part=>Transaction,
 33       Link_Part=>List);
 34 
 35    Transaction := new Check_Type'(Account_Part=>"432-1982-5432-FH",
 36       Amount_Part=>1_234.56, Check_Number_Part=>1_000);
 37    List := new Transaction_List_Cell_Type'(Transaction_Part=>Transaction,
 38       Link_Part=>List);
 39 
 40    while (List /= null) loop
 41       Put (List.Transaction_Part.all);
 42       List := List.Link_Part;
 43    end loop;
 44 
 45 end Main;
