The Ada Program: lower.adb

  1 -- lower.adb:  convert upper-case Latin-1 character to lower case
  2 
  3 -- This is uncessary as there is already the function
  4 --    Ada.Characters.Handling.To_Lower
  5 
  6 with Ada.Characters.Latin_1;
  7 use  Ada.Characters;
  8 
  9 function Lower (Ch: in Character) return Character is
 10 
 11    A_Grave:     Character renames Latin_1.UC_A_Grave;          -- 'À' 192
 12    O_Diaeresis: Character renames Latin_1.UC_O_Diaeresis;      -- 'Ö' 214
 13    Multi:       Character renames Latin_1.Multiplication_Sign; -- '×' 215
 14    O_Stroke:    Character renames Latin_1.UC_O_Oblique_Stroke; -- 'Ø' 216
 15    Thorn:       Character renames Latin_1.UC_Icelandic_Thorn;  -- 'Þ' 222
 16 
 17    -- consecutive, upper-case characters in the Latin-1 character set:
 18    --   'A' .. 'Z',  A_Grave .. O_Diaeresis, O_Stroke .. Thorn
 19 
 20    subtype R1 is Character range 'A'..'Z';
 21    subtype R2 is Character range A_Grave..Thorn;  -- except Multi!
 22 
 23    Shift: constant Integer := 32;
 24    Lower_Position: Integer;
 25 
 26 begin
 27 
 28    if (Ch in R1) or else (Ch in R2 and then Ch /= Multi) then
 29       Lower_Position := Character'Pos(Ch) + Shift;
 30       return Character'Val (Lower_Position);
 31    else
 32       return Ch;
 33    end if;
 34 
 35 end Lower;
