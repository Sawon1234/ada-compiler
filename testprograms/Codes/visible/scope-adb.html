  1 -- scope.adb:  example of lexical scoping
  2 -- (Do not program this way; choose distinct names!)
  3 
  4 procedure Scope is
  5 
  6     A: Integer;    -- Scope.A
  7     B: Boolean;    -- Scope.B
  8     -- Declaration of "A" or "B" here would be illegal
  9 
 10     procedure One (A: in  Integer;
 11                    C: out Boolean) is
 12         D: Integer;    -- One.D
 13         E: Integer;    -- One.E
 14     begin
 15         -- A=One.A, B, C, D, E, Scope, One
 16         -- Outer, Inner:  UNDEFINED
 17         null;
 18     end One;
 19 
 20     procedure Outer is
 21         A: Integer;
 22         C: Integer;
 23 
 24         procedure Inner is
 25             D: Integer;
 26             C: Integer;
 27         begin
 28             -- A=Outer.A, B, C=Inner.C, D=Inner.D,
 29             --   Scope, One, Outer, Inner
 30             -- E:  UNDEFINED
 31             null;
 32         end Inner;
 33 
 34     begin
 35         -- A=Outer.A, B, C, Scope, One, Outer, Inner
 36         -- D, E:  UNDEFINED
 37         null;
 38     end Outer;
 39 
 40 begin
 41     -- A, B, Scope, One, Outer
 42     -- C, D, E, Inner:  UNDEFINED
 43     null;
 44 end Scope;
