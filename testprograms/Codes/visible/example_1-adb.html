  1 -- example_1.adb:  scope rules and subprocedures
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Float_Text_IO;
  4 use Ada;
  5 
  6 procedure Example_1 is
  7 
  8     A: constant Integer := 17;
  9     B: Integer;     -- first declaration of variable "B"
 10     C: Integer;
 11 
 12     procedure One (C: in Float) is
 13         -- A formal parameter, eg., "C", is like a local declaration
 14 
 15         -- A second declaration of variable "B"; does not matter what
 16         -- its type is, it hides the first declaration.
 17         B: Float;
 18     begin
 19         B := 2.3;                                    -- local "B"
 20         Integer_Text_IO.Put (Item=>A, Width =>1);    -- non-local "A"
 21         Float_Text_IO.Put (Item=>B, Fore=>3, Aft=>1, Exp=>0);
 22         Float_Text_IO.Put (Item=>C, Fore=>3, Aft=>1, Exp=>0);
 23     end One;
 24 
 25 begin
 26 
 27     B := 4;
 28     C := 6;
 29     One (C => 42.0);
 30     B := 23;
 31     C := 53;
 32 
 33 end Example_1;
