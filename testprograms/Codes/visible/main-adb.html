The Ada Program: main.adb

  1 -- main.adb:  illustrate the visibility of subprograms
  2 
  3 -- Visibility of subprograms is just the same as other names. A
  4 -- subprogram name as a prefix makes eclipsed names visible,
  5 -- but does not make otherwise inaccessible names visible
  6 -- a package name as a prefix is a different matter.
  7 
  8 procedure Main is
  9    X: Integer;
 10    procedure A is
 11       procedure B is
 12          X: Integer;
 13          procedure A is
 14          begin
 15             null;
 16          end A;
 17       begin
 18          X      := 45;
 19          Main.X := 87;
 20          A;
 21          Main.A;
 22       end B;
 23    begin
 24       null;
 25    end A;
 26    procedure C is
 27       procedure D is
 28       begin
 29          null;
 30       end D;
 31    begin
 32       D;
 33       A;
 34       -- It is not possible to call procedure "B", it is not
 35       -- visible.  The prefix "A" does not make it visible either.
 36       --   B;
 37       --   A.B;
 38    end C;
 39 begin
 40    null;
 41 end Main;
