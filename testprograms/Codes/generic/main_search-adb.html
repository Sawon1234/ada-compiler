  1 -- main_search.adb:  find all words of command line in input stream
  2 with
  3   Ada.Strings.Bounded,  -- bounded-string type (Ada95 A.4.4)
  4   Ada.Command_Line,     -- access to external execution env (Ada95 A.15)
  5   Ada.Integer_Text_IO,  -- preinstantiated generic for Integer IO
  6   Ada.Text_IO,
  7   Gen_Bin_Search;       -- user-defined generic binary search procedure
  8 use Ada;
  9 procedure Main_Search is
 10    package String_25 is new Strings.Bounded.Generic_Bounded_Length (25);
 11    use String_25;
 12 
 13    type Word_List is array (Positive range <>) of Bounded_String;
 14    Max_Number_Of_Words : constant := 100;
 15    List : Word_List (1..Max_Number_Of_Words);
 16    Number_Of_Words : Integer range 0 .. Max_Number_Of_Words := 0;
 17 
 18    procedure String_Search is new Gen_Bin_Search (
 19       Element_Type => Bounded_String,
 20       List_Type => Word_List);
 21 
 22    Location     : Natural;        -- index of value in "List"
 23    C : constant Natural := Command_Line.Argument_Count;
 24 
 25    Line: String (1..100);
 26    Length: Integer;
 27 begin
 28 
 29    -- read and keep all words from standard input
 30    while not (Text_IO.End_Of_File) loop
 31       Text_IO.Get_Line (Line, Length);
 32       Number_Of_Words := Number_Of_Words + 1;
 33       List (Number_Of_Words) := To_Bounded_String (Line (1..Length));
 34    end loop;
 35 
 36    -- search the list for each word in command line
 37    for I in 1..C loop
 38       String_Search (List (1..Number_Of_words),
 39                      To_Bounded_String (Command_Line.Argument(I)),
 40                      Location);
 41       Text_IO.Put ("Searching for """&Command_Line.Argument(I)&""".  ");
 42       if Location /= 0 then
 43          Text_IO.Put ("Found at location ");
 44          Integer_Text_IO.Put (Location, Width=>2);
 45          Text_IO.Put_Line (".");
 46       else
 47          Text_IO.Put_Line ("Not found in list.");
 48       end if;
 49    end loop;
 50 
 51 end Main_Search;
