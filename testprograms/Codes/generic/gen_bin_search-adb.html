  3 -- Find "Location" such that "List[Location]=Value", if none, set
  4 -- "Location" to 0.  We assume that "List" is sorted in ascending order
  5 
  6 procedure Gen_Bin_Search (List    : in List_Type;
  7                           Value   : in Element_Type;
  8                           Location: out Natural) is
  9 
 10    Found       : Boolean := False;
 11    First_Index : Natural := List'First;
 12    Last_Index  : Natural := List'Last;
 13    Midpoint    : Natural;
 14 
 15 begin
 16 
 17    while (First_Index <= Last_Index) and not Found loop
 18 
 19       Midpoint := (First_Index + Last_Index) / 2;
 20       if List(Midpoint) = Value then
 21          Found := True;
 22       elsif Value < List(Midpoint) then
 23          Last_Index := Midpoint - 1;
 24       else
 25          First_Index := Midpoint + 1;
 26       end if;
 27 
 28    end Loop;
 29 
 30    if Found then
 31       Location := Midpoint;
 32    else
 33       Location := 0;
 34    end if;
 35 
 36 end Gen_Bin_Search;
