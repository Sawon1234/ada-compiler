  8    function Preorder (T: Tree) return Item_Array is
  9    begin
 10       if (T/=null) then
 11          return (Preorder (T.Left) & Preorder (T.Right));
 12       else
 13          return (Null_Array);
 14       end if;
 15    end Preorder;

 17    procedure Insert (I: Integer; T: in out Tree) is
 18    begin
 19       if T=null then
 20          T := new Node'(I, null, null);
 21       elsif I<T.Item then
 22          Insert (I, T.Left);
 23       else
 24          Insert (I, T.Right);
 25       end if;
 26    end Insert;

 28    procedure Insert (A: Item_Array; T: in out Tree) is
 29    begin
 30       for I in A'Range loop
 31          Insert (A(I), T);
 32       end loop;
 33    end Insert;
