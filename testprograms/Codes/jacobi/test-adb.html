 10    function GCD (X, Y : Natural) return Natural is
 11    begin
 12       if Y=0 then
 13          return (X);
 14       else
 15          return (GCD (Y, X mod Y));
 16       end if;
 17    end GCD;
 18 
 19    function Jacobi_1 (A,N: Positive) return Integer is
 20       My_N : Positive := N;
 21       My_A : Positive := A;
 22       R:  Integer := 1;
 23       P:  Boolean;
 24    begin
 25       loop
 26          P := False;
 27          while My_A mod 2 = 0 loop My_A:=My_A/2; P:=Not P; end loop;
 28          if P then
 29             case My_N mod 8 is
 30               when 3|5 => R := -R;
 31               when others => null;
 32             end case;
 33          end if;
 34 
 35          exit when My_A=1;
 36 
 37          if My_N mod 4 = 3 and then My_A mod 4 = 3 then R := -R; end if;
 38          declare
 39             I : constant Integer := My_A;
 40          begin
 41             My_A := My_N mod My_A;  My_N := I;
 42          end;
 43       end loop;
 44       return (R);
 45    end Jacobi_1;
 46 
 47    function Jacobi_2 (A,N: Positive) return Integer is
 48       R: Integer;
 49    begin
 50       if A=1 then
 51          return 1;
 52       elsif A mod 2 = 0 then
 53          R := Jacobi_2 (A/2,N);
 54          --  R * ((-1) ** (((N**2)-1)/8))
 55          if (((N**2)-1)/8) mod 2 = 0 then return R; else return -R; end if;
 56       else
 57          R := Jacobi_2 (N mod A, A);
 58          -- R * ((-1) ** ((A-1)*(N-1)/4))
 59          if (((A-1)*(N-1))/4) mod 2 = 0 then return R; else return -R; end if;
 60       end if;
 61    end Jacobi_2;

 81       end if;
