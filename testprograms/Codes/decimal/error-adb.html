  1 -- error.adb: round-off error may be avoided using "decimal" type
  2 
  3 with Ada.Text_IO, Ada.Float_Text_IO, Ada.Text_IO.Editing;
  4 use Ada, Ada.Text_IO;
  5 
  6 -- Decimal values may not be represented exactly in binary, e.g.,
  7 --       0.03 = 2#0.00000111101011100001010001111010111#
  8 
  9 procedure Error is
 10    Three_Hundreths: constant Float := 0.03;
 11    Total: Float := 0.0;
 12 
 13    type Money is delta 0.01 digits 8;   -- decimal fixed-point type
 14    package Money_IO is new Text_IO.Editing.Decimal_Output (Num=>Money);
 15    Format:constant Editing.Picture:=Editing.To_Picture("BB$ZZZ_ZZ9.99");
 16    Three_Pennies: constant Money := 0.03;
 17    Dollars: Money := 0.0;
 18 
 19 begin
 20 
 21    for Count in 1..100_000 loop
 22       Total   := Total + Three_Hundreths;
 23       Dollars := Dollars + Three_Pennies;
 24    end loop;
 25 
 26    Float_Text_IO.Put (Total, Fore=>10, Aft=>2, Exp=>0);
 27    Text_IO.New_Line;
 28    Money_IO.Put (Dollars, Format);
 29    Text_IO.New_Line;
 30 
 31 end Error;
