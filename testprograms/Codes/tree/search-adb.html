 15    procedure DFS (T: Tree) is
 16       Stack: Stack_Package.Stack_Type;
 17       Node: Tree;
 18    begin
 19       Stack_Package.Clear (Stack);

 22          Stack_Package.Pop (Stack, Node);
 23          Integer_Text_IO.Put (Node.Item);
 24          Text_IO.New_Line;
 25          if (Node.Right /= null) then

 27          end if;
 28          if (Node.Left /= null) then
 29             Stack_Package.Push (Stack, Node.Left);
 30          end if;
 31       end loop;
 32    end DFS;

 48          end if;
 49          if (Node.Right /= null) then

 51          end if;
 52       end loop;
 53    end BFS;
