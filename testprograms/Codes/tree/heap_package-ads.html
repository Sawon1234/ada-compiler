  1 -- heap_package.ads:  implementation of a heap using an array
  2 
  3 generic
  4 
  5    type Element_Type is private;       -- The type of element in the heap
  6    type Heap_Array is array (Positive range <>) of Element_Type;
  7    with function ">" (Left, Right: Element_Type) return Boolean is <>;
  8 
  9 package Heap_Package is
 10 
 11    -- Assuming that the root of the heap is the only element out of
 12    -- place, restore the heap property:  not (Heap(Child) > Heap(Parent))
 13    -- Time complexity: O(log N)
 14    procedure Reheap_Down (Heap: in out Heap_Array);
 15 
 16 
 17    -- Assuming that the last leaf of the heap is the only element out of
 18    -- place, restore the heap property:  not (Heap(Child) > Heap(Parent))
 19    -- Time complexity: O(log N)
 20    procedure Reheap_Up (Heap: in out Heap_Array);
 21 
 22 
 23    -- Form a heap out of an array of objects.
 24    -- Time complexity: O(N log N)
 25    procedure Heapify (Heap: in out Heap_Array);
 26 
 27 end Heap_Package;
