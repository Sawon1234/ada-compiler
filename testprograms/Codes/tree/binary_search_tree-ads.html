  1 -- The elements in the tree are ordered from smallest to largest using
  2 -- the "<" function supplied as a generic parameter.  The placement of
  3 -- each element into the binary tree must satisfy the binary search
  4 -- property:  The value of the key of an element is greater than the
  5 -- value of the key of any element in its left subtree, and less than
  6 -- the value of the key of any element in its right subtree.
  7 
  8 generic
  9 
 10    type Element_Type is private;       -- The type of element in the list
 11    type Key_Type is limited private;   -- The type of key in the element
 12    with function Key_Of (Element: Element_Type) return Key_Type;
 13    with function "=" (Left, Right: Key_Type) return Boolean;
 14    with function "<" (Left, Right: Key_Type) return Boolean;
