  3 package body Heap_Package is
  4 
  5    procedure Swap (Left, Right : in out Element_Type) is
  6       Temp: constant Element_Type := Left;
  7    begin
  8       Left := Right;  Right := Temp;
  9    end Swap;
 10 
 11    -- The calculation assumes that the root of the heap is index 1
 12    function Left_Subtree (I: Positive) return Positive is
 13    begin
 14       return I*2;
 15    end Left_Subtree;
 16 
 17    -- The calculation assumes that the root of the heap is index 1
 18    function Right_Subtree (I: Positive) return Positive is
 19    begin
 20       return I*2 + 1;
 21    end Right_Subtree;
 22 
 23    -- The calculation assumes that the root of the heap is index 1
 24    function Parent_Index (I: Positive) return Positive is
 25    begin
 26       return I/2;
 27    end Parent_Index;
 28 
 29    -- Return the index of the root node's child with the larger key.
 30    function Larger_Child (Heap: Heap_Array;
 31                           Root: Positive) return Positive is
 32       Left_Child  : constant Positive := Left_Subtree (Root);
 33       Right_Child : constant Positive := Right_Subtree (Root);
 34    begin
 35       if Right_Child <= Heap'Last and then
 36         not (Heap(Left_Child) > Heap(Right_Child)) then
 37          return Right_Child;
 38       else
 39          -- There must be a left subtree, since the root is not a leaf
 40          -- and this is a heap.
 41          return Left_Child;
 42       end if;
 43    end Larger_Child;
 44 
 45    -- Assuming that the root of the heap is the only element out of
 46    -- place, restore the heap property:  not (Heap(Child) > Heap(Parent))
 47    procedure Reheap_Down (Heap : in out Heap_Array) is
 48       Root      : Positive := Heap'First;
 49       Max_Child : Positive;  -- Index of child with larger key
 50    begin
 51       while Root <= Heap'Last/2 loop
 52          Max_Child := Larger_Child (Heap, Root);
 53          exit when not (Heap(Max_Child)>Heap(Root));
 54          Swap (Heap(Max_Child), Heap(Root));
 55          Root := Max_Child;
 56       end loop;
 57    end Reheap_Down;
 58 
 59    -- Assuming that the last leaf of the heap is the only element out of
 60    -- place, restore the heap property:  not (Heap(Child) > Heap(Parent))
 61    procedure Reheap_Up (Heap : in out Heap_Array) is
 62       Bottom: Positive := Heap'Last;
 63       Parent: Positive;  -- Index of the parent of Bottom
 64    begin
 65       while Bottom > Heap'First loop
 66          Parent := Parent_Index (Bottom);
 67          exit when not (Heap(Bottom) > Heap(Parent));
 68          Swap (Heap(Parent), Heap(Bottom));
 69          Bottom := Parent;
 70       end loop;
 71    end Reheap_Up;
 72 
 73    procedure Heapify (Heap: in out Heap_Array) is
 74       Last_Interior_Node: constant Positive := Heap'Last / 2;
 75    begin
 76       pragma Assert (Heap'First=1);
 77       for I in reverse 1..Last_Interior_Node loop
 78          Reheap_Down (Heap(I..Heap'Last));
 79       end loop;
 80    end Heapify;
 81 
 82 end Heap_Package;
