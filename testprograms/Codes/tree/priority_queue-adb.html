  4 package body Priority_Queue is
  5 
  6    function Empty (Queue: Queue_Type) return Boolean is
  7    begin
  8       return Queue.Number = 0;
  9    end Empty;
 10    function Full (Queue: Queue_Type) return Boolean is
 11    begin
 12       return (Queue.Number = Queue.Max_Size);
 13    end Full;
 14    function Size (Queue: Queue_Type) return Natural is
 15    begin
 16       return Queue.Number;
 17    end Size;
 18    procedure Clear (Queue: in out Queue_Type) is
 19    begin
 20       Queue.Number := 0;
 21    end Clear;
 22 
 23    function ">" (Left, Right: Element_Type) return Boolean is
 24    begin
 25       return (Priority_Of (Left) > Priority_Of (Right));
 26    end ">";
 27 
 28    package Heap is new Heap_Package (Element_Type, Heap_Array);
 29 
 30    procedure Enqueue (Queue: in out Queue_Type;
 31                       Item : in     Element_Type) is
 32    begin
 33       if not Full (Queue) then
 34          Queue.Number := Queue.Number + 1;
 35          Queue.Elements(Queue.Number) := Item;
 36          Heap.Reheap_Up (Queue.Elements (1..Queue.Number));
 37       else
 38          raise Overflow;
 39       end if;
 40    end Enqueue;
 41 
 42    procedure Dequeue (Queue: in out Queue_Type;
 43                       Item : out     Element_Type) is
 44    begin
 45       if not Empty (Queue) then
 46          Item := Queue.Elements(1);
 47          Queue.Elements(1) := Queue.Elements(Queue.Number);
 48          Queue.Number := Queue.Number - 1;
 49          Heap.Reheap_Down (Queue.Elements (1..Queue.Number));
 50       else
 51          raise Underflow;
 52       end if;
 53    end Dequeue;
 54 
 55 end Priority_Queue;
