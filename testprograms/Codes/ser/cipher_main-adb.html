  6 --  It is unclear what is meant by:
  7 --    "If it is possible to decrypt the message uniquely ..."
  8 --  Suppose there is more than one linear extension of the total
  9 --  ordering, and that all the letters in the encrypted message are
 10 --  in the same place in all linear extensions, the message could be
 11 --  uniquely decripted, but the problem becomes much harder.
 12 
 13 with Ada.Text_IO, Ada.Integer_Text_IO, Cipher;
 14 use Ada;
 15 
 16 procedure Cipher_Main is
 17 
 18    Number, A, K: Integer;
 19 
 20 begin
 21 
 22    Integer_Text_IO.Get (Number);
 23    Text_IO.Skip_Line;
 24 
 25    Input: for I in 1 .. Number loop
 26       Integer_Text_IO.Get (A);
 27       Integer_Text_IO.Get (K);
 28       Text_IO.Skip_Line;
 29 
 30       Text_IO.New_Line;
 31       Text_IO.Put ("Case #");
 32       Integer_Text_IO.Put (I, Width=>0);
 33       Text_IO.New_Line;
 34 
 35       Cipher (A, K);
 36 
 37    end loop Input;
 38 
 39 end Cipher_Main;
