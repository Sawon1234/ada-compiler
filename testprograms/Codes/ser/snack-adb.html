  1 with Ada.Text_IO, Ada.Integer_Text_IO;
  2 use Ada;
  3 
  4 procedure Snack is
  5 
  6    Distance_Joe, Distance_Fido: Natural;
  7    Speed_Joe, Speed_Fido:  Natural;
  8 
  9    function Compare (A,B,C,D: Natural) return Boolean is
 10    begin
 11       return (A*D < C*B);   -- A/B < C/D
 12    end Compare;
 13 
 14 begin
 15 
 16    Input: loop

 21       exit Input when Distance_Joe = 0;

 25          Text_IO.Put_Line ("Joe makes it.");
 26       end if;
