  4 procedure Pizza (Constraints: Positive) is
  5 
  6    subtype Toppings is Character range 'A' .. 'P';
  7    package Set is new Bit_Set (Toppings);
  8    use type Set.Set_Type;
  9    subtype Ingrediant_Set is Set.Set_Type;
 10 
 11    procedure Put (Pizza: Ingrediant_Set) is
 12    begin
 13       for T in Toppings loop
 14          if Set.Is_Member (T, Pizza) then
 15             Text_IO.Put (T);
 16          end if;
 17       end loop;
 18       Text_IO.New_Line;
 19    end Put;
 20 
 21    subtype Index is Positive range 1..Constraints;
 22 
 23    Likes, Hates: array (Index) of Ingrediant_Set :=
 24       (others=>Set.Empty_Set);
 25 
 26    procedure Get_Constraint (I: Index) is
 27       C,T: Character;
 28    begin
 29       loop
 30          Text_IO.Get (C);
 31          exit when C=';';
 32          Text_IO.Get (T);
 33          if (C='+') then
 34             Set.Add (Likes(I),T);
 35          else
 36             Set.Add (Hates(I),T);
 37          end if;
 38       end loop;
 39       Text_IO.Skip_Line;
 40    end Get_Constraint;
 41 
 42    -- Check with nonlocal "Likes" and "Hates" array to see
 43    -- if constraing "I" is satisfied.
 44    function OK_With (Pizza: Ingrediant_Set; I: Index) return Boolean is
 45       X: constant Ingrediant_Set := Pizza and Likes(I);
 46       Y: constant Ingrediant_Set := (not Pizza) and Hates(I);
 47    begin
 48       return (X/=Set.Empty_Set or else Y/=Set.Empty_Set);
 49    end OK_With;
 50 
 51    function OK_With (Pizza: Ingrediant_Set) return Boolean is
 52    begin
 53       for I in Index loop
 54          if not Ok_With (Pizza, I) then
 55             return False;
 56          end if;
 57       end loop;
 58       return True;
 59    end OK_With;
 60 
 61    Pizza: Ingrediant_Set;
 62 
 63 begin
 64 
 65    for I in 1..Constraints loop
 66      Get_Constraint (I);
 67    end loop;
 68    Pizza := Set.Search (OK_With'Access);

 74       Text_IO.Put_Line (".  No pizza will satisfy everyone.");
