  1 with Ada.Text_IO, Ada.Integer_Text_IO;
  2 use Ada;

  4 procedure Bendy is
  5 
  6    Length, N : Natural;
  7    type Vector is record X,Y: Integer; end record;
  8    type JT is array (Positive range <>) of Positive;
  9    Joints : JT (1..20);
 10    type Dir is mod 4;
 11    North: constant Dir := 0;
 12 
 13    function "*" (Left: Dir; Right: Integer) return Vector is
 14       V: Vector := (X=>0, Y=>0);
 15    begin
 16       case Left is
 17          when 0 => V.Y :=  Right;
 18          when 1 => V.X :=  Right;
 19          when 2 => V.Y := -Right;
 20          when 3 => V.X := -Right;
 21       end case;
 22       return (V);
 23    end;
 24 
 25    function "+" (Left, Right: Vector) return Vector is
 26    begin
 27       return Vector'(X=>Right.X+Left.X, Y=>Right.Y+Left.Y);
 28    end;
 29 
 30    -- Given the diameter is 5 inches; using units of tenths of inches
 31    Radius_Squared: constant Integer := (50/2)**2;
 32 
 33    function InDisk (V: Vector) return Boolean is
 34    begin
 35       return (V.X*V.X+V.Y*V.Y <= Radius_Squared);
 36    end InDisk;
 37 
 38    function Fits (J: JT; L: Integer; V:Vector; D: Dir) return Boolean is
 39       V1: Vector;
 40       I1,L1: Integer;
 41    begin
 42       if (J'Length<=0) then
 43          -- Remainder of straw must fit in disk
 44          return InDisk (V + D*(Length-L));   -- "Length" global!
 45       else
 46          L1   := J(J'First+1);  -- length from the beginning
 47          I1   := L1-L;          -- length from last joint
 48          V1   := V + D*I1;      -- location of this joint
 49          if (not InDisk (V1)) then return False; end if;
 50          case J(J'First) is
 51             when 1 => return
 52                Fits (J(J'First+2..J'Last), L+I1, V1, D) or else
 53                Fits (J(J'First+2..J'Last), L+I1, V1, D-1);  -- left turn
 54             when 2 => return
 55                Fits (J(J'First+2..J'Last), L+I1, V1, D) or else
 56                Fits (J(J'First+2..J'Last), L+I1, V1, D+1);  -- right turn
 57             when others => return
 58                Fits (J(J'First+2..J'Last), L+I1, V1, D) or else
 59                Fits (J(J'First+2..J'Last), L+I1, V1, D-1) or else
 60                Fits (J(J'First+2..J'Last), L+I1, V1, D+1);
 61          end case;
 62       end if;
 63    end Fits;
 64 
 65 begin
 66    Input: loop
 67       Integer_Text_IO.Get (Length);
 68       exit when Length<=0;
 69       Integer_Text_IO.Get (N);
 70       for I in 1..N loop
 71          Integer_Text_IO.Get (Joints(2*(I-1)+1));  -- type of joint
 72          Integer_Text_IO.Get (Joints(2*I));        -- length; in tenths
 73       end loop;
 74       if (Fits (Joints (1..2*N), 0, Vector'(0,-25), North)) then
 75          Text_IO.Put_Line ("YES");
 76       else
 77          Text_IO.Put_Line ("NO");
 78       end if;
 79    end loop Input;
 80 end;
