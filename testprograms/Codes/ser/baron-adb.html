  1 -- baron.adb:  Prim-Jarník algorithm for minimal spanning tree
  2 
  3 with Ada.Integer_Text_IO, Ada.Text_IO, Priority_Queue;
  4 use Ada;
  5 
  6 procedure Baron is
  7    N, Count: Natural := 0;
  8 begin
  9    Input: loop
 10       Integer_Text_IO.Get (N);
 11       exit when N<=0;
 12       Count := Count + 1;
 13       Text_IO.Put ("Case ");
 14       Integer_Text_IO.Put (Count, Width=>0);
 15       Text_IO.Put_Line (":");
 16 
 17       declare
 18          type Node is new Positive range 1..N;
 19          type Edge is record To,From: Node; end record;
 20          Deg: array (Node) of Natural := (others=>0);
 21          Adj: array (Node, Node) of Natural;
 22          package Q is new Priority_Queue (Edge,Natural);
 23          X: Q.Pair;
 24          Sum: Natural := 0;  -- sum of all edges in spanning tree
 25 
 26          function All_Visited return Boolean is
 27          begin
 28             for I in Deg'Range loop
 29                if Deg(I)=0 then return False; end if;
 30             end loop;
 31             return True;
 32          end All_Visited;
 33 
 34          procedure Queue_Neighbors (V: Node) is
 35          begin
 36             for Neighbor in Node loop
 37                if Deg(Neighbor)=0 and then Adj(V,Neighbor)>0 then
 38                   Q.Enqueue(Edge'(From=>V,To=>Neighbor),Adj(V,Neighbor));
 39                end if;
 40             end loop;
 41          end Queue_Neighbors;
 42 
 43       begin
 44          for Row in Node loop
 45             for Col in Node loop
 46                Integer_Text_IO.Get (Adj (Row, Col));
 47             end loop;
 48          end loop;
 49 
 50          Queue_Neighbors (1);
 51 
 52          while not Q.Is_Empty loop
 53             X := Q.Dequeue;
 54             -- reject edges to nodes visited before, and reject those
 55             -- from nodes with 3 or more incident edges.
 56             if Deg(X.Element.To)=0 and then Deg(X.Element.From)<3 then
 57                -- add edge to spanning tree
 58                Deg(X.Element.From) := Deg(X.Element.From) + 1;
 59                Deg(X.Element.To)   := Deg(X.Element.To) + 1;
 60                Sum := Sum + X.P;
 61                Queue_Neighbors (X.Element.To);
 62             end if;
 63          end loop;
 64 
 65          if All_Visited then
 66             Integer_Text_IO.Put (Sum, Width=>0);
 67          else
 68             Text_IO.Put ("Cannot be constructed.");
 69          end if;
 70          Text_IO.New_Line (2);
 71       end;
 72    end loop input;
 73 
 74 end Baron;
