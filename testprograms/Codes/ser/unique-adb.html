The Ada Program: unique.adb

  1 with Ada.Text_IO, Ada.Strings.Unbounded;
  2 use Ada, Ada.Strings;
  3 
  4 package body Unique is
  5 
  6    -- global, ordered list of words, no duplicates
  7    Words: array (1..900) of Unbounded.Unbounded_String;
  8    Count: Natural:= 0;
  9 
 10    procedure Print_All is
 11    begin
 12       for I in Words'First .. Count loop
 13          Text_IO.Put_Line (Unbounded.To_String(Words(I)));
 14       end loop;
 15       Count := 0;
 16    end Print_All;
 17 
 18    procedure Add (Word: String) is
 19       I:Natural:=Words'First;
 20    begin
 21       while (I<=Count and then Unbounded.To_String (Words(I))<Word) loop
 22          I:=I+1;
 23       end loop;
 24 
 25       -- reject duplicates
 26       if I<=Count and then Unbounded.To_String(Words(I))=Word then
 27          return;  -- nothing to do, so leave
 28       end if;
 29 
 30       -- move all later words back one slot in the global "Words" array
 31       for J in reverse I ..Count loop
 32          Words(J+1) := Words(J);
 33       end Loop;
 34 
 35       -- insert "Word" at its proper position, "I"
 36       Words (I) := Unbounded.To_Unbounded_String (Word);
 37       Count := Count + 1;
 38    end Add;
 39 
 40 end Unique;
