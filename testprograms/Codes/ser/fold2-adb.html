  1 with
  2    Ada.Text_Io,          -- usual text (Latin 1) input/output (A.10.1)
  3    Ada.Integer_Text_Io;  -- preinstantiated generic for Integer IO
  4 use Ada;
  5 
  6 procedure Fold2 is
  7 
  8    In_File_Name: constant String := "fold.in";
  9    In_File: Text_IO.File_Type;
 10 
 11    function Fr (X:String) return String is
 12       Y: String (X'First..X'Last);
 13    begin
 14       for I in X'Range loop
 15          case X(X'Length-I+1) is
 16             when 'A' => Y(I) := 'V';
 17             when 'V' => Y(I) := 'A';
 18             when others => null;
 19          end case;
 20       end loop;
 21       return Y&"V"&X;
 22    end Fr;
 23 
 24    function F (N: Positive) return String is
 25    begin
 26       if N=1 then
 27          return "V";
 28       else
 29          return Fr (F(N-1));
 30       end if;
 31    end F;
 32    N: Natural;
 33 begin
 34    Text_IO.Open (File=>In_File, Mode=>Text_IO.In_File, Name=>In_File_Name);
 35    loop
 36       Integer_Text_IO.Get (In_File, N);
 37       exit when N=0;
 38       Text_Io.Put_Line (F(N));
 39    end loop;
 40 
 41 end Fold2;
