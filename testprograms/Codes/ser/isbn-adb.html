The Ada Program: isbn.adb

  1 -- ACM North-Central Regional Programming Contest 1994
  2 with Ada.Text_IO; use Ada;
  3 procedure ISBN is
  4 
  5    AI, Sum1, Sum2: Integer;
  6 
  7    procedure Add (C:Character) is
  8       D: constant Integer := Character'Pos(C)-Character'Pos('0');
  9    begin
 10       AI:=AI+1;     -- keep track of length of ISBN
 11       Sum1 := Sum1 + D;
 12       Sum2 := Sum2 + Sum1;
 13    end Add;
 14 
 15    procedure AddX is
 16    begin
 17       AI:=AI+1;
 18       if AI = 10 then
 19          Sum1 := Sum1 + 10;
 20          Sum2 := Sum2 + Sum1;
 21       else
 22          AI := 999;   -- make ISBN illegal
 23       end if;
 24    end AddX;
 25 
 26    Buffer: String (1..80);
 27    Length, F, L: Integer;
 28 
 29 begin
 30    Input: while not (Text_IO.End_Of_File) loop
 31       Text_IO.Get_Line (Buffer, Length);
 32       AI := 0;  Sum1 := 0;  Sum2 := 0;
 33 
 34       -- trim leading and trailing blanks
 35       F := Buffer'First;
 36       while (F<=Length and then Buffer(F)=' ') loop F:=F+1; end loop;
 37       L  := Length;
 38       while (L>=F  and then Buffer(L)=' ')   loop L:=L-1; end loop;
 39 
 40       for I in F .. L loop
 41          case Buffer (I) is
 42             when '-'    => null;
 43             when '0'..'9' => Add (Buffer(I));
 44             when 'X'    => AddX;
 45             when others => AI := 999;
 46          end case;
 47       end loop;
 48 
 49       if (AI=10 and Sum2 mod 11=0) then
 50          Text_IO.Put_Line (Buffer(F..L)& " is correct.");
 51       else
 52          Text_IO.Put_Line (Buffer(F..L)& " is incorrect.");
 53       end if;
 54    end loop Input;
 55 end ISBN;
