  1 with Ada.Text_IO, Ada.Integer_Text_IO, Discrete_Set;
  2 use Ada;
  3 procedure Counter is
  4 
  5    subtype Coins is Character range 'A' .. 'L';
  6    package Set is new Discrete_set (Coins);
  7    subtype Coin_Set is Set.Set_Type;
  8    use type Coin_Set;
  9 
 10    procedure Put (CS: Coin_Set) is
 11    begin
 12       for T in Coins loop
 13          if Set.Is_Member (T, CS) then
 14             Text_IO.Put (T);
 15          end if;
 16       end loop;
 17    end Put;
 18 
 19    Less, More: Coin_Set := Set.Universal_Set;
 20 
 21    procedure Get_Line is
 22       C: Character;
 23       Even, Left, Right: Coin_Set := Set.Empty_Set;
 24    begin
 25       Text_IO.Get (C);
 26       while C in Coins loop Set.Add (Left,C); Text_IO.Get(C); end loop;
 27       while C=' ' loop Text_IO.Get (C); end loop;  -- skip blanks
 28       while C in Coins loop Set.Add (Right,C); Text_IO.Get(C); end loop;
 29       while C=' ' loop Text_IO.Get (C); end loop;  -- skip blanks
 30       if C='e' then
 31          Even := Even or Left or Right;
 32       elsif C='u' then
 33          Less := Less and Right;   More := More and Left;
 34       else
 35          Less := Less and Left;    More := More and Right;
 36       end if;
 37       Less := Less-Even;   More := More-Even;
 38       Text_IO.Skip_Line;
 39    end Get_Line;
 40 
 41 begin
 42    for I in 1..3 loop Get_Line; end loop;
 43 
 44    if Set.Cardinality(More)=1 and then Set.Cardinality(Less)/=1 then
 45       Put (More);
 46       Text_IO.Put_Line (" is the counterfeit coin and it is heavy.");
 47    elsif Set.Cardinality(Less)=1 and then Set.Cardinality(More)/=1 then
 48       Put (Less);
 49       Text_IO.Put_Line (" is the counterfeit coin and it is light.");
 50    end if;
 51 end Counter;
