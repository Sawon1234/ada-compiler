The Ada Program: jeep.adb

  1 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Float_Text_IO;
  2 use Ada;
  3 
  4 procedure Jeep is
  5 
  6    function Dist (
  7       N: Positive;  -- number of cans of gas, each containing C liters
  8       C: Float;     -- capacity of can and tank of jeep
  9       R: Float      -- rate of consumption in liters per 100 kilometers
 10    ) return Float is
 11 
 12       -- distance in kilometers the jeep can travel on one tank of gas
 13       Distance: Float := (C*Float(100)/R);
 14 
 15       -- journeys needed to move N-1 cans of gas (jeep can carry only 1)
 16       Journeys: Positive;
 17 
 18    begin
 19       if N=1 then
 20          return (Distance);
 21       else
 22          -- back and forth for N-2 cans; 1 outbound journey for last can
 23          Journeys := 2*(N-2) + 1;
 24          return (Distance/Float(Journeys) + Dist (N-1, C, R));
 25       end if;
 26    end Dist;
 27 
 28    Count: Natural:=0;
 29    Number: Integer;
 30    Capacity, Rate: Float;
 31 
 32 begin
 33 
 34    Input: loop
 35       Integer_Text_IO.Get (Number);
 36       exit Input when Number<=0;
 37       Count := Count + 1;
 38       Float_Text_IO.Get (Capacity);
 39       Float_Text_IO.Get (Rate);
 40       Integer_Text_IO.Put (Count, Width=>0);
 41       Text_IO.Put (". Distance is ");
 42       Float_Text_IO.Put (Dist (Number, Capacity, Rate), Exp=>0, Aft=>2);
 43       Text_IO.Put_Line (" kilometers");
 44    end loop Input;
 45 
 46 end Jeep;
