  4 procedure Binary (Number_Of_Pairs, Number_Of_Elements: Natural) is
  5 
  6    package Rel is new Relationships (N=>Number_Of_Elements);
  7    A: Rel.Relation := (others=>(others=>False));
  8 
  9    procedure Get (A: out Rel.Relation) is
 10       function El (C: Character) return Positive is
 11       begin
 12          return (Character'Pos(C) - Character'Pos('a') + 1);
 13       end El;
 14       D,R: Character;
 15    begin
 16       for K in 1..Number_Of_Pairs loop

 18          A(El(D),El(R)) := True;
 19       end loop;
 20    end Get;
 21 
 22    procedure Put (C: Boolean; S: String; M: Boolean:=False) is
 23    begin
 24       if C then
 25          Text_IO.Put (S);
 26          if M then
 27             Text_IO.Put (", ");
 28          end if;
 29       end if;
 30    end Put;
 31 
 32    R,S,T: Boolean;
 33 
 34 begin
 35    Get (A);
 36    R := Rel.Reflexive(A);  S := Rel.Symmetric(A); T := Rel.Transitive(A);
 37 
 38    if R or S or T then
 39       Text_IO.Put ("is ");
 40       Put (R, "reflexive", S or T);
 41       Put (S, "symmetric", T);
 42       Put (T, "transitive");
 43    else
 44       Text_IO.Put ("does not have any of the properties");
 45    end if;
