The Ada Program: cards.adb

  1 with Ada.Text_IO, Ada.Integer_Text_IO;
  2 use Ada;
  3 
  4 package body Cards is
  5 
  6    procedure Write (S: in Stack_Pointer) is
  7    begin
  8       if S=null then
  9          Text_IO.Put ("NULL");
 10       else
 11          for I in S'Range loop
 12             Integer_Text_IO.Put (S(I), Width=>3);
 13          end loop;
 14       end if;
 15       Text_IO.New_Line;
 16    end Write;
 17 
 18    procedure Write (St: in State) is
 19    begin
 20       Text_IO.Put ("Current=");
 21       Integer_Text_IO.Put (Integer(St.Current), Width=>0);
 22       Text_IO.New_Line;
 23       for I in St.Stacks'Range loop
 24          Integer_Text_IO.Put (Integer(I), Width=>4);
 25          Write (St.Stacks(I));
 26       end loop;
 27       Text_IO.Put ("Deck=");
 28       Write (St.Deck);
 29    end Write;
 30 
 31    procedure Read (S: out Stack_Pointer) is
 32    begin
 33       S := new Stack (1..52);
 34       for I in S'Range loop
 35          Integer_Text_IO.Get (S(I));
 36       end loop;
 37    end Read;
 38 
 39    procedure Read (S: out State) is
 40    begin
 41       Read (S.Deck);
 42    end Read;
 43 
 44    function "&" (Left: Stack; Right: Spots) return Stack is
 45       One: Stack (1..1) := (1=>Right);
 46    begin
 47       return (Left & One);
 48    end "&";
 49 
 50    procedure Deal_One (St: in out State) is
 51    begin
 52       St.Stacks (St.Current) := new Stack'(St.Stacks(St.Current).all & St.Deck(St.Deck'First));
 53       St.Deck := new Stack'(St.Deck(St.Deck'First+1 .. St.Deck'Last));  -- deal from top
 54       St.Current := S'Succ(St.Current);
 55    end Deal_One;
 56 
 57    function Nth (S: State; N: Natural:=0; Last:Boolean:=false) return Natural is
 58       Pile: Stack_Pointer := S.Stacks(S.Current);
 59    begin
 60       if (Last) then
 61          return Natural (Pile (Pile'Last-N));
 62       else
 63          return Natural (Pile (Pile'First+N));
 64       end if;
 65    end Nth;
 66 
 67    function Nth (Pile: Stack_Pointer; N: Natural:=0; Last:Boolean:=false) return Natural is
 68    begin
 69       if (Last) then
 70          return Natural (Pile (Pile'Last-N));
 71       else
 72          return Natural (Pile (Pile'First+N));
 73       end if;
 74    end Nth;
 75 
 76    function Cond_1 (S: State) return Natural is
 77    begin
 78       return (Nth (S,0)+Nth(S,1)+Nth(S,0,True));
 79    end Cond_1;
 80 
 81    function Cond_1 (Pile: Stack_Pointer) return Natural is
 82    begin
 83       return (Nth (Pile,0)+Nth(Pile,1)+Nth(Pile,0,True));
 84    end Cond_1;
 85 
 86    function Cond_2 (S: State) return Natural is
 87    begin
 88       return (Nth (S,0)+Nth(S,0,True)+Nth(S,1,True));
 89    end Cond_2;
 90 
 91    function Cond_3 (S: State) return Natural is
 92    begin
 93       return (Nth (S,0,True)+Nth(S,1,True)+Nth(S,2,True));
 94    end Cond_3;
 95 
 96    function Deal (S: State) return State is
 97       R: State := S;
 98    begin
 99       for I in R.Stacks'Range loop
100          R.Stacks (I) := new Stack (1..0);
101          Deal_One (R);
102       end loop;
103       Write (R);
104       return R;
105    end Deal;
106 
107 end Cards;
