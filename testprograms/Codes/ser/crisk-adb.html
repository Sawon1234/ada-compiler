The Ada Program: crisk.adb

  1 with Ada.Text_IO, Ada.Integer_Text_IO; use Ada;
  2 procedure Crisk (Countries: Positive) is
  3    --  Inf is > longest path (=Counties-1), much less than Natural'Last.
  4    Infinity : constant Natural := Countries;
  5    Dist: array (1..Countries, 1..Countries) of Natural :=
  6       (others=>(others=>Infinity));  -- all countries unconnected
  7    Via_K: Natural;
  8    B,C,Q,O,D: Integer;   -- input variables
  9 begin
 10    -- Get adjacency information
 11    for I in 1 .. Countries-1 loop
 12       Integer_Text_IO.Get (B);            -- B boundaries comming up
 13       for J in 1 .. B loop
 14          Integer_Text_IO.Get (C);
 15          Dist(I,C):= 1;  Dist(C,I):= 1;   -- I adjoins C and vice versa
 16       end loop;
 17       Text_IO.Skip_Line;                  -- ignore the rest of the line
 18    end loop;
 19    -- Initialize diagonal entries of "Dist" matrix
 20    for I in 1 .. Countries loop
 21       Dist(I,I) := 0;   -- distance to self is 0
 22    end loop;
 23 
 24    -- Find shortest path between and pairs of countries
 25    for K in 1 .. Countries loop
 26       for I in 1 .. Countries loop
 27          for J in I+1 .. Countries loop
 28             -- Is it shorter to go from I to J via K?
 29             Via_K := Dist(I,K)+Dist(K,J); -- be careful adding infinities
 30             if Via_K < Dist(I,J) then
 31                Dist(J,I):=Via_K; Dist(I,J):=Via_K;
 32             end if;
 33          end loop;
 34       end loop;
 35    end loop;
 36 
 37    -- Get queries (pairs of country numbers) and print responses
 38    Integer_Text_IO.Get (Q);
 39    for J in 1..Q loop
 40       Integer_Text_IO.Get (O);   Integer_Text_IO.Get (D);
 41       Integer_Text_IO.Put (O, Width=>0);  Text_IO.Put (" to ");
 42       Integer_Text_IO.Put (D, Width=>0);  Text_IO.Put (": ");
 43       Integer_Text_IO.Put (Dist(O,D), Width=>0);
 44       Text_IO.New_Line;
 45    end loop;
 46 end Crisk;
