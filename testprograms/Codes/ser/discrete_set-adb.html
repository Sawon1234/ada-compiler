  3 package body Discrete_Set is
  4 
  5    -- This function determines whether Element is a member of Set
  6    function Is_Member (Element : Element_Type;
  7                        Set     : Set_Type) return Boolean is
  8    begin
  9       return Set(Element);
 10    end Is_Member;
 11 
 12    function Cardinality (Set : Set_Type) return Natural is
 13       Count: Natural := 0;
 14    begin
 15       for T in reverse Element_Type loop
 16          if Set(T) then Count := Count + 1; end if;
 17       end loop;
 18       return Count;
 19    end Cardinality;
 20 
 21    -- Adds an element to a set
 22    procedure add (Left: in out Set_Type; Right: Element_Type) is
 23    begin
 24       Left(Right) := True;   -- Add the new element to the set
 25    end add;
 26 
 27    -- Returns the intersection of the two sets
 28    function "and" (Left, Right: Set_Type) return Set_Type is
 29    begin
 30       return (Standard."and" (Left, Right));
 31    end "and";
 32 
 33    -- Returns the union of the two sets
 34    function "or" (Left, Right: Set_Type) return Set_Type is
 35    begin
 36       return (Standard."or" (Left, Right));
 37    end "or";
 38 
 39    function "-" (Left, Right: Set_Type) return Set_Type is
 40    begin
 41       return (Left and not Right);
 42    end "-";
 43 
 44    function "not" (Right: Set_Type) return Set_Type is
 45    begin
 46       return (Standard."not"(Right));
 47    end "not";
 48 
 49    function Search (F: Search_Criterion) return Set_Type is
 50       Carry, Temp: Boolean;
 51       Set : Set_Type:= Empty_Set;
 52    begin
 53       loop
 54          if F(Set) then
 55             return (Set);
 56          end if;
 57 
 58          exit when Set=Universal_Set;
 59 
 60          -- next set
 61          Carry := True;
 62          for T in reverse Element_Type loop
 63             Temp    := Set(T) xor Carry;
 64             Carry   := Set(T) and Carry;
 65             Set(T)  := Temp;
 66             exit when not Carry;  -- add complete
 67          end loop;
 68 
 69       end loop;
 70       raise Search_Fail;
 71    end Search;
 72 
 73 end Discrete_Set;
