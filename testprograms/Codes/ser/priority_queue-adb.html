  1 package body Priority_Queue is
  2 
  3    The_Queue: array (1..500) of Pair;
  4    End_Of_Queue : Natural := 0;
  5 
  6    function Is_Empty return Boolean is
  7    begin
  8       return (End_Of_Queue = 0);
  9    end Is_Empty;

 11    function Dequeue return Pair is
 12        X: Pair;
 13    begin
 14       if Is_Empty then raise Queue_Empty; end if;
 15       X := The_Queue(1);
 16       for I in 2..End_Of_Queue loop
 17          The_Queue(I-1) := The_Queue(I);
 18       end loop;
 19       End_Of_Queue := End_Of_Queue - 1;
 20       return (X);
 21    end Dequeue;
 22 
 23    procedure Enqueue (New_Element: Element_Type; P: Priority) is
 24       N: Pair := (Element=>New_Element,P=>P);
 25       I: Natural:=1;
 26    begin
 27       while (I<=End_Of_Queue and then The_Queue(I).P <= P) loop
 28          I:=I+1;
 29       end loop;
 30       for J in reverse I..End_Of_Queue loop
 31          The_Queue(J+1) := The_Queue(J);
 32       end loop;
 33       The_Queue(I) := N;
 34       End_Of_Queue := End_Of_Queue + 1;
 35    end Enqueue;
 36 
 37 end Priority_Queue;
