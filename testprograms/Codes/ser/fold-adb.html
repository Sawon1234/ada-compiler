  1 with
  2    Ada.Command_Line,     -- access to external execution env (Ada95 A.15)
  3    Ada.Text_Io,          -- usual text (Latin 1) input/output (A.10.1)
  4    Ada.Integer_Text_Io;  -- preinstantiated generic for Integer IO
  5 use Ada;
  6 
  7 procedure Fold is
  8    N:constant Natural := Natural'Value (Command_Line.Argument(1));
  9 
 10    function Fr (X:String) return String is
 11       Y: String (X'First..X'Last);
 12    begin
 13       for I in X'Range loop
 14          case X(X'Length-I+1) is
 15             when 'A' => Y(I) := 'V';
 16             when 'V' => Y(I) := 'A';
 17             when others => null;
 18          end case;
 19       end loop;
 20       return Y&"V"&X;
 21    end Fr;
 22 
 23    function F (N: Positive) return String is
 24    begin
 25       if N=1 then
 26          return "V";
 27       else
 28          return Fr (F(N-1));
 29       end if;
 30    end F;
 31 begin
 32    Text_Io.Put_Line (F(N));
 33 end Fold;
