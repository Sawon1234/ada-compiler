The Ada Program: bit_set.ads

  1 -- bit_set.ads:  generic package for sets of discrete elements
  2 with Interfaces;
  3 generic
  4 
  5    type Element_Type is (<>);          -- any discrete type
  6 
  7 package Bit_Set is
  8 
  9    type Set_Type is private;
 10 
 11    Empty_Set     : constant Set_Type;  -- Set with no elements
 12    Universal_Set : constant Set_Type;  -- Set with all elements
 13 
 14    -- This function determines whether Element is a member of Set
 15    function Is_Member (Element : Element_Type;
 16                        Set     : Set_Type) return Boolean;
 17 
 18    -- Adds an element to a set
 19    procedure add (Left: in out Set_Type; Right: Element_Type);
 20 
 21    -- Returns the intersection of the two sets
 22    function "and" (Left, Right: Set_Type) return Set_Type;
 23 
 24    -- Returns the complement of the set
 25    function "not" (Right: Set_Type) return Set_Type;
 26 
 27    type Search_Criterion is access function (X: Set_Type) return Boolean;
 28 
 29    Search_Fail: exception;
 30 
 31    function Search (F: Search_Criterion) return Set_Type;
 32 
 33 private
 34 
 35    --  'Pos attribute of Element_Type'First may not be 0
 36    Position_Of_First : constant Natural :=
 37       Element_Type'Pos (Element_Type'First);
 38 
 39    Number_Of_Elements: constant Natural :=
 40       Element_Type'Pos (Element_Type'Last) - Position_Of_First + 1;
 41 
 42    -- We hope "Number_Of_Elements" <= 64, but we can't check.
 43    type Set_Type is new Interfaces.Unsigned_64;  -- same as "mod 2**64"
 44 
 45    -- Completion of deferred constants
 46    Empty_Set         : constant Set_Type := Set_Type(0);
 47    Universal_Set     : constant Set_Type := 2 ** Number_Of_Elements - 1;
 48 
 49 end Bit_Set;
