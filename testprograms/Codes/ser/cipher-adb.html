The Ada Program: cipher.adb

  1 with Ada.Text_IO;
  2 use Ada;
  3 
  4 procedure Cipher (A, K: Positive) is
  5 
  6    -- The number A determines that last letter in the alphabet
  7    Last: constant Character := Character'Val (Character'Pos('a') + A-1);
  8    subtype Alpha is Character range 'a' .. Last;
  9 
 10    type Str is access String;
 11 
 12    procedure Get (X: out Str) is
 13       Buffer: String (1..100);
 14       Last: Natural;
 15    begin
 16       Text_IO.Get_Line (Buffer, Last);
 17       X := new String' (Buffer(1..Last));
 18    end;
 19 
 20    Order: array (Alpha, Alpha) of Boolean := (others=>(others=>False));
 21 
 22    procedure Lex (X,Y: Str) is
 23       J: Integer := X'First;  K: Integer := Y'First;
 24    begin
 25       while (J<=X'Last and then K<=Y'Last and then X(J)=Y(K)) loop
 26          J := J+1;  K := K+1;
 27       end loop;
 28       if (J>X'Last or else K>Y'Last) then
 29          return;  -- we learned nothing
 30       else
 31          Order (X(J),Y(K)) := True;
 32          Text_IO.Put_Line (X(J)&" is less than "&Y(K));
 33       end if;
 34    end Lex;
 35 
 36 
 37    Key: array (Alpha) of Alpha;
 38    Ambiguous: exception;
 39 
 40    procedure Top is
 41 
 42       InDegree: array (Alpha) of Integer;
 43 
 44       function FindInDegree (A: Alpha) return Natural is
 45          Count: Natural := 0;
 46       begin
 47          for C in Alpha loop
 48             if Order (C,A) then Count:=Count+1; end if;
 49          end loop;
 50          return Count;
 51       end FindInDegree;
 52 
 53       procedure Remove_Edges (A: Alpha) is
 54       begin
 55          for C in Alpha loop
 56             if Order (A,C) then InDegree(C):=InDegree(C)-1; end if;
 57          end loop;
 58       end Remove_Edges;
 59 
 60       Count: Natural;
 61 
 62    begin
 63       for U in Alpha loop InDegree(U):=FindInDegree(U); end loop;
 64       for C in Alpha loop
 65          Count := 0;
 66          for Z in Alpha loop
 67             if (InDegree(Z)=0) then Count:=Count+1; end if;
 68          end loop;
 69          if Count/=1 then
 70             raise Ambiguous;  -- give up right away
 71          end if;
 72          for Z in Alpha loop
 73             if InDegree(Z)=0 then
 74                -- We now know that Z stands for C in the cipher
 75                Key(Z):=C;
 76                Remove_Edges (Z);  InDegree(Z):=-1;
 77                exit;
 78             end if;
 79          end loop;
 80       end loop;
 81    end Top;
 82 
 83    procedure Decode (X: in out String) is
 84    begin
 85       for I in X'Range loop
 86          if X(I) in Alpha then
 87             X(I) := Key(X(I));
 88          end if;
 89       end loop;
 90    end Decode;
 91 
 92    Previous, Current, Encrypted: Str;
 93 
 94 begin
 95    -- Get the K encrypted words
 96    Get (Previous);
 97    for I in 2..K loop
 98       Get (Current);
 99       Lex (Previous, Current);
100       Previous := Current;
101    end loop;
102 
103    -- Get the encrypted string
104    Get (Encrypted);
105 
106    -- Do a topological sort
107    Top;
108 
109    Text_IO.Put_Line ("Encrypted text: " & Encrypted.all);
110    Decode (Encrypted.all);  -- decode encrypted string in place
111    Text_IO.Put_Line ("Plain text:     " & Encrypted.all);
112 
113 exception
114    when Ambiguous => Text_IO.Put_Line ("Message cannot be decrypted.");
115 end Cipher;
