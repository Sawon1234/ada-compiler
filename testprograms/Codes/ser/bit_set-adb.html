The Ada Program: bit_set.adb

  1 -- bit_set.ads:  generic package for sets of discrete elements
  2 
  3 -- Using "Shift_Left(1,Amount)" makes "Pizza" run about twice as
  4 -- fast as using "2**Amount".
  5 
  6 with Interfaces; use Interfaces;
  7 
  8 package body Bit_Set is
  9 
 10    function Singleton (Element: Element_Type) return Set_Type is
 11    begin
 12       return Shift_Left (1, Element_Type'Pos(Element)-Position_Of_First);
 13    end Singleton;
 14 
 15    -- Determine whether Element is a member of Set
 16    function Is_Member (Element : Element_Type;
 17                        Set     : Set_Type) return Boolean is
 18       Inter: constant Set_Type := Set and Singleton (Element);
 19    begin
 20       return (Inter /= 0);
 21    end Is_Member;
 22 
 23    -- Adds an element to a set
 24    procedure add (Left: in out Set_Type; Right: Element_Type) is
 25    begin
 26       Left := Left or Singleton (Right);
 27    end add;
 28 
 29    -- Returns the intersection of the two sets
 30    function "and" (Left, Right: Set_Type) return Set_Type is
 31    begin
 32       return (Standard."and" (Left, Right));
 33    end "and";
 34 
 35    function "not" (Right: Set_Type) return Set_Type is
 36    begin
 37       return (Standard."not"(Right));
 38    end "not";
 39 
 40    function Search (F: Search_Criterion) return Set_Type is
 41    begin
 42       for Set in Empty_Set .. Universal_Set loop
 43          if F(Set) then return (Set); end if;
 44       end loop;
 45       raise Search_Fail;
 46    end Search;
 47 
 48 begin
 49    if Number_Of_Elements > 64 then raise Constraint_Error; end if;
 50 end Bit_Set;
