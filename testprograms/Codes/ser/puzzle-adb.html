  6    procedure All_Words (Line: String) is
  7       Buffer: String (1..100);
  8       Count: Natural := 0;
  9    begin
 10       for I in Line'Range loop
 11          -- we require all words, even those at the end of the line,
 12          -- to end with a '_' character.
 13          if Line(I)='_' then
 14             if Count>1 then
 15                Unique.Add (Buffer (1..Count));
 16             end if;
 17             Count:=0;
 18          else
 19             Count := Count + 1;
 20             Buffer (Count) := Line(I);
 21          end if;
 22       end loop;
 23    end All_Words;

 33

 36       Text_IO.New_Line;
 37 
 38       declare
 39          Grid: array (1..Number_Of_Rows, 1..Number_Of_Cols) of Character;
 40          Row_Buffer: String (1..Number_Of_Cols+1) := (others=>'_');
 41          Col_Buffer: String (1..Number_Of_Rows+1) := (others=>'_');
 42       begin
 43          for Row in Grid'Range(1) loop
 44             for Col in Grid'Range(2) loop
 45                Text_IO.Get (Grid(Row,Col));

 52                Row_Buffer (Col) := Grid(Row,Col);
 53             end loop;

 55          end loop;

 60             end loop;

 62          end loop;
