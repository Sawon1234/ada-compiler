The Ada Program: trains.adb

  1 with Ada.Integer_Text_IO, Ada.Text_IO;
  2 use Ada;
  3 
  4 procedure Trains is
  5    Distance, Speed_1, Speed_2, Time_1, Time_2, Simulation_End: Natural;
  6    type Interval is record Start,Stop: Natural; end record;
  7 
  8    function Intersection (Left, Right: Interval) return Boolean is
  9       Contemp:  constant Boolean := Right.Start <= Left.Stop or else Left.Start <= Right.Stop;
 10       Complete: constant Boolean := Left.Stop<=Simulation_End or else Right.Stop<=Simulation_End;
 11       D1, D2: Natural;
 12    begin
 13       D1 := Speed_2 * (Simulation_End - Left.Start);
 14       D2 := Speed_1 * (Simulation_End - Right.Start);
 15       -- The trains must have had some part of their journey at the same time,
 16       -- in order to have a chance of passing.  If either completed the journey
 17       -- then they must have passed.  If neither completed the journey before
 18       -- the simulation ends, then to have reached each other, the total distance
 19       -- they traveled must greater or equal to the length of the track.
 20       return (Contemp and then (Complete or else D1+D2>=Distance));
 21    end Intersection;
 22 
 23    function Nth (N, Depart, Length: Natural) return Interval is
 24       X: constant Natural := N * Depart;
 25       I: Interval := (Start=>X, Stop=>X+Length);
 26    begin
 27       return (I);
 28    end Nth;
 29 
 30    Units, LJ_1, LJ_2: Natural;
 31 
 32    function Rendezvous return Natural is
 33       Count, N_1, N_2 : Natural := 0;
 34    begin
 35       -- trains depart at regular intervals one way
 36       while N_1 * Time_1 <= Simulation_End loop
 37          N_2 := 0;
 38          -- trains depart at regular intervals going the other way
 39          while N_2 * Time_2 <= Simulation_End loop
 40             if Intersection (Nth (N_1, Time_1, LJ_1), Nth (N_2, Time_2, LJ_2)) then
 41                Count := Count + 1;
 42             end if;
 43             N_2 := N_2 + 1;
 44          end loop;
 45          N_1 := N_1 + 1;
 46       end loop;
 47       return Count;
 48    end Rendezvous;
 49 
 50 begin
 51    Input: loop
 52       Integer_Text_IO.Get (Distance);
 53 
 54       exit when Distance <= 0;
 55 
 56       Integer_Text_IO.Get (Speed_1);
 57       Integer_Text_IO.Get (Speed_2);
 58 
 59       -- Instead of dividing distance by speed to get time, we convert everthing to
 60       -- units of the multiple of the speeds.  Now all calculations are in whole integers.
 61       Units := Speed_1 * Speed_2;
 62       LJ_1 := Speed_2 * Distance;   -- Length of journey for trains going one way ...
 63       LJ_2 := Speed_1 * Distance;   -- and trains going the other way (in funny units).
 64       Distance := Units * Distance; -- Total distance between cities (in funny units).
 65 
 66       Integer_Text_IO.Get (Time_1);
 67       Time_1 := Units * Time_1;
 68 
 69       Integer_Text_IO.Get (Time_2);
 70       Time_2 := Units * Time_2;
 71 
 72       Integer_Text_IO.Get (Simulation_End);
 73       Simulation_End := Units * Simulation_End;
 74 
 75       --  Compute and print number of rendezvous
 76       Integer_Text_IO.Put (Rendezvous, Width=>0);
 77       Text_IO.New_Line;
 78 
 79    end loop Input;
 80 end Trains;
