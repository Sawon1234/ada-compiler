The Ada Program: vending.adb

  1 -- vending.adb:  example finite state machine for vending machine
  2 with Ada.Text_IO, Ada.Integer_Text_IO; use Ada, Ada.Text_IO;
  3 
  4 procedure Vending is
  5 
  6    type Input_Type is (Push, Nickel, Dime);
  7    package Input_IO is new Text_IO.Enumeration_IO (Enum=>Input_Type);
  8    Input_Action  : Input_Type;
  9 
 10    type State_Type is (Zero, Five, Ten, Fifteen);
 11    State : State_Type := Zero;
 12 
 13    Transition : constant array (Input_Type, State_Type) of State_Type :=
 14       (Push   => (Zero, Five, Ten, Zero),
 15        Nickel => (Five, Ten, Fifteen, Fifteen),
 16        Dime   => (Ten, Fifteen, Ten, Fifteen));
 17 begin
 18 
 19    Input: loop
 20 
 21       Text_IO.Put ("Enter an input action -- ");
 22       exit when Text_IO.End_Of_File;
 23       Input_IO.Get (Input_Action);
 24       Text_IO.Skip_Line;
 25 
 26       case Input_Action is
 27          when Push =>
 28             if State=Fifteen then    -- item costs 15¢
 29                Put_Line ("Item dispensed.");
 30             else
 31                Put_Line ("Not enough money deposited!");
 32             end if;
 33          when Nickel =>
 34             if State=Fifteen then
 35                Put_Line ("Already enough money; nickel returned.");
 36             end if;
 37          when Dime =>
 38             if State=Ten or else State=Fifteen then
 39                Put_Line ("Already enough money; dime returned.");
 40             end if;
 41       end case;
 42       State := Transition (Input_Action, State);
 43 
 44    end loop Input;
 45 
 46 end Vending;
