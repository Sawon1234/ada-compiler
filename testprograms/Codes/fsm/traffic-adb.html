  1 with Ada.Text_IO, Ada.Integer_Text_IO; use Ada;
  2 
  3 procedure Traffic is
  4 
  5    type Direction is (Eastbound, Westbound);
  6    type Lane is (Left_Turn, Straight);
  7    type Input_Type is (West_Turn, West, East_Turn, East, Time);
  8    package Input_IO is new Text_IO.Enumeration_IO (Enum=>Input_Type);
  9    Input_Action  : Input_Type;
 10 
 11    type State_Type is array (Direction, Lane) of Boolean;
 12    State : State_Type := (Direction=>(Lane=>False));
 13 
 14    procedure Transition (I: Input_Type; S: in out State_Type) is
 15    begin
 16       case I is
 17          when West_Turn => S(WestBound, Left_Turn) := True;
 18          when West      => S(WestBound, Straight) := True;
 19          when East_Turn => S(EastBound, Left_Turn) := True;
 20          when East      => S(EastBound, Straight) := True;
 21          When Time      => null;
 22       end case;
 23    end Transition;
 24 
 25    procedure Signal (S: in State_Type) is
 26       East: Boolean := S(EastBound, Left_Turn) or else S(EastBound, Straight);
 27       West: Boolean := S(WestBound, Left_Turn) or else S(WestBound, Straight);
 28    begin
 29       if (East and not West) then
 30          if (S(EastBound, Left_Turn)) then
 31             Text_IO.Put_Line ("Eastbound green with left turn light.");
 32          else
 33             Text_IO.Put_Line ("Eastbound green.");
 34          end if;
 35       elsif (West and not East) then
 36          if (S(WestBound, Left_Turn)) then
 37             Text_IO.Put_Line ("Westbound green with left turn light.");
 38          else
 39             Text_IO.Put_Line ("Westbound green.");
 40          end if;
 41       elsif (East and West) then
 42          if (S(EastBound, Left_Turn) and not S(WestBound, Left_Turn)) then
 43             Text_IO.Put_Line ("Eastbound green with left turn light.");
 44             Text_IO.Put_Line ("Eastbound left turn off.");
 45             Text_IO.Put_Line ("Now also westbound green.");
 46          elsif  (S(WestBound, Left_Turn) and not S(EastBound, Left_Turn)) then
 47             Text_IO.Put_Line ("Westbound green with left turn light.");
 48             Text_IO.Put_Line ("Westbound left turn off.");
 49             Text_IO.Put_Line ("Now also eastbound green.");
 50          elsif  (S(WestBound, Left_Turn) and S(EastBound, Left_Turn)) then
 51             Text_IO.Put_Line ("Both left turn signals.");
 52             Text_IO.Put_Line ("Green both east and west bound.");
 53          else
 54             Text_IO.Put_Line ("Green both east and west bound.");
 55          end if;
 56       else
 57          Text_IO.Put_Line ("Skip cycle.");
 58       end if;
 59    end Signal;
 60 
 61 begin
 62 
 63    Input: loop
 64 
 65       Text_IO.Put ("Enter an input action -- ");
 66       Input_IO.Get (Input_Action);
 67       exit when Input_Action = Time;
 68       Text_IO.Skip_Line;
 69       Transition (Input_Action, State);
 70 
 71    end loop Input;
 72 
 73    Signal (State);
 74 
 75 end Traffic;
