The Ada Program: chill.adb

  1 -- chill.adb:  Compute wind chill given wind speed and temerature
  2 
  3 with Ada.Text_Io, Ada.Integer_Text_Io;
  4 with Ada.Numerics.Elementary_Functions;
  5 use Ada, Ada.Numerics.Elementary_Functions;
  6 
  7 -- Adapted from Kamin and Reingold, "Programming with Class," 
  8 -- pages 52, 53.
  9 
 10 procedure Chill is
 11    Wind_Speed, Temperature, Wind_Chill : Integer;
 12    X, Y: Float;  -- temporary variables 
 13 begin
 14    Text_IO.Put ("Enter wind speed in mph => ");
 15    Integer_Text_IO.Get (Wind_Speed);
 16    Text_IO.Put ("Enter temperature in degrees Farenheit => ");
 17    Integer_Text_IO.Get (Temperature);
 18    if Wind_Speed <= 4 then
 19       Wind_Chill := Temperature;
 20    elsif Wind_Speed <= 45 then
 21       X := 10.45 + 6.69 * Sqrt (Float (Wind_Speed)) -
 22            0.447 * (Float (Wind_Speed));
 23       Y := (91.4 - Float (Temperature))/22.0;
 24       Wind_Chill := Integer (91.4 - X*Y);
 25    else
 26       Wind_Chill := Integer (1.6 * Float (Temperature) - 55.0);
 27    end if;
 28    Integer_Text_Io.Put (Wind_Chill);
 29 end Chill;
