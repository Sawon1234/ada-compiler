  1 with Priority_Queue;
  2 
  3 separate (Main_Path)
  4 
  5 procedure SFS (Graph: in out Air_Routes.Graph_Type;
  6                Start_City, End_City: in City_String) is
  7 
  8    type Flight is
  9       record
 10          Start, Stop:  City_String;
 11          Total: Natural := 0;
 12       end record;
 13 
 14    function Distance (F: Flight) return Natural is
 15    begin
 16       return F.Total;
 17    end Distance;
 18 
 19    package City_Queue is new Priority_Queue (Flight, Natural, Distance, "<");
 20 
 21    Stack: City_Queue.Queue_Type (100);
 22    F: Flight;
 23 begin
 24    Air_Routes.Clear_All_Marks (Graph);
 25    City_Queue.Clear (Stack);
 26    City_Queue.Enqueue (Stack, Flight'(Start_City, Start_City, 0));
 27    while not City_Queue.Empty (Stack) loop
 28       City_Queue.Dequeue (Stack, F);
 29       Text_IO.Put (F.Start);
 30       Text_IO.Put (F.Stop);
 31       Text_IO.Put (Natural'Image(F.Total));
 32       Text_IO.New_Line;
 33       if (F.Stop=End_City) then
 34          return;  -- RETURN!! Found it!!
 35       end if;
 36       Air_Routes.Mark_Vertex (Graph, F.Stop);
 37       declare
 38          Adjacent : Air_Routes.Vertex_List := Air_Routes.Successors (Graph, F.Stop);
 39       begin
 40          for I in Adjacent'Range loop
 41             if not Air_Routes.Marked (Graph, Adjacent(I)) then
 42                City_Queue.Enqueue (Stack, Flight'(F.Stop,Adjacent(I),F.Total+Air_Routes.Weight_Of (Graph, F.Stop,Adjacent(I))));
 43             end if;
 44          end loop;
 45       end;
 46    end loop;
 47 end SFS;
