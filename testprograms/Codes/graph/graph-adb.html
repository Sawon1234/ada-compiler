The Ada Program: graph.adb

  1 --  graph.adb:  Generic package for weighted digraphs
  2 
  3 package body Graph is
  4 
  5    procedure Clear (Graph : in out Graph_Type) is
  6    begin
  7       Graph.Number := 0;
  8    end Clear;
  9 
 10    function Index_Of (Graph : Graph_Type;
 11                       Vertex: Vertex_Type) return Positive is
 12    begin
 13       for I in Graph.Vertices'First..Graph.Number loop
 14          if Graph.Vertices(I).Info=Vertex then
 15             return I;
 16          end if;
 17       end loop;
 18       return Graph.Number+1;  --  if not there, return next index
 19    end Index_Of;
 20 
 21    procedure Add_Vertex (Graph  : in out Graph_Type;
 22                          Vertex : in     Vertex_Type) is
 23       Index: Positive := Index_Of (Graph, Vertex);
 24    begin
 25       if Index <= Graph.Number then
 26          raise Vertex_Error;   --  already there
 27       elsif Index > Graph.Max_Vertices then
 28          raise Overflow;       --  no space for any more
 29       else
 30          Graph.Vertices (Index) := (Vertex, False);
 31          Graph.Number := Index;
 32          --  no edges for this vertex (yet)
 33          for Destination in Graph.Vertices'Range loop
 34             Graph.Edges(Index,Destination).Defined := False;
 35          end loop;
 36       end if;
 37    end Add_Vertex;
 38 
 39    procedure Add_Edge (Graph: in out Graph_Type;
 40                        Head, Tail: Vertex_Type; Weight: Edge_Weight) is
 41       H: Positive := Index_Of (Graph, Head);
 42       T: Positive := Index_Of (Graph, Tail);
 43    begin
 44       if H>Graph.Number or T>Graph.Number then
 45          raise Edge_Error;
 46       else
 47          Graph.Edges (H,T) := (Weight, True);
 48       end if;
 49    end Add_Edge;
 50 
 51    function Weight_Of (Graph: Graph_Type;
 52                        Head, Tail: Vertex_Type) return Edge_Weight is
 53       H: Positive := Index_Of (Graph, Head);
 54       T: Positive := Index_Of (Graph, Tail);
 55    begin
 56       if H>Graph.Number or T>Graph.Number then
 57          raise Edge_Error;
 58       else
 59          return Graph.Edges(H,T).Weight;
 60       end if;
 61    end Weight_Of;
 62 
 63    function Successors (Graph: Graph_Type;
 64                         Vertex: Vertex_Type) return Vertex_List is
 65       Index: Positive := Index_Of (Graph, Vertex);
 66       List: Vertex_List (1..Graph.Number);  -- max possible outdegree
 67       N: Natural := 0;
 68    begin
 69       for Destination in Graph.Vertices'First..Graph.Number loop
 70          if Graph.Edges (Index, Destination).Defined then
 71             N := N+1;
 72             List(N) := Graph.Vertices(Destination).Info;
 73          end if;
 74       end loop;
 75       return List(1..N);
 76    end Successors;
 77 
 78    procedure Clear_All_Marks (Graph: in out Graph_Type) is
 79    begin
 80       for I in Graph.Vertices'First..Graph.Number loop
 81          Graph.Vertices(I).Marked := False;
 82       end loop;
 83    end Clear_All_Marks;
 84 
 85    function Marked (Graph: Graph_Type;
 86                     Vertex: Vertex_Type) return Boolean is
 87       Index: Positive := Index_Of (Graph, Vertex);
 88    begin
 89       if Index > Graph.Number then
 90          raise Vertex_Error;
 91       else
 92          return (Graph.Vertices(Index).Marked);
 93       end if;
 94    end Marked;
 95 
 96    procedure Mark_Vertex (Graph : in out Graph_Type;
 97                           Vertex: in     Vertex_Type) is
 98       Index: Positive := Index_Of (Graph, Vertex);
 99    begin
100       if Index > Graph.Number then
101          raise Vertex_Error;
102       else
103          Graph.Vertices(Index).Marked := True;
104       end if;
105    end Mark_Vertex;
106 
107 end Graph;
