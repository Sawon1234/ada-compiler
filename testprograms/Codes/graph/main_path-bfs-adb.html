  1 with Linked_Queue;
  2 
  3 separate (Main_Path)
  4 
  5 procedure BFS (Graph: in out Air_Routes.Graph_Type;
  6                Start_City, End_City: in City_String) is
  7 
  8    package City_Queue is new Linked_Queue (City_String);
  9    Stack: City_Queue.Queue_Type;
 10    City: City_String;
 11 
 12 begin
 13 
 14    Air_Routes.Clear_All_Marks (Graph);
 15    City_Queue.Clear (Stack);
 16    City_Queue.Enqueue (Stack, Start_City);
 17 
 18    while not City_Queue.Empty (Stack) loop
 19       City_Queue.Dequeue (Stack, City);
 20 
 21       Text_IO.Put (City);
 22       Text_IO.New_Line;
 23 
 24       if (City=End_City) then
 25          return;  -- RETURN!! Found it!!
 26       end if;
 27 
 28       Air_Routes.Mark_Vertex (Graph, City);
 29 
 30       declare
 31          Adjacent : Air_Routes.Vertex_List :=
 32             Air_Routes.Successors (Graph, City);
 33       begin
 34          for I in Adjacent'Range loop
 35             if not Air_Routes.Marked (Graph, Adjacent(I)) then
 36                City_Queue.Enqueue (Stack, Adjacent(I));
 37             end if;
 38          end loop;
 39       end;
 40    end loop;
 41    -- not found
 42 end BFS;
