 39 private
 40 
 41    type Vertex_Rec is
 42       record
 43          Info   : Vertex_Type;       -- Information declared by user
 44          Marked : Boolean := False;  -- Has the Vertex been visited?
 45       end record;
 46 
 47    type Edge_Record is
 48       record
 49          Weight  : Edge_Weight;       -- Weight assigned to the edge
 50          Defined : Boolean := False;  -- Is this edge in the graph?
 51       end record;
 52 
 53    type Vertex_Array is array (Positive range <>) of Vertex_Rec;
 54    type Matrix_Type  is array (Positive range <>,
 55                                Positive range <>) of Edge_Record;
 56 
 57    type Graph_Type (Max_Vertices : Positive) is
 58       record
 59          Number : Natural := 0;
 60          Vertices     : Vertex_Array (1..Max_Vertices);
 61          Edges        : Matrix_Type  (1..Max_Vertices, 1..Max_Vertices);
 62       end record;
 63 
 64 end Graph;
