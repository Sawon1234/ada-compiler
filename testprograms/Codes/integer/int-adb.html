The Ada Program: int.adb

  1 -- int.adb:  integer type, subtypes, and derived types
  2 
  3 with Ada.Integer_Text_IO; use Ada;
  4 
  5 procedure int is
  6 
  7    type T is range 1 .. 1_000;             -- an integer type
  8    type R is new Integer range 1 .. 1_000; -- derived type of integer
  9    subtype S is Integer range 1 .. 1_000;  -- a subtype of integer
 10 
 11    X: Integer;
 12    Y: T := 1;
 13    Z: R := 1;
 14    W: S := 1;
 15    U: Integer range 1 .. 1_000;
 16 
 17 begin
 18 
 19    -- X,W,U are Integer as far as compiler is concerned
 20    X := X+W;
 21    X := X+W+U;
 22    Integer_Text_IO.Put (W);
 23    Integer_Text_IO.Put (U);
 24 
 25    -- X, Y, Z, are mutually type-incompatible, i.e.,
 26    --    X:=Y;  X:=Z;  Y:= Z;  Integer_Text_IO.Put (Y);  Integer_Text_IO.(Z);
 27    -- are illegal.
 28 
 29    -- But explict conversion between them is possible.
 30    X := Integer (Y);
 31    Z := R (X);
 32    Y := T (Z);
 33 
 34    -- The types T and R have the same operations as the type Integer.
 35    -- *  Integer operations (having been defined on "root_integer")
 36    --    work on all integer types.
 37    -- *  Derived types inherit all the operations of the parent.  So
 38    --    no difference when type derived from Integer.
 39    Y := Y-Y/2;
 40    Z := Z+2*Z;
 41 
 42 end Int;
