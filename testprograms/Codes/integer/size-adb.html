The Ada Program: size.adb

  1 -- size.adb:  universal integer expressions and large integers
  2 
  3 with Ada.Text_IO, Ada.Long_Long_Integer_Text_IO, System;
  4 use Ada;
  5 
  6 procedure Size is
  7 
  8    type I1 is range              -32_768 ..              32_767;
  9    type I2 is range       -4_294_967_296 ..       4_294_967_295;
 10    type I3 is range      -10_000_000_000 ..      10_000_000_000;
 11    type I4 is range -9223372036854775808 .. 9223372036854775807;
 12    type I5 is range       System.Min_Int ..      System.Max_Int;
 13    -- any more than Min_Int..Max_Int is too big for an integer type range
 14 
 15    -- named numbers and universal integer expressions can be bigger
 16    Big1 : constant := (2**63 - 1) * 1000;
 17    Big2 : constant := 9_223_372_036_854_775_807_000;
 18    E    : Boolean  := Big1=Big2;               -- True
 19 
 20    X : Integer      :=                 100;
 21    Y : I1           :=              10_000;
 22    Z : I4           :=  10_000_000_000_000;
 23    Q : Long_Integer;     -- GNAT has "Long_Integer" = "Integer" (32 bits)
 24    R : Long_Long_Integer;-- "Long_Long" is 64 bits
 25 
 26 begin
 27 
 28    Long_Long_Integer_Text_IO.Put (Long_Long_Integer'Last);
 29    Long_Long_Integer_Text_IO.Put (Long_Long_Integer(Integer'Last));
 30    Long_Long_Integer_Text_IO.Put (System.Max_Int);  -- 2**63-1
 31    Long_Long_Integer_Text_IO.Put (System.Min_Int);  -- - (2**63)
 32 
 33    X := Integer (Y);
 34    Y := I1 (X);
 35 
 36    X := Integer (Long_Integer'Last);
 37 
 38    --  Raises "Constraint_Error"; GNAT won't compile it
 39    --  X := Integer (Long_Long_Integer'Last);
 40 
 41    Z := I4 (X);
 42 
 43 end Size;
