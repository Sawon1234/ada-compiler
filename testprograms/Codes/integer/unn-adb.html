The Ada Program: unn.adb

  1 -- unn.adb:  integer type, subtypes, dervied types, universal named numbers
  2 
  3 with Ada.Text_IO, Ada.Float_Text_IO, Ada.Integer_Text_IO, Ada.Numerics;
  4 use Ada;
  5 
  6 procedure Unn is
  7 
  8    type Byte_Type is range 0..256;
  9    subtype Byte_Sub_Type is Integer range 0..256;
 10    type Byte_Derived_Type is new Integer range 0..256;
 11 
 12    Two  : constant := 2 * Ada.Numerics.Pi;  -- universal named number
 13    Three: constant := 3;                    -- universal named number
 14 
 15    A: Integer;
 16    B: Byte_Type;
 17    C: Byte_Sub_Type;
 18    D: Byte_Derived_Type;
 19 
 20 begin
 21 
 22    Float_Text_IO.Put (Ada.Numerics.Pi);
 23    Float_Text_IO.Put (Ada.Numerics.e);
 24    Float_Text_IO.Put (Two);
 25 
 26    B := Three;  C := Three;  D := Three;
 27    B := 4; C := 4;  D := 4;
 28 
 29    -- Byte_Sub_Type is Integer as far as compiler is concerned
 30    A := A+C;
 31    Integer_Text_IO.Put (C);
 32 
 33    -- A, B, D, are mutually type-incompatible, i.e.,
 34    --
 35    -- A:=B; A:=D; B:=D;
 36    -- Integer_Text_IO.Put (B); Integer_Text_IO.Put (D);
 37    --
 38    -- are illegal.
 39 
 40    B := B+2*B;  -- Byte_Type and Byte_Derived_Type have same
 41    D := D-D/2;  -- operations as integer.
 42 
 43 end Unn;
