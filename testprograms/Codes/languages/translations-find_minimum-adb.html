The Ada Program: translations-find_minimum.adb

  1 -- translations-find_minimum.adb:  subunit to find small set of languages
  2 
  3 separate (Translations)
  4 
  5 -- This function uses a greedy algorithm to determine some "small" set
  6 -- of languages that satisfy the needs of all delegates in the List
  7 
  8 function Find_Minimum (List: Language_Set_Array)
  9    return Language_Set.Set_Type is
 10 
 11    -- Is the given set acceptable to all delegates?
 12    function Adequate (Set: Language_Set.Set_Type) return Boolean is
 13    begin
 14       for Del in List'Range loop
 15          if (List(Del) * Set = Language_Set.Empty_Set) then
 16             return (False);  -- The set inadequate for delegate "Del"
 17          end if;
 18       end loop;
 19       return (True);
 20    end Adequate;
 21 
 22    -- The set of all languages will certainly work, start with it
 23    Min_Set : Language_Set.Set_Type := Language_Set.Universal_Set;
 24 
 25 begin
 26 
 27    -- Try to remove the languages one by one from the set
 28    for Candidate in Language_Type'Range loop
 29       Min_Set := Min_Set - Candidate;  -- Remove language from the set
 30       -- Is the current set of languages inadequate?
 31       if not Adequate (Set=>Min_Set) then
 32          Min_Set := Min_Set + Candidate;  -- Put the language back in
 33       end if;
 34    end loop;
 35 
 36    return (Min_Set);
 37 
 38 end Find_Minimum;
