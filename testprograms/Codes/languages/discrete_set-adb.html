  3 package body Discrete_Set is
  4 
  5    -- This function determines whether Element is a member of Set
  6    function Is_Member (Element : Element_Type;
  7                        Set     : Set_Type) return Boolean is
  8    begin
  9       return Set(Element);
 10    end Is_Member;
 11 
 12    -- Returns the union of the two sets
 13    -- (elements that are in either or both sets)
 14    function "+" (Left, Right: Set_Type) return Set_Type is
 15    begin
 16       return (Left or Right);
 17    end "+";
 18 
 19    -- Adds an element to a set
 20    -- (the set is not changed if the element is already a member)
 21    function "+" (Left: Set_Type; Right: Element_Type) return Set_Type is
 22       Result: Set_Type := Left;
 23    begin
 24       Result(Right) := True;   -- Add the new element to the set
 25       return Result;
 26    end "+";
 27 
 28    -- Returns the intersection of the two sets
 29    -- (elements occurring in both sets)
 30    function "*" (Left, Right: Set_Type) return Set_Type is
 31    begin
 32       return (Left and Right);
 33    end "*";
 34 
 35    -- Returns the difference of the two sets
 36    -- (elements in Left and not in Right)
 37    function "-" (Left, Right: Set_Type) return Set_Type is
 38    begin
 39       return (Left and not Right);
 40    end "-";
 41 
 42    -- Removes an element from the set
 43    -- (the set is not changed if the element is not a member)
 44    function "-" (Left: Set_Type; Right: Element_Type) return Set_Type is
 45       Result: Set_Type := Left;
 46    begin
 47       Result(Right) := False;   -- Remove the element from the set
 48       return Result;
 49    end "-";
 50 
 51 
 52    -- Now subset and proper subset operations ...
 53 
 54    -- If the union of the sets Left and Right is equal to Right,
 55    -- then the set Left is a subset of Right.
 56    function "<=" (Left, Right: Set_Type) return Boolean is
 57    begin
 58       return ((Left+Right) = Right);
 59    end "<=";
 60 
 61    -- A proper subset is a subset but at least one element must
 62    -- be missing making the sets unequal
 63    function "<" (Left, Right: Set_Type) return Boolean is
 64    begin
 65       return ((Left <= Right) and then Left /= Right);
 66    end "<";
 67 
 68 
 69    -- The set Left is a superset of Right, if Right is a subset of Left
 70    function ">=" (Left, Right: Set_Type) return Boolean is
 71    begin
 72       return (Right <= Left);
 73    end ">=";
 74 
 75    -- Similarly for a proper superset
 76    function ">" (Left, Right: Set_Type) return Boolean is
 77    begin
 78       return (Right < Left);
 79    end ">";
 80 
 81 end Discrete_Set;
