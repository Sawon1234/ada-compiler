The Ada Program: translations.adb

  1 -- translations.adb:  use sets to determine minimum number of languages
  2 
  3 -- This program finds a "minimum" number of languages needed for an
  4 -- international conference.  Because a greedy algorithm is used,
  5 -- the languages selected may not be the minimum number overall.
  6 
  7 with Ada.Text_IO, Discrete_Set;
  8 use Ada;
  9 
 10 procedure Translations is
 11 
 12    Max_People : constant := 100;  -- Maximum number of people
 13 
 14    type Language_Type is (Dutch, English, Finnish, French, German,
 15       Chinese, Italian, Norwegian, Portuguese, Spanish, Swedish);
 16 
 17    -- Create a set class whose components are the languages
 18    package Language_Set is
 19       new Discrete_Set (Element_Type=>Language_Type);
 20    use type Language_Set.Set_Type;   -- Make set ops directly visisble
 21    -- A package for I/O of languages
 22    package Language_IO is new Text_IO.Enumeration_IO (Language_Type);
 23 
 24    -- Types for an array of language sets
 25    subtype Index_Range is Positive range 1..Max_People;
 26    type Language_Set_Array is
 27       array (Index_Range range <>) of Language_Set.Set_Type;
 28 
 29    function Find_Minimum (List: Language_Set_Array)
 30       return Language_Set.Set_Type is separate;
 31 
 32    procedure Get_Data (File_Name     : in String;
 33                        Delegate_List : out Language_Set_Array;
 34                        Delegate_Count: out Natural) is separate;
 35 
 36    -- A list of the languages spoken by each delegate.  Each element
 37    -- in the array is a set of languages spoken by one delegate.
 38    Delegate_List  : Language_Set_Array (Index_Range);
 39    Delegate_Count : Natural range 0..Max_People;  -- Actual number
 40 
 41    File_Name : String := "delegate.dat"; -- Filename of delegate info
 42    Min_Set   : Language_Set.Set_Type;    -- Set of sufficent languages
 43 
 44 begin  -- Program Translations
 45 
 46    -- Fill Delegate_List with the language data for all the delegates
 47    Get_Data (File_Name, Delegate_List, Delegate_Count);
 48 
 49    -- See if there are any languages spoken by ALL delegates by
 50    -- determining the intersection of all their language sets
 51    Min_Set := Delegate_List (1);  -- 1st delegate's languages
 52    for Count in 2..Delegate_Count loop
 53       Min_Set := Min_Set * Delegate_List (Count);
 54    end loop;
 55 
 56     -- Is there a set of common languages, use one of them.
 57     -- Else find some small set that satisfies all delegates.
 58    if Min_Set /= Language_Set.Empty_Set then
 59       Text_IO.Put_Line ("Use any one of the following languages.");
 60    else
 61       Text_IO.Put_Line ("Use all of the following languages.");
 62       Min_Set := Find_Minimum (Delegate_List (1..Delegate_Count));
 63    end if;
 64 
 65    -- Print each language in the set "Min_Set"
 66    for Language in Language_Type loop
 67       If Language_Set.Is_Member (Element=>Language, Set=>Min_Set) then
 68          Language_IO.Put (Language);
 69          Text_IO.New_Line;
 70       end if;
 71    end loop;
 72 
 73 exception
 74    when IO_EXCEPTIONS.NAME_ERROR =>
 75       Text_IO.Put_Line ("File " & File_Name & " not found.");
 76 end Translations;
