  1 separate (Translations)

  3 -- Get the language data for all the list of delegates
  4 
  5 procedure Get_Data (File_Name     : in String;
  6                     Delegate_List : out Language_Set_Array;
  7                     Delegate_Count: out Natural) is
  8 
  9    Data_File : Text_IO.File_Type;
 10 
 11    -- This procedure gets all the language data for one delegate.
 12    -- We assume that the languages spoken by this delegate are on a
 13    -- single line.  Nothing else can come after the last language.
 14    procedure Get_Delegate_Data (Languages: out Language_Set.Set_Type) is
 15       One_Language: Language_Type;  -- Variable to store input
 16    begin
 17 
 18       -- Initialize this delegate's set of languages
 19       Languages := Language_Set.Empty_Set;
 20 
 21       -- For each language on the line, add it to delegate's set
 22       while (not Text_IO.End_Of_Line (Data_File)) loop
 23          Language_IO.Get (File => Data_File, Item => One_Language);
 24          Languages := Languages + One_Language;
 25       end loop;
 26 
 27    end Get_Delegate_Data;
 28 
 29 begin
 30 
 31    Text_IO.Open (File=>Data_File,Name=>File_Name,Mode=>Text_IO.In_File);
 32 
 33    Delegate_Count := 0;
 34 
 35    -- For each line of the file, get the delegate's language set
 36    while (not Text_IO.End_Of_File (Data_File)) loop
 37       Delegate_Count := Delegate_Count + 1;
 38       Get_Delegate_Data (Languages=>Delegate_List(Delegate_Count));
 39       Text_IO.Skip_Line (Data_File);
 40    end loop;
 41 
 42    Text_IO.Close (Data_File);
 43 
 44 end Get_Data;
