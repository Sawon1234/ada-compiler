The Ada Program: quadratic.adb

  1 -- quadratic.adb:  given user inputs compute roots of quadratic equations
  2 
  3 with Ada.Text_IO, Ada.Float_Text_IO, Ada.Numerics.Elementary_Functions;
  4 use Ada, Ada.Numerics;
  5 
  6 procedure Quadratic is
  7 
  8     A,B,C: Float;
  9     Discriminant, Q: Float;
 10 
 11 begin
 12    Float_Text_IO.Get (A);
 13    Float_Text_IO.Get (B);
 14    Float_Text_IO.Get (C);
 15    Discriminant := (B**2) - 4.0*A*C;
 16 
 17    if (A = 0.0 and then B = 0.0) then
 18       -- neither quadratic nor linear
 19       null;
 20    elsif (A = 0.0) then
 21       Text_IO.Put_Line ("The linear equation: ");
 22       Text_IO.Put ("    ");
 23       Float_Text_IO.Put (B, 4, 2, 0);
 24       Text_IO.Put (" X + ");
 25       Float_Text_IO.Put (C, 4, 2, 0);
 26       Text_IO.Put_Line (" = 0.0");
 27       Text_IO.Put_Line ("has one solution:  ");
 28       Text_IO.Put ("    X = ");
 29       Float_Text_IO.Put (-C / B, 4, 2, 0);
 30       Text_IO.New_Line;
 31    else
 32       Text_IO.Put_Line ("The quadratic equation: ");
 33       Text_IO.Put ("    ");
 34       Float_Text_IO.Put (A, 4, 2, 0);
 35       Text_IO.Put (" X**2 + ");
 36       Float_Text_IO.Put (B, 4, 2, 0);
 37       Text_IO.Put (" X + ");
 38       Float_Text_IO.Put (C, 4, 2, 0);
 39       Text_IO.Put_Line (" = 0.0");
 40 
 41       if (Discriminant < 0.0) then
 42          Text_IO.Put_Line ("has no real roots.");
 43       elsif (Discriminant = 0.0) then
 44          Text_IO.Put_Line ("has one real root:  ");
 45          Text_IO.Put ("    X = ");
 46          Float_Text_IO.Put (-B / (2.0 * A), 4, 2, 0);
 47          Text_IO.New_Line;
 48       else
 49          Q := Elementary_Functions.Sqrt (Discriminant);
 50          Text_IO.Put_Line ("has two real roots:  ");
 51          Text_IO.Put ("    X = ");
 52          Float_Text_IO.Put ((-B + Q)/ (2.0 * A), 4, 2, 0);
 53          Text_IO.New_Line;
 54          Text_IO.Put ("    X = ");
 55          Float_Text_IO.Put ((-B - Q)/ (2.0 * A), 4, 2, 0);
 56          Text_IO.New_Line;
 57       end if;
 58    end if;
 59 
 60 end Quadratic;
