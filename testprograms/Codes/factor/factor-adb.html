  2 
  3 package body Factor is
  4 
  5    -- global, static, array (bigger than required, we hope)
  6    -- the largest number of factors for any number less than 2**31
  7    -- is 223,092,870 = 2*3*5*7*11*13*17*19*23.
  8    F: Factors_Array_Type (1..10);
  9 
 10 
 11    -- return an array of all the prime factors (>1) of input
 12 
 13    function Factorize (N: GT1) return Factors_Array is
 14       Number  : Positive := N;  -- local copy we can modify downto 1
 15       Count   : Positive := 1;  -- number of factors discovered so far
 16       Divisor : Positive := 2;
 17       Exponent: Natural;
 18    begin
 19       loop
 20 
 21          while (Number mod Divisor /= 0 and then Divisor*Divisor<=Number) loop
 22             -- no value <= "Divisor" remains as a factor of "Number"
 23             if (Divisor = 2) then
 24                Divisor := 3;
 25             else
 26                Divisor := Divisor+2;
 27             end if;
 28          end loop;
 29 
 30          if (Number mod Divisor /= 0) then
 31             Divisor := Number;
 32          end if;
 33 
 34          Exponent := 0;
 35          while (Number mod Divisor = 0) loop
 36             Number := Number / Divisor;
 37             Exponent := Exponent + 1;
 38          end loop;
 39 
 40          F(Count) := Factor_Type' (Divisor, Exponent);
 41 
 42          exit when Number = 1;
 43 
 44          -- if Number is > 1, then there is at least one more factor
 45          Count := Count + 1;
 46 
 47       end loop;
 48 
 49       return (new Factors_Array_Type' (F(1..Count)));
 50    end;
 51 
 52 end Factor;
