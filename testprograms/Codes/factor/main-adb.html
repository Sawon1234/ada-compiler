The Ada Program: main.adb

  1 -- main.adb:  compute some some numbers based on factorization of input
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO, Factor;
  4 use Ada;
  5 
  6 procedure Main is
  7 
  8    A : Factor.Factors_Array;
  9    N : Factor.GT1;
 10    Product, Sum : Positive;
 11    Möbius : Integer;
 12 
 13 begin
 14 
 15    Text_IO.Put ("Enter number to factor => ");
 16    Integer_Text_IO.Get (N);
 17    A := Factor.Factorize (N);
 18 
 19    Text_IO.Put_Line ("Prime factorization ...");
 20    for I in A'Range loop
 21       Integer_Text_IO.Put (I);
 22       Integer_Text_IO.Put (A(I).Prime);
 23       Integer_Text_IO.Put (A(I).Exponent);
 24       Text_IO.New_Line;
 25    end loop;
 26    Text_IO.New_Line;
 27 
 28    Text_IO.Put_Line ("Number of positive divisors");
 29    Product := 1;
 30    for I in A'Range loop
 31       Product := Product * (A(I).Exponent + 1);
 32    end loop;
 33    Integer_Text_IO.Put (Product);
 34    Text_IO.New_Line (2);
 35 
 36    Text_IO.Put_Line ("Möbius function");
 37    if A'Length mod 2 = 0 then
 38       Möbius := 1;
 39    else
 40       Möbius := -1;
 41    end if;
 42    for I in A'Range loop
 43       if A(I).Exponent > 1 then
 44          Möbius := 0;
 45          exit;
 46       end if;
 47    end loop;
 48    Integer_Text_IO.Put (Möbius);
 49    Text_IO.New_Line (2);
 50 
 51    Text_IO.Put_Line ("Sum of all positive divisors");
 52    Product := 1;
 53    for I in A'Range loop
 54       Sum := 1;
 55       for J in 1 .. A(I).Exponent loop
 56          Sum := Sum + A(I).Prime ** J;
 57       end loop;
 58       Product := Product * Sum;
 59    end loop;
 60    Integer_Text_IO.Put (Product);
 61    Text_IO.New_Line (2);
 62 
 63 
 64    Text_IO.Put_Line ("Euler totient function");
 65    Product := N;
 66    for I in A'Range loop
 67       Product := Product / A(I).Prime;
 68       Product := Product * (A(I).Prime-1);
 69    end loop;
 70    Integer_Text_IO.Put (Product);
 71    Text_IO.New_Line (2);
 72 
 73 end Main;
