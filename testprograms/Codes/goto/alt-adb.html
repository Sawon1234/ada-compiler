The Ada Program: alt.adb

  1 -- alt.adb:  a version of "main.adb" without gotos
  2 
  3 with Ada.Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Alt is
  7 
  8    A: array (1..10,1..20) of Integer;
  9    Some_Nonzero :  Boolean;
 10 
 11 begin
 12 
 13    Rows:
 14    for Row in A'Range(1) loop
 15 
 16       Some_Nonzero := False;
 17 
 18       Zeros:
 19       for Col in A'Range(2) loop
 20          if A(Row,Col)/=0 then Some_Nonzero:=True; exit Zeros; end if;
 21       end loop Zeros;
 22 
 23       if not Some_Nonzero then
 24          Text_IO.Put ("First all-zero row is ");
 25          Integer_Text_IO.Put (Row);
 26          Text_IO.New_Line;
 27          exit Rows;
 28       end if;
 29 
 30    end loop Rows;
 31 
 32 end Alt;
