  1 separate (Machine_Inventory)
  2 
  3 -- This procedure queries for and obtains the information for a new
  4 -- machine.  The information is written to the inventory file
  5 
  6 procedure New_Machine (Inventory_File: in out Machine_IO.File_Type) is
  7 
  8    Machine : Machine_Rec;    -- The information on the new machine
  9    Last    : Natural;        -- String index
 10    Index   : Natural;        -- File component number
 11 
 12 begin
 13    -- Machine ID
 14    Text_IO.Put_Line ("Enter the ID Number of the new machine.");
 15    Integer_Text_IO.Get (Machine.ID_Number);
 16    Text_IO.Skip_Line;  -- Move reading marker to next line
 17 
 18    -- Machine Description
 19    Text_IO.Put_Line ("Enter a description of the new machine.");
 20    Text_IO.Get_Line (Item => Machine.Description,
 21                      Last => Last);
 22    -- Pad description with blanks
 23    Last := Last + 1;  -- Start padding after last character
 24    Blank_Loop:   -- Each iteration, one blank is assigned to the string
 25    loop
 26       exit Blank_Loop when Last > Description_Length;
 27       Machine.Description (Last) := ' ';
 28       Last := Last + 1;
 29    end loop Blank_Loop;
 30 
 31    -- Machine Cost
 32    Text_IO.Put_Line ("Enter the cost of the new machine.");
 33    Float_Text_IO.Get (Machine.Cost);
 34 
 35    -- Initialize the remaining fields in the machine record
 36 
 37    -- Get today's date and put it into the purchase date field
 38    Get_Today (Today => Machine.Purchase_Date);
 39    -- Assign the same date to the last serviced record
 40    Machine.History.Last_Serviced := Machine.Purchase_Date;
 41    -- Assign zero to the failure fields
 42    Machine.History.Fail_Rate := 0.0;
 43    Machine.History.Down_Days := 0;
 44 
 45    -- Write the record to the next spot in the inventory file
 46    Index := Natural (Machine_IO.Size (Inventory_File)) + 1;
 47    Machine_IO.Write (File => Inventory_File,
 48                      Item => Machine,
 49                      To   => Machine_IO.Count (Index));
 50 end New_Machine;
