The Ada Program: write_cd.adb

  1 -- write_cd.adb:  write records to binary file
  2 
  3 with Ada, Text_IO, Ada.Integer_Text_IO, Ada.Sequential_IO;
  4 use Ada;
  5 
  6 procedure Write_CD is
  7 
  8    subtype Year_Range is Integer range 1900..2990;
  9    subtype Track_Range is Integer range 1..30;
 10    type Disc is record
 11       Title, Artist, Album: String(1..20);
 12       Year: Year_Range;
 13       Track: Track_Range;
 14    end record;
 15 
 16    package CD_Bin_IO is new Sequential_IO (Disc);
 17    File      : CD_Bin_IO.File_Type;
 18    File_Name : constant String := "disk.data";
 19 
 20    Song_1, Song_2: Disc;
 21 
 22 begin
 23 
 24    CD_Bin_IO.Create (File=>File, Name=>File_Name);
 25    Song_1 := Disc'(
 26       Album => "Freewheelin'        ",
 27       Artist=> "Bob Dylan           ",
 28       Year  => 1963,
 29       Title => "Blowin' in the Wind ",
 30       Track => 1);
 31 
 32    Song_2 := Disc'(
 33       Album => "Highway 61 Revisited",
 34       Artist=> "Bob Dylan           ",
 35       Year  => 1965,
 36       Title => "Like a Rolling Stone",
 37       Track => 1);
 38 
 39    CD_Bin_IO.Write (File=>File, Item=>Song_1);
 40    CD_Bin_IO.Write (File=>File, Item=>Song_2);
 41    CD_Bin_IO.Close (File=>File);
 42 
 43 end Write_CD;
