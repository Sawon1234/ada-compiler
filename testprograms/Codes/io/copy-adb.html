The Ada Program: copy.adb

  1 --  copy.adb:  copy standard input to output numbering lines
  2 
  3 with Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Copy is
  7 
  8    Line   : String (1 .. 100);
  9    Length : Integer;
 10    Num    : Integer := 0;
 11 
 12 begin
 13 
 14    while not (Text_IO.End_Of_File) loop
 15       Text_IO.Get_Line (Line, Length);
 16       Num := Num + 1;
 17       Integer_Text_IO.Put (Item => Num, Width => 4);
 18       Text_IO.Put (Item => "  " & Line (1 .. Length));
 19       Text_IO.New_Line;
 20    end loop;
 21 
 22 end Copy;
