  7    package Kind_Random is new Numerics.Discrete_Random (Rec_Package.Kind);
  8    Kind_Seed: Kind_Random.Generator;
  9    subtype Some_Int is Integer range -100 .. 100;
 10    package Int_Random is new Numerics.Discrete_Random (Some_Int);
 11    Int_Seed: Int_Random.Generator;
 12 
 13    Float_Seed: Float_Random.Generator;
 14    function Random_Float return Float is
 15    begin
 16       return 200.0*Float_Random.Random (Float_Seed)-100.00;
 17    end Random_Float;
 18 
 19    subtype Some_Char is Character range '!' .. '~';
 20    package Char_Random is new Numerics.Discrete_Random (Some_Char);
 21    Char_Seed: Char_Random.Generator;
 22 
 23    function Random_Element return Rec_Package.Mixed is
 24    begin
 25       case Kind_Random.Random (Kind_Seed) is
 26          when Rec_Package.Int  =>
 27             return (Rec_Package.To_Mixed (Int_Random.Random (Int_Seed)));
 28          when Rec_Package.Real =>
 29             return (Rec_Package.To_Mixed (Random_Float));
 30          when Rec_Package.Char =>
 31             return (Rec_Package.To_Mixed (Char_Random.Random (Char_Seed)));
 32       end case;
 33    end Random_Element;
 34 
 35 begin
 36    -- Initialize each seed to a different, randomly chosen, value
 37    -- each time the program is run.  Without this, the program will
 38    -- start with the same sequence of random numbers every time.
 39    Kind_Random.Reset (Kind_Seed);
 40    Int_Random.Reset (Int_Seed);
 41    Float_Random.Reset (Float_Seed);
 42    Char_Random.Reset (Char_Seed);
 43 end Random_Rec;
