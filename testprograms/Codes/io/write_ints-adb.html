The Ada Program: write_ints.adb

  1 -- write_ints.adb:  write integers to binary file
  2 
  3 with Ada, Text_IO, Ada.Integer_Text_IO, Ada.Sequential_IO;
  4 use Ada;
  5 
  6 procedure Write_Ints is
  7 
  8    package Integer_Bin_IO is new Sequential_IO (Integer);
  9    File      : Integer_Bin_IO.File_Type;
 10    File_Name : constant String := "integers.data";
 11    Num       : Integer;
 12 
 13 begin
 14 
 15    Integer_Bin_IO.Create (File=>File, Name=>File_Name);
 16    while not (Text_IO.End_Of_File) loop
 17       Integer_Text_IO.Get (Item=>Num);
 18       Integer_Bin_IO.Write (File=>File, Item=>Num);
 19    end loop;
 20    Integer_Bin_IO.Close (File=>File);
 21 
 22 end Write_Ints;
