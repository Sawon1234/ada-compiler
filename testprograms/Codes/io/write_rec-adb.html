  3 with Ada, Ada.Sequential_IO;
  4 with Rec_Package, Random_Rec;
  5 use Ada;
  6 
  7 procedure Write_Rec is
  8 
  9    package Bin_IO is new Sequential_IO (Rec_Package.Mixed);
 10    File      : Bin_IO.File_Type;
 11    File_Name : constant String := "mixed.data";
 12 
 13    subtype Domain is Integer range 1..30;
 14    Data : array (Domain) of Rec_Package.Mixed;
 15 
 16 begin
 17 
 18    for I in Domain loop
 19       Data (I) := Random_Rec.Random_Element;
 20    end loop;
 21 
 22    Bin_IO.Create (File=>File, Name=>File_Name);

 24       Bin_IO.Write (File=>File, Item=>Data(I));
 25    end loop;
 26    Bin_IO.Close (File=>File);
 27 
 28 end Write_Rec;
