  1 with Ada.Text_IO, Ada.Integer_Text_IO, Ada.Float_Text_IO;
  2 use Ada;
  3 
  4 package body Rec_Package is
  5 
  6    function To_Mixed (X: Integer) return Mixed is
  7    begin
  8       return (Mixed'(Tag=>Int, Int_Part=>X));
  9    end To_Mixed;
 10 
 11    function To_Mixed (X: Float) return Mixed is
 12    begin
 13       return (Mixed'(Tag=>Real, Real_Part=>X));
 14    end To_Mixed;
 15 
 16    function To_Mixed (X: Character) return Mixed is
 17    begin
 18       return (Mixed'(Tag=>Char, Char_Part=>X));
 19    end To_Mixed;
 20 
 21    procedure Put_Line (Item: Mixed) is
 22    begin
 23       case Item.Tag is
 24          when Int  => Integer_Text_IO.Put(Item=>Item.Int_Part,Width=>0);
 25          when Real => Float_Text_IO.Put (Item=>Item.Real_Part, Exp=>0);
 26          when Char => Text_IO.Put (Item=>Item.Char_Part);
 27       end case;
 28       Text_IO.New_Line;
 29    end Put_Line;
 30 
 31 end Rec_Package;
