The Ada Program: palindrome.adb

  1 -- palindrome.adb:  check standard input for palidroms
  2 
  3 with Text_IO, Ada.Integer_Text_IO;
  4 use Ada;
  5 
  6 procedure Palindrome is
  7 
  8    function Check (Line: String) return Boolean is
  9    begin
 10      for I in 0 .. Line'Length/2 - 1 loop
 11         if (Line(Line'First+I)/=Line(Line'Last-I)) then
 12            return False;
 13         end if;
 14      end loop;
 15      return True;
 16    end Check;
 17 
 18    Line: String (1..100);
 19    Length: Integer;
 20 
 21 begin
 22 
 23    while not (Text_IO.End_Of_File) loop
 24       Text_IO.Get_Line (Line, Length);
 25       if Check (Line (1..Length)) then
 26          Text_IO.Put ("Palindrome");
 27          Text_IO.New_Line;
 28       end if;
 29       if Check (Line (1..Length-1)) then
 30          Text_IO.Put ("R-palindrome");
 31          Text_IO.New_Line;
 32       end if;
 33       if Check (Line (2..Length)) then
 34          Text_IO.Put ("L-palindrome");
 35          Text_IO.New_Line;
 36       end if;
 37       Text_IO.New_Line;
 38    end loop;
 39 
 40 end Palindrome;
