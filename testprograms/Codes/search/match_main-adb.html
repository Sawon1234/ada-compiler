  1 with
  2   Ada.Command_Line,     -- Access to external execution env (Ada95 A.15)
  3   Ada.Text_IO;          -- Usual string oriented IO package
  4 use Ada;

  6 procedure Match_Main is
  7 
  8    function Match (Pattern, Text: String) return Boolean is
  9       N : constant Natural := Pattern'Length;
 10    begin
 11       for I in Text'First .. Text'Last-N+1 loop
 12          if Text(I..I+N-1) = Pattern then
 13             return (True);
 14          end if;
 15       end loop;
 16       return (False);
 17    end Match;
 18 
 19    M : constant Boolean := Match (
 20       Command_Line.Argument (1), Command_Line.Argument (2));
 21 begin
 22 
 23    if M then
 24       Text_IO.Put_Line ("Match.");
 25    else
 26       Text_IO.Put_Line ("No match.");
 27    end if;
 28 
 29 end Match_Main;
