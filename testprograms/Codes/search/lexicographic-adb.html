  1 -- lexicographic.adb:  is the string "Left" < "Right"?

  3 function Lexicographic (Left, Right: String) return Boolean is
  4    I: Integer := Left'First;
  5    J: Integer := Right'First;
  6 begin
  7    loop
  8       if (I>Left'Last) then
  9          -- Left'Length <= Right'Length;
 10          return (Left'Length < Right'Length);
 11       elsif (J>Right'Last) then
 12          -- Left'Length > Right'Length);
 13          return (False);
 14       elsif (Left(I)/=Right(J)) then
 15          return (Left(I)<Right(J));
 16       else
 17          -- Left (Left'First..I) = Right (Right'First..J)
 18          I := I + 1;
 19          J := J + 1;
 20       end if;
 21    end loop;
 22 end Lexicographic;
