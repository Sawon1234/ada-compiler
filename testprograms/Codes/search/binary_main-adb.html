  1 -- binary_main.adb:  binary search of array
  2 with Ada.Text_IO, Ada.Integer_Text_IO;
  3 use Ada;
  4 procedure Binary_Main is
  5    type Item_Array_Type is array (Integer range <>) of Integer;
  6    -- search for "Item" in non-decreasing "List"
  7    procedure Binary_Search (List: in Item_Array_Type;
  8                             Item: in Integer;
  9                             Index: out Integer;
 10                             Found: out Boolean) is
 11       First_Index: Integer := List'First;
 12       Last_Index : Integer := List'Last;
 13       Midpoint   : Integer;
 14    begin
 15       Found := False;
 16       Index := 0;    -- meaningless
 17       while (First_Index <= Last_Index) and not Found loop
 18          Midpoint := (First_Index + Last_Index) / 2;
 19          if List(Midpoint) = Item then
 20             Found := True;
 21             Index := Midpoint;
 22          elsif Item < List(Midpoint) then
 23             Last_Index := Midpoint - 1;
 24          else
 25             First_Index := Midpoint + 1;
 26          end if;
 27       end loop;
 28    end Binary_Search;
 29    A: Item_Array_Type (-3..11) :=
 30       (-3=>1, -2=>2, -1=>3, 0=>5, 1=>23, 2=>77, others=>416);
 31    Found : Boolean;
 32    Index : Integer;
 33    Item  : Integer := 77;
 34 begin
 35    Binary_Search (A, Item, Index, Found);
 36    if Found then
 37       Text_IO.Put ("The item ");
 38       Integer_Text_IO.Put (Item=>Item, Width=>2);
 39       Text_IO.Put (" was found at index ");
 40       Integer_Text_IO.Put (Item=>Index, Width=>2);
 41       Text_IO.Put_Line (".");
 42    else
 43       Text_IO.Put ("The item");
 44       Integer_Text_IO.Put (Item=>Item, Width=>2);
 45       Text_IO.Put_Line ("was not found.");
 46    end if;
 47 end Binary_Main;
