  1 -- main-insertion_sort.adb:  subunit for insertion sort
  2 
  3 separate (Main)
  4 
  5 -- This subunit shares the type "Array_Type" with the parent unit
  6 -- "Main".  The elements of the array are of type "Element_Type"
  7 -- and we compare them using "<".
  8 
  9 -- Sort array "List" in non-decreasing order using insertion sort.
 10 
 11 procedure Insertion_Sort (List : in out Array_Type) is
 12 
 13 begin
 14 
 15    for I in List'First+1 .. List'Last loop
 16 
 17       -- List (List'First .. I-1) is sorted.
 18       -- Insert List(I) into its proper place.
 19 
 20       declare
 21          Temp : Element_Type := List (I);
 22          J : Integer := I-1;
 23       begin
 24          -- The proper place for List(I) is an index J such that
 25          --    List(List'First..J) <= List(I) < List(J+1..I-1)
 26          -- Find J and move elements List(J+1..I-1) to List(J+2..I).
 27          while J >= List'First and then List(J) > Temp loop
 28            List (J+1) := List (J);
 29            J := J-1;
 30          end loop;
 31          List (J+1) := Temp;
 32       end;
 33 
 34    end loop;
 35 
 36 end Insertion_Sort;
