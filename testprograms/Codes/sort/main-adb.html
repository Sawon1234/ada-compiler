  1 with Ada.Text_IO, Ada.Integer_Text_IO;
  2 use Ada;

  4 procedure Main is
  5 
  6    -- Sort in non-decreasing order.  The arrays can be of any element type
  7    -- as long as we can compare elements using "<".
  8 
  9    subtype Element_Type is Integer;    -- Just another name for Integer
 10    type Array_Type is array (Positive range <>) of Element_Type;
 11 
 12    procedure Selection_Sort (List : in out Array_Type) is separate;
 13    -- GNAT looks for subunit in file main-selection_sort.adb
 14 
 15    procedure Insertion_Sort (List : in out Array_Type) is separate;
 16    -- GNAT looks for subunit in file main-insertion_sort.adb
 17 
 18    A : Array_Type := (3, 6, 1, 9, 2, 4, 8, 5, 7);
 19    B : Array_Type := (3, 5, 9, 5, 7);
 20 
 21 begin
 22 
 23    -- Sort array "A" using selection sort and print result
 24    Selection_Sort (A);
 25    for Index in A'Range loop
 26       Integer_Text_IO.Put (A(Index), 5);
 27    end loop;
 28    Text_IO.New_Line;
 29 
 30 
 31    -- Sort array "B" using insertion sort and print result
 32    Insertion_Sort (B);
 33    for Index in B'Range loop
 34       Integer_Text_IO.Put (B(Index), 5);
 35    end loop;
 36    Text_IO.New_Line;
 37 
 38 end Main;
