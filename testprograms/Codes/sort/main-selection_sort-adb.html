  1 -- main-selection_sort.adb:  subunit for selection sort
  2 
  3 separate (Main)

  5 -- This subunit shares the type "Array_Type" with the parent unit
  6 -- "Main".  The elements of the array are of type "Element_Type"
  7 -- and we compare them using "<".
  8 
  9 -- Sort array "List" in non-decreasing order using selection sort.
 10 
 11 procedure Selection_Sort (List : in out Array_Type) is
 12 
 13 begin
 14 
 15    for I in List'First .. List'Last-1 loop
 16 
 17       -- List (List'First..I-1) are the smallest elements
 18       -- of List in order
 19 
 20       declare
 21          Temp : Element_Type := List (I);
 22          Min_Index : Integer := I;
 23       begin
 24 
 25          -- Find index of next smallest element
 26          for J in I+1 .. List'Last loop
 27            if List(J)<List(Min_Index) then
 28               Min_Index := J;
 29            end if;
 30          end loop;
 31 
 32          -- Exchange List (Min_Index) and List (I)
 33          List (I)         := List (Min_Index);
 34          List (Min_Index) := Temp;
 35       end;
 36 
 37    end loop;
 38 
 39 end Selection_Sort;
