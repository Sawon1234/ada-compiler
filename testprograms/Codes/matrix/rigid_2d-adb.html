The Ada Program: rigid_2d.adb

  1 with Ada.Numerics.Elementary_Functions;
  2 use  Ada.Numerics.Elementary_Functions;
  3 
  4 with Ada.Text_IO, Ada.Float_Text_IO;
  5 use Ada;
  6 
  7 package body Rigid_2D is
  8 
  9    --                        |  1  0  DX  |
 10    --  Translation (DX,DY) = |  0  1  DY  |
 11    --                        |  0  0   1  |
 12    function Translation (DX, DY: Float) return Transformation is
 13    begin
 14       return (Transformation'((1.0,0.0,DX),(0.0,1.0,DY),(0.0,0.0,1.0)));
 15    end Translation;
 16 
 17    --                    | SX  0  0  |
 18    --    Scale (DX,DY) = |  0 SY  0  |
 19    --                    |  0  0  1  |
 20    function Scaling (SX, SY: Float) return Transformation is
 21    begin
 22       return (Transformation'((SX,0.0,0.0),(0.0,SY,0.0),(0.0,0.0,1.0)));
 23    end Scaling;
 24 
 25    --                 | cos(Th)  -sin(Th)  0  |
 26    --   Rotate (Th) = | sin(Th)   cos(Th)  0  |
 27    --                 |   0         0      1  |
 28    function Rotation (Theta: Float) return Transformation is
 29    begin
 30       return (Transformation'(
 31          (Cos(Theta,360.0),-Sin(Theta,360.0),0.0),
 32          (Sin(Theta,360.0), Cos(Theta,360.0),0.0),
 33          (  0.0,              0.0,           1.0)));
 34    end Rotation;
 35 
 36    -- Convert point to homogeneous coordinate system
 37    function Vector (P: Point) return Matrix is
 38    begin
 39       return (Matrix'((1=>P.X), (1=>P.Y), (1=>1.0)));
 40    end;
 41 
 42    procedure Put (P: Point) is
 43    begin
 44       Text_IO.Put ("(");
 45       Float_Text_IO.Put (P.X, Aft=>2, Exp=>0);  Text_IO.Put (",");
 46       Float_Text_IO.Put (P.Y, Aft=>2, Exp=>0);  Text_IO.Put (")");
 47    end Put;
 48 
 49    -- Matrix multiplication
 50    function Multiply (A,B: in Matrix) return Matrix is
 51       C: Matrix (A'Range(1), B'Range(2));
 52    begin
 53       pragma Assert (A'First(2)=B'First(1) and A'Last(2)=B'Last(1));
 54       for Row in A'Range loop
 55          for Col in B'Range(2) loop
 56             declare
 57                Sum: Float := 0.0;
 58             begin
 59                for I in A'Range(2) loop
 60                   Sum := Sum + A(Row,I)*B(I,Col);
 61                end loop;
 62                C(Row,Col) := Sum;
 63             end;
 64          end loop;
 65       end loop;
 66       return (C);
 67    end Multiply;
 68 
 69    -- Transformation composition is just matrix multiplication
 70    function "*" (Second, First: Transformation) return Transformation is
 71    begin
 72       return (Multiply (Second, First));
 73    end "*";
 74 
 75    -- Rotation about a point is a composition of transformations
 76    function Rotation (Theta:Float; About:Point) return Transformation is
 77    begin
 78       return (Translation(About.X, About.Y)*Rotation(Theta)*
 79               Translation(-About.X,-About.Y));
 80    end Rotation;
 81 
 82    -- To transform, just do matrix multiplication
 83    function Transform (X: Point; By: Transformation) return Point is
 84       A: Matrix := Multiply (By, Vector (X));
 85    begin
 86       pragma Assert (abs(A(3,1)-1.0)<0.001);  -- A(3,1)=1.0, nearly
 87       return (Point'(X=>A(1,1), Y=>A(2,1)));
 88    end Transform;
 89 
 90 end Rigid_2D;
