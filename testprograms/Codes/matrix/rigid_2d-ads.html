The Ada Program: rigid_2d.ads

  1 -- rigid_2d.ads :  rigid 2D transformation using homogeneous coordinates
  2 
  3 package Rigid_2D is
  4 
  5    -- If points are expressed in homogeneous coordinates, then
  6    -- translation can be expressed with matrix multiplication like
  7    -- rotation and scaling.  Thus unifiying the expression of all
  8    -- common transformation.
  9    -- See Foley, van Dam, et al., "Computer Graphics".
 10 
 11    type Matrix is
 12        array (Positive range <>, Positive range <>) of Float;
 13 
 14    type Point is record
 15       X, Y: Float;
 16    end record;
 17 
 18    subtype Transformation is Matrix (1..3, 1..3);
 19 
 20    -- construct affine transformations
 21    function Translation (DX, DY: Float) return Transformation;
 22    function Scaling (SX, SY: Float) return Transformation;
 23    function Rotation (Theta: Float) return Transformation;
 24    function Rotation (Theta: Float; About: Point) return Transformation;
 25 
 26    -- the composition of transformation
 27    function "*" (Second, First: Transformation) return Transformation;
 28 
 29    -- transform 2D point using a transformation
 30    function Transform (X: Point; By: Transformation) return Point;
 31 
 32    -- print a point
 33    procedure Put (P: Point);
 34 
 35 end Rigid_2D;
